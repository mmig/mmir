 28 var silenceCount = 0, //how many silent blobs have there been in a row now?
 29 	speechCount = 0, //how many blobs have been loud in a row now?
 30 	lastInput = 0, // how long has there been no good loud input?
 31 	recording= false,
 32 	noiseTreshold = 0.1, //the bigger, the more is counted as silent
 33 	sampleRate = 0,
 34 	pauseCount = 3,
 35 	resetCount = 15,
 36 	maxBlobSize = 15,
 37 	blobSizeCount = 0,
 38 	blobNumber = 0;
 39   
 40 self.onmessage = function(e){
 41   switch(e.data.command){
 42     case 'init':
 43       init(e.data.config);
 44       break;
 45     case 'start':
 46       start();
 47       break;
 48     case 'isSilent':
 49       isSilent(e.data.buffer);
 50       break;
 51     case 'stop':
 52       stop();
 53       break;
 54   }
 55 };
 56 /**
 57  * sets the config and echos back
 58  * @param config
 59  */
 60 function init(config){
 61   if (config.sampleRate)sampleRate = config.sampleRate;
 62   if (config.noiseTreshold) noiseTreshold = config.noiseTreshold;
 63   if (config.pauseCount) pauseCount = config.pauseCount;
 64   if (config.resetCount) resetCount = config.resetCount;
 65   self.postMessage('Silence Detection initialized');
 66 }
 67 
 68 /**
 69  * recieves an audioBlob and decides whether or not there has been a real input (min. 3 loud blobs in a row)
 70  * and a real pause (min. <pauseCount> silent blobs in a row) afterwards. In this case it dictates a pause. 
 71  * If some time has gone by without any real input, it sends a signal to clear the buffers.
 72  * @param inputBuffer
 73  */
 74 function isSilent(inputBuffer){
 75 	if (recording){
 76 		blobNumber++;
 77 		if (blobNumber==3){
 78 			self.postMessage('Silence detected!');
 79 		}
 80 		var thisSilent = true;
 81 		var bound = 0;
 82 		for (var i = 0; i < inputBuffer.length; i++) {
 83 			if (( inputBuffer[i]> noiseTreshold)||( inputBuffer[i]<0-noiseTreshold)){
 84 				if (inputBuffer[i]>bound) bound= inputBuffer[i];
 85 				thisSilent = false;
 86 			}
 87 		}
 88 		if (thisSilent){
 89 			if (silenceCount>=pauseCount){
 90 				self.postMessage('Silence detected!');
 91 				speechCount = 0;
 92 				silenceCount = 0;
 93 				lastInput = 0;
 94 				blobsizeCount = 0;
 95 			}
 96 			if (speechCount>=pauseCount){
 97 				blobSizeCount++;
 98 				silenceCount++;
 99 			} 
100 			else {
101 				speechCount = 0;
102 				lastInput++;
103 			}
104 		} 
105 		else {
106 			if (speechCount>=pauseCount){
107 				silenceCount=0;
108 				blobSizeCount++;
109 			} 
110 			else {
111 				speechCount++;
112 				lastInput++;
113 			}
114 		}
115 		if (speechCount>pauseCount){
116 			
117 		}
118 		if (blobSizeCount >= maxBlobSize){
119 			self.postMessage('Send partial!');
120 			blobSizeCount = 0;
121 		}
122 		if (speechCount==0 && lastInput>resetCount){
123 			this.postMessage('clear');
124 			lastInput= 0;
125 		}
126 		
127 	}
128 }
129 
130 /**
131  * resets everything and switches the worker to recording mode.
132  */
133 function start(){
134 	silenceCount=0;
135 	speechCount =0;
136 	lastInput = 0;
137 	recording = true;
138 	self.postMessage('Silence Detection started');
139 	blobNumber = 0;
140 }
141 function stop(){
142 	recording = false;
143 	if (speechCount>0){
144 		self.postMessage('Silence detected!');
145 		speechCount = 0;
146 		silenceCount = 0;
147 		lastInput = 0;
148 		blobsizeCount = 0;
149 	}
150 	self.postMessage('Silence Detection stopped');
151 }
