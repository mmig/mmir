 91 GrammarConverter.prototype.loadGrammar = function(successCallback, errorCallback, grammarUrl, doLoadSynchronously){
 92 	var self = this;
 93 	var success = function(data, status, xhr){
 94 		
 95 		//DISABLED: old-style masking for umlauts:
 96 //		data = self.recodeJSON(data, self.encodeUmlauts);
 97 		
 98 		//if auto-upgrading is enabled:
 99 		//   decode old-style umlaut masking before continuing
100 		if(self.convertOldFormat){
101 			data = self.recodeJSON(data, self.decodeUmlauts);
102 		}
103 		
104 		self.json_grammar_definition = data;
105 		
106 		if (typeof successCallback == "function") {
107 			successCallback.call(this, self, xhr);
108 		}
109 	};
110 	var error = function(xhr, status, data){
111 		alert("failed to load the grammar! error: "+ JSON.stringify(data));
112 		if (typeof errorCallback == "function") {
113 			errorCallback.call(this, self);
114 		}
115 	};
116 	this.loadResource(success, error, grammarUrl, doLoadSynchronously);
117 };

167 /**
168  * HELPER creates a copy of the stopword list and encodes all non-ASCII chars to their unicode
169  *        representation (e.g. for save storage of stringified stopword list, even if file-encoding
170  *        does not support non-ASCII letters).
171  * 
172  * @returns {Array<String>} a copy of the stopword list, from the current JSON grammar
173  * 							(or empty list, if no grammar is present)
174  */
175 GrammarConverter.prototype.getEncodedStopwords = function(){
176 	var list = this.getStopWords();
177 	if(!list){
178 		return [];
179 	}
180 	
181 	//use copy, since recoding works in-place (we do not want to modify the stored stopword list here)
182 	list = list.slice(0, list.length);
183 	
184 	//store stopwords with their Unicode representation (only for non-ASCII chars)
185 	return this.recodeJSON(
186 			list, this.maskAsUnicode
187 	);
188 };

190 //this is the original / main implementation for creating the RegExp for stopword removal
191 GrammarConverter.prototype.parseStopWords = function(){
192 
193 	//create RegExp for stop words:
194 	var json_stop_words = this.json_grammar_definition.stop_word;
195 	var size = json_stop_words.length;
196 	var stop_words = "";
197 	
198 	//FIX for encoded chars: if a word begins or ends with an encoded char, \b cannot detect the word's boundaries
199 	//	-> FIX if we encounter such words, create a separate RegExpr that uses 
200 	//         whitespaces & START-/END-expression for detecting word-boundaries, i.e. something like: (\s|^)(~~ ... words ... ~~)(\s|$)
201 	//
202 	//  NOTE: the word-boundaries expression \b seems to have no effect in case of non-ASCII chars in general
203 	//        (e.g. for Japanese characters / words)
204 	//        .... so we would need to use this alternative mechanism (e.g. using whitespaces & START-/END-expr.) 
205 	//        even if these characters were not encoded! 
206 	var encStartTester = new RegExp("^" + this.enc_regexp_str      ,"gm");
207 	var encEndTester   = new RegExp(      this.enc_regexp_str + "$","gm");
208 	var enc_stop_words = "";
209 	var isEncWord = function(str){
210 		return encStartTester.test(str) || encEndTester.test(str); 
211 	};
212 	
213 	
214 	if(size > 0){
215 		
216 		//... then the RegExp matches each stopword:
217 		for(var index=0; index < size ; ++index){
218 			var stop_word = json_stop_words[index];
219 			
220 			//special treatment for word that begin/end with encoded chars:
221 			if(isEncWord(stop_word)){
222 				if(enc_stop_words.length === 0){
223 					enc_stop_words = "(\\s|^)(";
224 				}
225 				else {
226 					enc_stop_words += "|";
227 				}
228 				
229 				enc_stop_words += stop_word;
230 				
231 				continue;
232 			}
233 			
234 			//... for "normal" stopwords:
235 			
236 			if (stop_words.length > 0){
237 				stop_words +=	"|";    //... if there is already a previous stopword-entry: do add OR-matching ...
238 			}
239 	
240 			stop_words +=	stop_word;  //... add the stopword "stop_word"
241 		}
242 	}
243 	
244 	if(stop_words.length > 0){
245 		
246 		stop_words = 
247 				 "\\b("             //starting at a word-boundary (-> ignore within-word matches)
248 			   
249 			+ stop_words 
250 			   
251 			   + ")"
252 		       + "\\b"	            //... ending with a word-boundary -> avoid "cutting out" matching partial strings
253 	                                //    e.g. without \b: '(in)\s?' would match (and cut out all matches) within "winning" -> "wng"
254 		       
255 		       + "\\s?";	        //... and optionally: one white-character that follows the stopword
256 	}
257 	else {
258 		//for empty stopword definition: match empty string
259 		//  (basically: remove nothing)
260 		stop_words += '^$';
261 	}
262 	this.stop_words_regexp = new RegExp(stop_words,"igm");	//RegExp options: 
263 															// ignore-case (i),
264 															// match globally i.e. all occurrences in the String (g), 
265 															// do not stop at line breaks (m)
266 	
267 	
268 	//only create ReExp for special stopwords, if we actually have at least 1 of those:
269 	//NOTE for replacement, we need to use a space-char (i.e. replace these with spaces, not empty strings: str.replace(..., ' '); )
270 	if(enc_stop_words.length > 0){
271 			enc_stop_words += ")(\\s|$)";
272 			this.stop_words_regexp_enc = new RegExp(enc_stop_words,"igm");	
273 	}
274 
275 	//DISABLED: only create these if necessary (i.e. if getStopWordsRegExpr_alt() is called)
276 //	//initialize the alternative version / regular expression for stopwords:
277 //	this.parseStopWords_alt();
278 };

280 //initialize alternative version / regular expression for stopwords:
281 GrammarConverter.prototype.parseStopWords_alt = function(){
282 	
283 	var json_stop_words = this.json_grammar_definition.stop_word;
284 	var size = json_stop_words.length;
285 	var stop_words = "";
286 	
287 	if(size > 0){
288 		stop_words += "(";
289 
290 		for(var index=0; index < size ; ++index){
291 			var stop_word = json_stop_words[index];
292 			if (index > 0) {
293 				stop_words += "|";
294 			}
295 			//create match pattern for: (1) stopword enclosed in spaces, (2) the stopword at 'line end' preceded by a space, (3) the stopword at 'line start' followed by a space
296 			stop_words += " " + stop_word + " | " + stop_word + "$|^" + stop_word
297 					+ " ";
298 		}
299 		
300 		stop_words += ")";
301 	}
302 	else {
303 		//for empty stopword definition: match empty string
304 		//  (basically: remove nothing)
305 		stop_words += '^$';
306 	}
307 	this.stop_words_regexp_alt = new RegExp(stop_words,"igm");
308 };

590 /**
591  * Masks unicoded characters strings.
592  * 
593  * Unicode characters are mask by replacing them with
594  * <code>~~XXXX~~</code>
595  * where <code>XXXX</code> is the four digit unicode HEX number.
596  * 
597  * <p>
598  * NOTE that this function is <em>stable</em> with regard to
599  * multiple executions:
600  * 
601  * If the function is invoked on the returned String again, the
602  * returned String will be the same / unchanged, i.e.
603  * maskings (i.e. "~~XXXX~~") will not be masked again.
604  * </p>
605  * <p>
606  * NOTE: currently, the masking pattern cannot be escaped,
607  * 		 i.e. if the original String contains a substring
608  * 		 that matches the masking pattern, it cannot
609  * 		 be escaped, so that the unmask-function
610  * 		 will leave it untouched.
611  * </p>
612  * 
613  * @param {String} str
614  * 				the String to process
615  * @param {String} [prefix] OPTIONAL
616  * 				an alternative prefix used for masking, i.e instead of <code>~~</code>
617  * 				(ignored, if argument has other type than <code>string</code>)
618  * @param {String} [postfix] OPTIONAL
619  * 				an alternative postfix used for masking, i.e instead of <code>~~</code>
620  * 				(ignored, if argument has other type than <code>string</code>)
621  * @returns {String} 
622  * 				the masked string
623  */
624 GrammarConverter.prototype.maskString = function (str, prefix, postfix) {
625 	var i, s, ch, peek, result,
626 		next, endline, push, mask,
627 		spaces, source = str;
628 	
629 	var ESC_START = typeof prefix  === 'string'? prefix  : '~~';
630 	var ESC_END   = typeof postfix === 'string'? postfix : '~~';
631 	
632 	// Stash the next character and advance the pointer
633 	next = function () {
634 		peek = source.charAt(i);
635 		i += 1;
636 	};
637 	
638 	// Start a new "line" of output, to be joined later by <br />
639 	endline = function () {
640 		result.push('\n');
641 	};
642 	
643 	function mask(theChar) {
644 		
645 		result.push(ESC_START);
646 		
647 		var theUnicode = theChar.charCodeAt(0).toString(16).toUpperCase();
648 		var j = theUnicode.length;
649 		while (j < 4) {
650 //			theUnicode = '0' + theUnicode;
651 			result.push('0');
652 			++j;
653 		}
654 		result.push(theUnicode);
655 
656 		result.push(ESC_END);
657 	};
658 	
659 	// Push a character or its entity onto the current line
660 	push = function () {
661 		
662 		//handle NEWLINE:
663 		if (ch === '\r' || ch === '\n') {
664 			if (ch === '\r') {
665 				if (peek === '\n') {
666 					next();
667 				}
668 				endline();
669 			}
670 			if (ch === '\n') {
671 				if (peek === '\r') {
672 					next();
673 				}
674 				endline();
675 			}
676 		}
677 		//handle tabs
678 		else if (ch === '\t') {
679 			result.push(ch);
680 		}
681 		//handle NON-ASCII
682 		else if (ch < ' ' || ch > '~') {
683 			mask( ch );
684 		} 
685 		//handle normal chars
686 		else {
687 			result.push(ch);
688 		}
689 	};
690 	
691 	
692 	result = [];
693 
694 	i = 0;
695 	next();
696 	while (i <= source.length) { // less than or equal, because i is always one ahead
697 		ch = peek;
698 		next();
699 		
700 		push();
701 	}
702 	
703 	return result.join('');
704 };

707  * HELPER uses #maskString for encoding non-ASCII chars to their Unicode representation,
708  * i.e. <code>\uXXXX</code> where XXXX is the Unicode HEX number.
709  * 
710  * 
711  * SHORTCUT for calling <code>maskString(str, '\\u', '')</code>.
712  * 
713  * @example
714  * //for Japanese "" ("please")
715  * maskAsUnicode("") -> "\u4E0B\u3055\u3044"

717  * //... and using default masking:
718  * maskString("") -> "~~4E0B~~~~3055~~~~3044~~"

720 GrammarConverter.prototype.maskAsUnicode = function (str) {
721 	return this.maskString(str, '\\u', '');

724 /**
725  * Unmasks <i>masked unicoded characters</i> in a string.
726  * 
727  * Masked unicode characters are assumed to have the pattern:
728  * <code>~~XXXX~~</code>
729  * where <code>XXXX</code> is the four digit unicode HEX number.
730  * 
731  * <p>
732  * NOTE that this function is <em>stable</em> with regard to
733  * multiple executions, <b>IF</b> the original String <tt>str</tt> did not
734  * contain a sub-string that conforms to the encoding pattern 
735  * (see remark for {@link #maskString}):
736  * 
737  * If the function is invoked on the returned String again, the
738  * returned String will be the same, i.e. unchanged. 
739  * </p>
740  * 
741  * @param {String} str
742  * @param {RegExp} [detector] OPTIONAL
743  * 				an alternative detector-RegExp:
744  * 				the RegExp must conatin at least one grouping which detects a unicode number (HEX),
745  * 				e.g. default detector is <code>~~([0-9|A-F|a-f]{4})~~</code> (note the grouping
746  * 				for detecting a 4-digit HEX number within the brackets).
747  * @returns {String} the unmasked string
748  */
749 GrammarConverter.prototype.unmaskString = function (str, detector) {
750 	var match, source = str, result = [], pos = 0, i, len = str.length;
751 	
752 	//RegExpr for: ~~XXXX~~
753 	// where XXXX is the unicode HEX number: ~~([0-9|A-F|a-f]{4})~~
754 	var REGEXPR_ESC = detector? detector : new RegExp( this.enc_regexp_str, "igm");
755 	
756 	while(match = REGEXPR_ESC.exec(source)){
757 		i =  match.index;
758 		//add previous:
759 		if(i > pos){
760 			result.push(source.substring(pos, i));
761 		}
762 		
763 		//add matched ESC as UNICODE:
764 		result.push(String.fromCharCode(  parseInt(match[1], 16) ));
765 		
766 		//update position:
767 		pos = i + match[0].length;
768 	}
769 	
770 	if(pos < len){
771 		result.push(source.substring(pos));
772 	}
773 
774 	return result.join('');
775 };

786 /**
787  * Recodes Strings of a JSON-like object.
788  * 
789  * @function
790  * @param {Object} json 
791  * 					the JSON-like object (i.e. PlainObject)
792  * 
793  * @param {Function} recodeFunc
794  * 								the "recoding" function for modifying String values:
795  * 								 must accecpt a String argument and return a String
796  * 									<code>String recodeFunc(String)</code>.
797  * 								The <tt></tt> function is invoked in context of the GrammarConverter object.
798  * 								Example: this.maskString().
799  * 								See {@link #maskString}.k
800  * 
801  * @param {Boolean} [isMaskValues] OPTIONAL
802  * 								 if true, the object's property String values will be processed
803  * 								 NOTE: in case this parameter is specified, then <code>recodeFunc</code> must
804  * 									   also be specified!
805  * 								 DEFAULT: uses property {@link #maskValues}
806  * @param {Boolean} [isMaskNames]  OPTIONAL
807  * 								 if true, the property names will be processed
808  * 								 NOTE: in case this parameter is specified, then <code>recodeFunc</code> and
809  * 									   <code>isMaskValues</code> must also be specified!
810  * 								 DEFAULT: uses property {@link #maskNames}
811  * 
812  * @returns {Object} the recoded JSON object
813  * 
814  * @requires {@link mmir.CommonUtils#isArray} or {@link Array#isArray}
815  */
816 GrammarConverter.prototype.recodeJSON = (function () {//<- NOTE this is only the initializer (i.e. see returned function below)
817 	
818 	var isArray;
819 	if(typeof commonUtils !== 'undefined'){
820 		isArray = commonUtils.isArray;//FIXME this requires ArrayExtension.js !!!
821 	} 
822 	else { 
823 		isArray = Array.isArray;
824 	}
825 	
826 	//recursive processing for an object
827 	//returns: the processed object
828 	var processJSON = function(obj, recodeFunc, isMaskValues, isMaskNames){
829 		
830 		//different treatments for: STRING, ARRAY, OBJECT types (and 'REST' type, i.e. all ohters)
831 		if(typeof obj === 'string' && isMaskValues){
832 			//STRING: encode the string
833 			return recodeFunc.call(this, obj);
834 		}
835 		else if( isArray(obj) ) {
836 			//ARRAY: process all entries:
837 			for(var i=0, size = obj.length; i < size; ++i){
838 				obj[i] = processJSON.call(this, obj[i], recodeFunc, isMaskValues, isMaskNames);
839 			}
840 			
841 			return obj;
842 		}
843 		else if(typeof obj === 'object') {
844 			//OBJECT: process all the object's properties (but only, if they are not inherited)
845 			for(var p in obj){
846 				if(obj.hasOwnProperty(p)){
847 					
848 					obj[p] = processJSON.call(this, obj[p], recodeFunc, isMaskValues, isMaskNames);
849 					
850 					//if the property-name should also be encoded:
851 					if(typeof p === 'string' && isMaskNames){
852 						
853 						var masked = recodeFunc.call(this, p);
854 						if(masked !== p){
855 							obj[masked] = obj[p];
856 							delete obj[p];
857 						}
858 					}
859 				}
860 			}
861 			return obj;
862 		}
863 		else {
864 			return obj;
865 		}
866 	};
867 	
868 	return function (json, recodeFunc, isMaskValues, isMaskNames){
869 		//evalate arguments:
870 		if(typeof isMaskValues === 'undefined'){
871 			isMaskValues = this.maskValues;
872 		}
873 		if(typeof isMaskNames === 'undefined'){
874 			isMaskNames = this.maskNames;
875 		}
876 		
877 		return processJSON.call(this, json, recodeFunc, isMaskValues, isMaskNames);
878 	};
879 	
880 })();
