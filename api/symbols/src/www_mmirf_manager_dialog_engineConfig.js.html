 96     var _browserFactory = {
 97 		createWorker: function(_instance, gen) {
 98 
 99 			var scionQueueWorker = new Worker(
100 					constants.getWorkerPath()+ 'ScionQueueWorker.js'
101 			);
102 
103 			scionQueueWorker.onmessage = function(e) {
104 
105 				if (e.data.command == "toDo") {
106 
107 					_instance._log.debug('raising:' + e.data.toDo.event);
108 
109 					gen(e.data.toDo.event, e.data.toDo.eventData);
110 
111 					// @chsc03: attention if something goes wrong along the transition,
112 					// the worker is not ready for any incoming jobs
113 					this.postMessage({
114 						command : 'readyForJob'
115 					});
116 
117 					_instance.onraise();
118 				}
119 			};
120 
121 			return scionQueueWorker;
122 
123 		},
124 		createRaise: function(_instance){
125 			return function(event, eventData) {
126 
127 				if (eventData){
128 					this._log.debug('new Job:' + event);
129 				}
130 
131 				_instance.worker.postMessage({
132 					command : 'newJob',
133 					job : {
134 						event : event,
135 						eventData : eventData
136 					}
137 				});
138 			};
139 		}

143     var _androidFactory = {
144 		createWorker: (function initWorkerFactory() {
145 
146 			//"global" ID-list for all queues (i.e. ID-list for all engines)
147 			var callBackList = [];
148 
149 			function successCallBackHandler(args){
150 				if (args.length=2){
151 //  				console.debug('QueuePlugin: success '+ JSON.stringify(args[0]) + ', '+JSON.stringify(args[1]));//DEBUG
152 					callBackList[args[0]](args[1]);
153 				}
154 			}
155 
156 
157 
158 			return function workerFactory(_instance, gen){
159 
160 				var id = callBackList.length;
161 
162 				function failureFallbackHandler(err){
163 
164 					_instance._log.error('failed to initialize SCION extension for ANDROID evn');
165 					_instance.worker = (function(gen){
166 						return { 
167 							raiseCordova: function fallback(event, eventData){
168 								setTimeout(function(){
169 									gen(event, eventData);
170 								}, 1);
171 							}
172 						};
173 					})();//END: fallback
174 				}
175 
176 				callBackList.push(function(data){
177 					if(_instance._log.isv()) _instance._log.debug('raising:'+ data.event);
178 					var generatedState = gen(data.event, data.eventData);
179 					if(_instance._log.isv()) _instance._log.debug('QueuePlugin: processed event '+id+' for '+ data.event+' -> new state: '+JSON.stringify(generatedState)+ ' at ' + _instance.url);
180 					plugins.queuePlugin.readyForJob(id, successCallBackHandler, failureFallbackHandler);
181 
182 					_instance.onraise();
183 				});
184 				plugins.queuePlugin.newQueue(id, function(args){
185 					if(_instance._log.isv()) _instance._log.debug('QueuePlugin: entry '+id+' created.' + ' at ' + _instance.url);
186 				}, failureFallbackHandler
187 				);
188 
189 				return {
190 					raiseCordova: function (event, eventData){
191 						if(_instance._log.isv()) _instance._log.debug('QueuePlugin: new Job: '+ id + ' at ' + _instance.url);
192 						plugins.queuePlugin.newJob(id, {event: event, eventData: eventData}, successCallBackHandler,failureFallbackHandler);
193 					}
194 				};
195 				
196 			};//END: workerFactory(_instance, gen)
197 
198 		})(),//END: initWorkerFactory()
199 		
200 		createRaise: function(_instance){
201 			return function(event, eventData) {
202 
203 				if (eventData) _instance._log.log('new Job:' + event);
204 
205 				_instance.worker.raiseCordova(event, eventData);
206 			};
207 		}
208 
209     };
