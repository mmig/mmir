 27 define(['core', 'jquery', 'commonUtils', 'module', 'engineConfig', 'logger' ],
 28 	/**
 29 	 * The InputManager handles input events.
 30 	 * 
 31 	 * <p>
 32 	 * On initialization, the InputManager also creates the {@link mmir.InputEngine},
 33 	 * and returns it as the second argument of the {@link #init}() function's callback
 34 	 * (or the Promise's triggered callbacks).
 35 	 * 
 36 	 * In addition, the InputEngine is exported as module <code>"inputEngine"</code> via
 37 	 * RequireJS' <code>define()</code> function.
 38 	 * 
 39 	 * @example
 40 	 * //initialization of inputManager
 41 	 * require('inputManager').init().then( function(inputManagerInstance, inputEngineInstance){
 42 	 * 		//do something
 43 	 * });
 44 	 * 
 45 	 * 
 46 	 * @name mmir.InputManager
 47 	 * @static
 48 	 * @class
 49 	 * 
 50 	 * 
 51      *  @requires jQuery.Deferred
 52      *  @requires jQuery.extend
 53 	 * 
 54 	 */
 55 	function(
 56 		mmir, $, commonUtils,module, engineConfig, Logger
 57 ){
 58 
 59 	//next 2 comments are needed by JSDoc so that all functions etc. can
 60 	// be mapped to the correct class description
 61 	/** @scope mmir.InputManager.prototype */
 62 	/**
 63 	 * #@+
 64 	 * @memberOf mmir.InputManager.prototype 
 65 	 */
 66 	
 67 	/**
 68 	 * @memberOf mmir.InputManager#
 69 	 */
 70 	var _instance = {
 71 
 72 		/** @scope mmir.InputManager.prototype */
 73 		
 74 		/** 
 75 		 * @deprecated instead: use mmir.InputManager object directly.
 76 		 * 
 77 		 * @memberOf mmir.InputManager.prototype
 78 		 */
 79 		getInstance : function() {
 80 			return this;
 81 		},
 82 		
 83 		/**
 84 		 * This function raises an event. 
 85 		 * 
 86 		 * @function raise
 87 		 * @param {String} eventName
 88 		 * 				The name of the event which is to be raised
 89 		 * @param {Object} [eventData] OPTIONAL
 90 		 * 				Data belonging to the event
 91 		 * @throws {Error} if this function is invoked while the internal
 92 		 * 				   event/state engine (i.e. {@link mmir.InputEngine}
 93 		 * 				   is not initialized yet
 94 		 * @public
 95 		 */
 96 		raise : function(eventName, eventData) {
 97 			//NOTE the functional implementation will be set during initialization (see below #init())
 98 			throw new Error('InputEngine not initialized yet: '
 99 					+'call mmir.InputManager.init(callback) and wait for the callback.'
100 			);
101 		}
102 
103 	};
104 
105 	return $.extend(true, _instance, {
106 
107 		/**
108 		 * @function
109 		 * @name init
110 		 * @returns {Deferred}
111 		 * 
112 		 * @memberOf mmir.InputManager.prototype
113 		 */
114 		init : function() {
115 			delete this.init;
116 			
117 			//"read" settings from requirejs' config (see mainConfig.js):
118 			var url = module.config().scxmlDoc;
119 			var mode = module.config().mode;
120 			
121 			//create a SCION engine:
122 			var engine = engineConfig(url, mode);
123 			
124 			this._log = Logger.create(module);
125 			engine._log = Logger.create(module.id+'Engine', module.config().logLevel);
126 
127 //			var _self = this;
128 
129 			return $.Deferred(function(theDeferredObj) {
130 				
131 				engine.load().done(function(_engine) {
132 					
133 					_instance.raise = function raise(){
134 						_engine.raise.apply(_engine, arguments);
135 					};
136 					
137 //					mmir.InputEngine = _engine;
138 					delete _engine.gen;
139 
140 					//register the InputEngine with requirejs as module "inputEngine":
141 					define("inputEngine", function(){
142 						return _engine;
143 					});
144 					//immediately load the module-definition:
145 					require(['inputEngine'], function(){
146 						//signal end of initialization process:
147 						theDeferredObj.resolve(_instance, _engine);	
148 					});
149 					
150 				});
151 				
152 			}).promise();
153 		}
154 	});
155 	
156 	/** #@- */
157 
158 });
