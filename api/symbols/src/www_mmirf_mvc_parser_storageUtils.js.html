 25 /**
 26  * Creates the appropriate object from a JSON-like <tt>storedObject</tt>.
 27  * 
 28  * <p>
 29  * NOTE that in difference to a real JSON object, the <tt>storedObject</tt> 
 30  * may contain function definitions.
 31  * 
 32  * <p>
 33  * The storedObject must have a String property <strong>classConstructor</strong>
 34  * <ul>
 35  * 	<li>that must correspond to a constructor function (which will be invoked with <tt>new</tt>)</li>
 36  * 	<li>the constructor function must be invokable without parameters</li>
 37  * 	<li>the constructor function must be accessable from the global namespace
 38  * 		(or <tt>classConstuctor</tt> must contain the code for retrieving the constructor
 39  * 		 function from the global namespace)</li>
 40  * </ul>
 41  * 
 42  * 
 43  * <p>
 44  * If <tt>storedObject</tt> contains a function <tt>init</tt>, then this function will be invoked
 45  * before returning the new newly created object.
 46  * 
 47  * 
 48  * @function
 49  * @static
 50  * @public
 51  * 
 52  * @param {Object} storedObject 
 53  * 				    a JSON-like object with fields and functions (which will be transfered to the returned object).
 54  * @param {Boolean} [isTriggerPublish] OPTIONAL
 55  * 					if <code>true</code> then the restore function call
 56  * 					<code>initPublish()</code> on the restored object before returning.
 57  * 					This should only be <code>true</code> for the root-object
 58  * 					(e.g. the View-object or Partial-object; nested objects should NOT invoke
 59  * 					 restoreObject() with this argument set to true).
 60  * @param {Number} [fileFormatNo] OPTIONAL
 61  * 					NOTE: if argument <code>isTriggerPublish</code> was used with value <code>true</code>,
 62  * 						  then this argument MUST be used too!
 63  * 					If the number given does not match {@link parser.STORAGE_FILE_FORMAT_NUMBER}
 64  * 					the file format is assumed to be out-dated and an Error will be thrown.
 65  * @returns {Object} 
 66  *          an new instance created with the constructor <tt>classConstructor</tt> and
 67  * 			set with all properties (fields and functions) from <tt>storedObject</tt>.
 68  * 
 69  * @throws Error if <code>fileFormatNo</code> does not match STORAGE_FILE_FORMAT_NUMBER.
 70  */
 71 parser.restoreObject = function(storedObject, isTriggerPublish, fileFormatNo){
 72 	
 73 	if(isTriggerPublish && fileFormatNo != parser.STORAGE_FILE_FORMAT_NUMBER){
 74 		
 75 		throw new Error('Compiled template file has wrong format: need file with format version '
 76 				+ parser.STORAGE_FILE_FORMAT_NUMBER +', but got: '+ fileFormatNo
 77 				+ '. Please re-compile views / templates.'
 78 		);
 79 		
 80 	}
 81 	
 82 	var classExtender;
 83 	if(parser.CLASS_EXTENDER && typeof parser.CLASS_EXTENDER.extend === 'function'){
 84 		classExtender = parser.CLASS_EXTENDER;
 85 	}
 86 	else {
 87 		classExtender = $;
 88 	}
 89 	
 90 //	//NOTE: classConstructor contains a list of Strings:
 91 //	//       * the constructor-function is either in global namespace
 92 //	//         e.g. "View" -> then the list contains exactly one entry ["View"]
 93 //	//
 94 //	//       * if the constructor-function is in a sub-namespace (e.g. "sub-module")
 95 //	//         then the list contains the "path" to the constructorf-function
 96 //	//         starting with the module that is in the global namespace
 97 //	//         e.g. "mobileDS.parser.ParsingResult" -> ["mobileDS, "parser", "ParsingResult"]
 98 //	//
 99 //	var constructor = window[storedObject.classConstructor[0]];//FIXME need to convert this to require
100 //	for(var i=1, size = storedObject.classConstructor.length; i < size; ++i){
101 //		constructor = constructor[storedObject.classConstructor[i]];
102 //	}
103 	
104 	//requirejs version (for this to work, all Classe (i.e. JS-files) need to already have been loaded & required!)
105 	var constructor = require(storedObject.classConstructor);
106 	
107 	var obj = classExtender.extend( new constructor(), storedObject);
108 	if(typeof obj.init === 'function'){
109 		obj.init();
110 	}
111 	
112 	if(isTriggerPublish && typeof obj.initPublish === 'function'){
113 		obj.initPublish();
114 	}
115 	
116 	return obj;
117 };

119 /**
120  * Creates String-representations (JSON-like) for the specified properties and appends them to the StringBuffer.
121  * 
122  * <p>
123  * This function iterates over the Array <tt>propertyNames</tt>:
124  * If a property with that name exists in <tt>obj</tt>, a JSON-like String-representation is generated
125  * and appended at the end of the array <tt>stringBuffer</tt>.
126  * <br>
127  * Multiple representations are separated by comma entry <code>','</code> in <tt>stringBuffer</tt>.
128  * <br>
129  * The last entry in <tt>stringBuffer</tt> is a comma entry <code>','</code> if at least one property 
130  * entry was inserted in <tt>stringBuffer</tt>.
131  * 
132  * <p>
133  * NOTE that the String-representation inserted into <tt>stringBuffer</tt> may not have a 1:1 correspondence
134  * with properties (only the last entry is guaranteed to be <code>','</code>, if a property was inserted).
135  * <br>
136  * For pratical use, the returned (or modified) <tt>stringBuffer</tt> should be converted into a String
137  * e.g. by <code>stringBuffer.join('')</code>.
138  * 
139  * 
140  * @function
141  * @static
142  * @public
143  * 
144  * @param {Object} obj 
145  * 				the object, that contains the properties for which String representations should be generated.
146  * @param {Array<String>} propertyNames 
147  * 				the names of the properties, for which String-representations should be generated.
148  * @param {Array<String>} stringBuffer 
149  * 				the buffer: String-representations will be appended as entries at the end of the buffer
150  * @param {String} [propertyNamePostfix] 
151  * 				OPTIONAL if present, this postfix will be appended to each property name, before processing it.
152  * 				This is a convenience method, e.g. if all properties in <tt>propertyNames</tt> should end with
153  * 				the same String / postfix.
154  * @param {Function} [valueFunc] 
155  * 				OPTIONAL by default, value representations are generated using the <code>JSON.stringify</code>
156  * 				function. If instead this argument is present, this function will be invoked for creating
157  * 				the string representation of the property-value.
158  * 				The function signature is <code>valueFunc(propertyName : String, propertyValue : Object) : String</code>.
159  * 
160  * @returns {Array<String>} the modified <tt>stringBuffer</tt>
161  * 
162  * @requires JSON.stringify
163  * 
164  * @example
165  * 
166  * var obj = {
167  * 	some: "properties",
168  * 	including: function(arg1,arg2){ return 'functions' }
169  * };
170  * 
171  * var sb = mobileDS.parser.appendStringified(obj, ['some'], []);
172  * var str = sb.join(',');
173  * //str will be: "some:\"properties\","
174  * 
175  */
176 parser.appendStringified = function(obj, propertyNames, stringBuffer, propertyNamePostfix, valueFunc){
177 	
178 	//"shift" arguments, if necessary
179 	if(typeof propertyNamePostfix === 'function' && ! valueFunc){
180 		valueFunc = propertyNamePostfix;
181 		propertyNamePostfix = null;
182 	}
183 	
184 	var prop, val;
185 	for(var i=0, size = propertyNames.length; i < size; ++i){
186 		prop = propertyNames[i];
187 		
188 		if(propertyNamePostfix){
189 			prop += propertyNamePostfix;
190 		}
191 
192 		if(typeof obj[prop] === 'undefined'){
193 			continue;
194 		}
195 		
196 		
197 		if(valueFunc){
198 			val = valueFunc(prop, obj[prop]);
199 		}
200 		else {
201 			val = JSON.stringify(obj[prop]);
202 		}
203 		
204 		stringBuffer.push( prop );
205 		stringBuffer.push( ':' );
206 		stringBuffer.push( val );
207 
208 		stringBuffer.push( ',' );
209 	}
210 	
211 	return stringBuffer;
212 };
213 
214 return parser;
