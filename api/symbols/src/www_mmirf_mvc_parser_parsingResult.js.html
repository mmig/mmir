 43 /**
 44  * ParsingResult represents an element that was detected during parsing.
 45  * 
 46  * <p>
 47  * The detected element is referenced by the properties <code>start</code> and <code>end</code>
 48  * that refer to the start-index and end-index within the parsed text.
 49  * 
 50  * <p>
 51  * The ParsingResult has a <code>type</code> property which refers to the kind of element
 52  * that was detected (see constants in {@link mmir.parser.element}).
 53  * 
 54  * <p>
 55  * In addition, the ParsingResult may have several properties that depend of its type. In general,
 56  * these properties refer to detected parts of the element (e.g. for a invocation-statement, these
 57  * may refer to its arguments).
 58  * 
 59  * 
 60  * <p>
 61  * Properties for <strong>INCLUDE_SCRIPT</strong> type:
 62  * <ul>
 63  * 	<li><strong>scriptPath</strong>: the path / URL to the resource</li>
 64  * 	<li><strong>scriptPathType</strong>: the type of the <tt>scriptPath</tt>
 65  * 		 field: one of <code>StringLiteral</code>, <code>Identifier</code>, <code>IdentifierNameAmpersatStart</code></li>
 66  * </ul>
 67  * <p>
 68  * Properties for <strong>INCLUDE_STYLE</strong> type:
 69  * <ul>
 70  * 	<li><strong>stylePath</strong>: </li>
 71  * 	<li><strong>stylePathType</strong>: the type of the <tt>stylePath</tt>
 72  * 		 field: one of <code>StringLiteral</code>, <code>Identifier</code>, <code>IdentifierNameAmpersatStart</code></li>
 73  * </ul>
 74  * <p>
 75  * Properties for <strong>LOCALIZE</strong> type:
 76  * <ul>
 77  * 	<li><strong>name</strong>: the name/identifier for the localized String</li>
 78  * 	<li><strong>nameType</strong>: the type of the <tt>name</tt>
 79  * 		 field: one of <code>StringLiteral</code>, <code>Identifier</code>, <code>IdentifierNameAmpersatStart</code></li>
 80  * </ul>
 81  * <p>
 82  * Properties for <strong>YIELD_DECLARATION</strong> type:
 83  * <ul>
 84  * 	<li><strong>name</strong>: the name/identifier for the yield section</li>
 85  * 	<li><strong>nameType</strong>: the type of the <tt>name</tt>
 86  * 		 field: one of <code>StringLiteral</code>, <code>Identifier</code>, <code>IdentifierNameAmpersatStart</code></li>
 87  * </ul>
 88  * Properties for <strong>BLOCK</strong> type:
 89  * <ul>
 90  * 	<li><strong>scriptContent</strong> {String}: <tt>OPTIONALLY</tt> the script code as a String</li>
 91  * 	<li><strong>scriptEval</strong> {Function}: the compiled script code in form of a function. The 
 92  * 							function takes one argument: the current data-object.</li>
 93  * </ul>
 94  * <p>
 95  * Properties for <strong>STATEMENT</strong> type:
 96  * <ul>
 97  * 	<li><strong>scriptContent</strong> {String}: <tt>OPTIONALLY</tt> the script code as a String</li>
 98  * 	<li><strong>scriptEval</strong> {Function}: the compiled script code in form of a function. The 
 99  * 							function takes one argument: the current data-object.</li>
100  * </ul>
101  * <p>
102  * Properties for <strong>HELPER</strong> type:
103  * <ul>
104  * 	<li><strong>helper</strong>: the name of the helper function</li>
105  * 	<li><strong>helperType</strong>: the type of the <tt>helper</tt>
106  * 		 field: one of <code>StringLiteral</code>, <code>Identifier</code>, <code>IdentifierNameAmpersatStart</code></li>
107  * 	<li><strong>argsEval</strong> {Function}: OPTIONALLY compiled getter Function for retrieving the current ARGS 
108  * 			(optional argument) of the helper expression. The function takes one argument: the current data-object.</li>
109  * </ul>
110  * <p>
111  * Properties for <strong>IF</strong> type:
112  * <ul>
113  * 	<li><strong>ifEval</strong> {Function}: the condition statement, that was compiled into a Function. The 
114  * 							function takes one argument: the current data-object.</li>
115  * 	<li><strong>content</strong> {ContentElement}: the HTML / template content that should be render, in case the
116  * 					if-expression evaluates to <code>true</code>.</li>
117  *  <li><strong>elseContent</strong> {@link mmir.parser.ParsingResult}: OPTIONALLY a ParsingResult
118  *  					representing an else-expression, see {@link mmir.parser.element.ELSE}.</li>
119  * </ul>
120  * <p>
121  * Properties for <strong>ELSE</strong> type:
122  * <ul>
123  * 	<li><strong>content</strong> {ContentElement}: the HTML / template content that should be render, in case the
124  * 					if-expression (to which the else-expression belongs) evaluates to <code>false</code>.</li>
125  * </ul>
126  * <p>
127  * Properties for <strong>FOR</strong> type:
128  * <ul>
129  * 	<li><strong>forControlType</strong> {String}: the type of for-loop, either <code>FORITER</code> or <code>FORSTEP</code></li>
130  * 	<li><strong>forInitEval</strong> {Function}: the initialization statement of the for-expression, compiled into
131  * 									a Function. The function takes one argument: the current data-object.</li>
132  * 		
133  * 	<li><code>FORITER</code>: <code>@for(PROP in OBJ){ ... }@</code>
134  * 	 <ul>
135  * 		<li><strong>forIterator</strong> {Object}: an iterator object with functions <code>hasNext() : Boolean</code> and
136  * 					<code>next() : String</code> (which returns the name of the property currently iterated).</li>
137  * 		<li><strong>forPropName</strong> {String}: the variable name for the property which is currently iterated over.</li>
138  * 	 </ul>
139  * 	</li>
140  * 	<li><code>FORSTEP</code>: <code>@for(INIT; CONDITION; INCREMENT){ ... }@</code>
141  * 	 <ul>
142  * 		<li><strong>forConditionEval</strong> {Function}: the condition statement of the for-expression, compiled into
143  * 											a Function. The function takes one argument: the current data-object.</li>
144  * 		<li><strong>forIncrementEval</strong> {Function}: the increment statement of the for-expression, compiled into
145  * 											a Function. The function takes one argument: the current data-object.</li>
146  * 	 </ul>
147  * 	</li>
148  * 	<li><strong>content</strong> {ContentElement}: the HTML / template content that should be rendered
149  * 								 during each iteration of the for-loop.</li>
150  * </ul>
151  * <p>
152  * Properties for <strong>RENDER</strong> type:
153  * <ul>
154  * 	<li><strong>partial</strong>: the name of the partial view</li>
155  * 	<li><strong>partialType</strong>: the type of the <tt>partial</tt>
156  * 		 field: one of <code>StringLiteral</code>, <code>Identifier</code>, <code>IdentifierNameAmpersatStart</code></li>
157  * 	<li><strong>controller</strong>: the name of the controller, to which the partial view definition belongs</li>
158  * 	<li><strong>controllerType</strong>: the type of the <tt>controller</tt>
159  * 		 field: one of <code>StringLiteral</code>, <code>Identifier</code>, <code>IdentifierNameAmpersatStart</code></li>
160  * 	<li><strong>argsEval</strong> {Function}: OPTIONALLY compiled getter Function for retrieving the current ARGS 
161  * 								(optional argument) of the render expression. The function takes one argument: the current data-object.</li>
162  * </ul>
163  * <p>
164  * Properties for <strong>ESCAPE_ENTER</strong> type:
165  * <ul>
166  * 	<li><strong>text</strong> {String}: the text that will be rendered (i.e. without the escape-character(s) itself).</li>
167  * </ul>
168  * <p>
169  * Properties for <strong>ESCAPE_EXIT</strong> type:
170  * <ul>
171  * 	<li><strong>text</strong> {String}: the text that will be rendered (i.e. without the escape-character(s) itself).</li>
172  * </ul>
173  * <p>
174  * Properties for <strong>VAR_DECLARATION</strong> type:
175  * <ul>
176  * 	<li><strong>name</strong>: the name for the variable (without the leading <tt>@</tt> of template variables)</li>
177  * 	<li><strong>nameType</strong>: the type of the <tt>name</tt> field: <code>StringLiteral</code></li>
178  * </ul>
179  * 
180  * <p>
181  * 
182  * @class ParsingResult
183  * @param {org.antlr.runtime.CommonTokenStream} thetokens (optional) the TokenStream that corresponds to this parsed element;
184  * 												when provided, the TokenStream is used to set the start- and end-
185  * 												property of the new instance.
186  * 		  {org.antlr.runtime.Token} thetokens (optional) if the parameter is a single Token object, then the start- and end-
187  * 												property for the new instance is set by this token object
188  * 
189  * @category parser
190  */
191 parser.ParsingResult = function (thetokens){
192 	var isSet = false;
193 	
194 	//try to extract start-/end-indexes from the argument:
195 	if(thetokens){
196 		
197 		if(typeof org !== 'undefined'){
198 			//NOTE: must invoke getTokens() for initializing size() etc.!
199 			if(thetokens instanceof org.antlr.runtime.CommonTokenStream && thetokens.getTokens() && thetokens.size() > 0){
200 				this.start = thetokens.getTokens()[0].getStartIndex();
201 				this.end = thetokens.getTokens()[thetokens.size()-1].getStopIndex();
202 				
203 				isSet = true;
204 			}
205 			else if(thetokens instanceof org.antlr.runtime.CommonToken || thetokens instanceof org.antlr.runtime.Token){
206 				this.start = thetokens.getStartIndex();
207 				this.end = thetokens.getStopIndex();
208 							
209 				isSet = true;
210 			}
211 		}
212 		
213 		//if not already set, try...
214 		if( isSet === false && typeof thetokens.getToken !== 'undefined' && (typeof thetokens.getToken().getStartIndex !== 'undefined' && typeof thetokens.getToken().getStopIndex !== 'undefined')){
215 			this.start = thetokens.getToken().getStartIndex();
216 			this.end = thetokens.getToken().getStopIndex();
217 						
218 			isSet = true;
219 		}
220 		else if(typeof thetokens.getStartIndex !== 'undefined' && typeof thetokens.getStopIndex !== 'undefined'){
221 			this.start = thetokens.getStartIndex();
222 			this.end = thetokens.getStopIndex();
223 						
224 			isSet = true;
225 		}
226 		else if(isSet === false) {
227 			var type = Object.prototype.toString.call(thetokens);//.match(/^\[object (.*)\]$/)[1];
228 			console.warn('unknown argument type: '+type);//debug
229 		}
230 	}
231 	
232 	if(isSet === false) {
233 		this.start = -1;
234 		this.end   = -1;
235 	}
236 };

238 /**
239  * Set the start position (index) for this parsed element with regard to the TokenStream of the complete input.
240  * 
241  * @function setStartFrom
242  * @param {org.antlr.runtime.CommonTokenStream} thetokens (optional) the TokenStream that corresponds to this parsed element;
243  * 												when provided, the TokenStream is used to set the start-property of this object.
244  * 
245  * @public
246  */
247 parser.ParsingResult.prototype.setStartFrom = function(thetokens){
248 	//NOTE: must invoke getTokens() for initializing size() etc.!
249 	if(thetokens.getTokens() && thetokens.size() > 0){
250 		this.start = thetokens.getTokens()[0].getStartIndex();
251 	} 
252 	else {
253 		this.start = -1;
254 	}
255 };

259  * 
260  * @function setEndFrom
261  * @param {org.antlr.runtime.CommonTokenStream} thetokens (optional) the TokenStream that corresponds to this parsed element;
262  * 												when provided, the TokenStream is used to set the end-property of this object.

264  * @public
265  */

269 		this.end = thetokens.getTokens()[thetokens.size()-1].getStopIndex();

271 	else {
272 		this.end = -1;
273 	}

296 //helper function for converting properties to the correct value.
297 // By default, the ParsingResult only contains "raw" property values.
298 // Which properties are available, depends on the type of the ParsingResult (see templateProcessor.js)
299 parser.ParsingResult.prototype.getValue = function(rawPropertyValue, proptertyType, data){
300 	
301 	if(proptertyType === 'StringLiteral'){
302 		return rawPropertyValue.substring(1, rawPropertyValue.length-1);
303 	}
304 	else if(proptertyType === 'Identifier'){
305 		if(data){
306 			return data['@' + rawPropertyValue];
307 		}
308 		else {
309 			//just return variable name
310 			return rawPropertyValue;
311 		}
312 	}
313 	else if(proptertyType === 'IdentifierNameAmpersatStart'){
314 		if(data){
315 			return data[rawPropertyValue];
316 		}
317 		else {
318 			//just return variable name, but remove leading @:
319 			return rawPropertyValue.substring(1);
320 		}
321 	}
322 	else if(proptertyType === 'OBJECT'){
323 		return rawPropertyValue;//TODO
324 	}
325 	else if(proptertyType === 'DecimalLiteral'){
326 		return parseFloat(rawPropertyValue);
327 	}
328 //	else if(typeof proptertyType === 'undefined'){
329 //		return rawPropertyValue;
330 //	}
331 	else
332 		return rawPropertyValue;
333 };

487 parser.ParsingResult.prototype.stringify = function(){
488 	
489 	//TODO use constants for lists
490 	
491     //typed properties:
492 	// for each typed properties '<prop>' there is an additional property with name '<prop>Type'
493 	// (both of these properties are Strings themselves)
494 	var typedPropList = [
495 	     'scriptPath',
496 	     'stylePath', 
497 	     'name',
498 	     'helper',
499 	     'partial',
500 	     'controller'
501 	];
502 	
503 	//primitive-type properties:
504 	// write values 'as is' for these properties
505 	var propList = [
506    	     'start',
507 	     'end',
508 	     'type',
509 	     'dataType',
510 	     'text',
511 	     'forControlType',
512 	     'forPropName'
513 	];
514 	
515 	//"stringify-able" object properties:
516 	var strPropList = [
517    	     'content',
518    	     'elseContent'
519    	];
520 	
521 	//function properties:
522 	var funcPropList = [
523    	     'scriptEval',
524    	     'argsEval',
525    	     'ifEval',
526    	     'forInitEval',
527    	     'forIterator',
528    	     'forConditionEval',
529    	     'forIncrementEval'
530    	];
531 	
532 
533 	//function for iterating over the property-list and generating JSON-like entries in the string-buffer
534 	var  appendStringified = parser.appendStringified;
535 
536 	var sb = ['require("storageUtils").restoreObject({ classConstructor: "parsingResult"', ','];
537 
538 	//TODO property dataPos: {start: Number, end: Number}
539 	if(this['dataPos']){
540 		sb.push( 'dataPos:{start:' );
541 		sb.push( this.dataPos.start );
542 		sb.push( ',end:' );
543 		sb.push( this.dataPos.end );
544 		sb.push( '}' );
545 		//NOTE: need to add comma in a separate entry 
546 		//      (-> in order to not break the removal method of last comma, see below)
547 		sb.push( ',' );
548 	}
549 		
550 	
551 	appendStringified(this, typedPropList, sb);
552 	appendStringified(this, typedPropList, sb, 'Type');
553 	appendStringified(this, propList, sb);
554 	//non-primitives with stringify() function:
555 	appendStringified(this, strPropList, sb, null, function valueExtractor(name, value){
556 		return value.stringify();
557 	});
558 	//function values
559 	appendStringified(this, funcPropList, sb, null, function valueExtractor(name, value){
560 		return value.toString();
561 	});
562 	
563 	//if last element is a comma, remove it
564 	if(sb[sb.length - 1] === ','){
565 		sb.splice( sb.length - 1, 1);
566 	}
567 	
568 	sb.push(' })');
569 	return sb.join('');
570 };
571 
572 return parser.ParsingResult;
