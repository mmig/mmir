286 				
287 				return (function (res){
288 					
289 					console.log("androidAudioInput"+(repeat? "(REPEAT_MODE)" : "")+": " + JSON.stringify(res));//FIXME DEBUG
290 
291 					var asr_result = null;
292 					var asr_score = -1;
293 					var asr_type = -1;
294 					var asr_alternatives = [];
295 					var asr_unstable = null;
296 
297 					if(res) {
298 
299 						if(typeof res['result'] !== "undefined"){
300 							asr_result = res['result'];
301 						}
302 						if  (typeof res["score"] !== "undefined"){
303 							asr_score = res["score"];
304 						}
305 						if  (typeof res["type"] !== "undefined"){
306 							asr_type = res["type"];
307 						}
308 						if  (typeof res["alternatives"] !== "undefined"){
309 							asr_alternatives = res["alternatives"];
310 						}
311 						if  (typeof res["unstable"] !== "undefined"){
312 							asr_unstable = res["unstable"];
313 						}
314 					}
315 
316 					//call voice recognition again, if repeat is set to true
317 					if (repeat === true) {
318 						if (asr_type === result_types.RECORDING_BEGIN){
319 							// only call success-callback, if we started recording again.
320 							_wait.ready();
321 							call_callback_with_last_result();
322 						} else if (asr_type === result_types.RECORDING_DONE){
323 							// Do nothing right now at the recording done event
324 							console.log("androidAudioInput"+(repeat? "(REPEAT_MODE)" : "")+": RECORDING_DONE, last_result: " + JSON.stringify(last_result));//FIXME DEBUG
325 							
326 						} else if (asr_type === result_types.FINAL){
327 							// its the final result
328 							// post last result
329 							call_callback_with_last_result();
330 							last_result = [asr_result, asr_score, asr_type, asr_alternatives, asr_unstable];
331 							// post current result
332 							call_callback_with_last_result();
333 						} else if (asr_type === result_types.INTERIM){
334 							// its an interim result
335 							// -> save the last result
336 							call_callback_with_last_result();
337 							last_result = [asr_result, asr_score, asr_type, asr_alternatives, asr_unstable];
338 							// post current result
339 							call_callback_with_last_result();
340 						} else if (asr_type === result_types.INTERMEDIATE){
341 							_wait.preparing();
342 							// save the last result and start recognition again
343 							// (callback for intermediate results will be triggered on next RECORDING_BEGIN)
344 							last_result = [asr_result, asr_score, asr_type, asr_alternatives, asr_unstable];
345 
346 							window.plugins.androidSpeechPlugin.recognizeNoEOS(
347 									languageManager.getLanguageConfig(_pluginName),
348 									successCallbackWrapper(currentSuccessCallback),
349 									failureCallbackWrapper(currentFailureCallback),

352 
353 						} else {

364 						}

366 					} else {
367 						// no repeat, there won't be another recording, so call callback right now with previous and current result
368 
369 						_wait.ready();
370 
371 						if (asr_type === result_types.RECORDING_DONE){
372 						} else if (asr_type === result_types.FINAL){
373 						}
374 						call_callback_with_last_result();
375 
376 						if(typeof res !== "undefined") {
377 							if (cb){
378 								cb(asr_result, asr_score, asr_type, asr_alternatives, asr_unstable);

396 				
397 				return (function (res){
398 					var error_code = -1;
399 					var error_msg = "";
400 					var error_suggestion = "";
401 					var error_type = -1;
402 
403 					if (typeof res !== "undefined"){
404 						if(typeof res['error_code'] !== "undefined") {
405

407 						}

410 						}
411 
412 						if (typeof res["suggestion"] !== "undefined"){
413 							error_suggestion = res["suggestion"];
414 						}
415 
416 						if  (typeof res["type"] !== "undefined"){
417 							error_type = res["type"];
418 						}
419 					}
420 
421 					console.error("androidAudioInput ERROR \""+error_msg+"\" (code "+error_code+", type "+error_type+")...");
422 					
423 					_wait.ready();
424 
425 					// TEST: if there is still a pending last result, call successcallback first.
426 					call_callback_with_last_result();
427 					if (repeat === true){
428 						
429 						
430 //								"NETWORK_TIMEOUT":			1, //Nuance: SERVER_CONNECTION
431 //								"NETWORK":					2, //Nuance: SERVER_RETRY
432 //								"AUDIO": 					3, //Nuance: RECOGNIZER
433 //								"SERVER": 					4, //Nuance: VOCALIZER

445 								||	error_code == error_codes_enum.NETWORK_TIMEOUT

484 									
485 								});
486 							}
487 							else {
488 								
489 								//just restart recognition:
490 								restartFunc();
491 							}
492 							
493 						}
494 						//on minor errors that do not stop the recognizer -> do nothing
495 						else if(
496 								error_code == error_codes_enum.CLIENT
497 						){
498 							console.info("androidAudioInput error: "+error_msg+" (code "+error_code+") - continueing ASR process...");
499 						}
500 						// call error callback on "severe" errors

504 //								|| 	error_code == error_codes_enum.SERVER
505 //								|| 	error_code == error_codes_enum.AUDIO

509 						{
510 							
511 							if (cb){
512 								console.warn("androidAudioInput: Calling error callback (" + error_code + ": " + error_msg + ").");
513 								cb(error_msg, error_code, error_suggestion);
514 							}
515 							else {
516 								console.error("androidAudioInput Error: No callback function defined for failure.");
517 							}
518 						}
519 						
520 					}
521 					// "one-time recogintion call", i.e. without intermediate results, that is no repeat-mode requested
522 					//  --> just call error callback:
523 					else {
524 						
525 						// do no repeat, just call errorCallback

527 							console.debug("androidAudioInput: Calling error callback (" + error_code + ").");
528 							cb(error_msg, error_code, error_suggestion);

530 							console.error("androidAudioInput Error: No callback function defined for failure.");
531 						}
