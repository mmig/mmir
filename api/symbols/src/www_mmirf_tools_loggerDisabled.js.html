 30 /**
 31  * A "silent" replacement for the Logger factory.<br>
 32  * Has the same interaface (functions etc.) as {@link mmir.Logging}.
 33  * 
 34  * <p>
 35  * 
 36  * This logging module allows to disable logging completely.
 37  * 
 38  * <p>
 39  * 
 40  * Which logging module is used, can be configured in core
 41  * module in {@link mmir}.
 42  *  
 43  * 
 44  * @example <code>mmir.Logging.create('SomeClass')</code>
 45  * @class DisabledLogging
 46  * @name DisabledLogging
 47  * @exports DisabledLogging as mmir.DisabledLogging
 48  * @category tools
 49  * @static
 50  * 
 51  * @see mmir.Logging
 52  */
 53 function(){
 54 
 55 /**
 56  * logging levels
 57  * 
 58  * 0: verbose
 59  * 1: debug
 60  * 2: info
 61  * 3: warn
 62  * 4: error
 63  * 5: critical
 64  * 6: disabled
 65  */
 66 
 67 
 68 /** NO-OP */
 69 function noop(){}
 70 /** @returns false **/
 71 function deny(){return false;}
 72 
 73 /**
 74  * Constructor-Method of Class DisabledLogger:
 75  * This logger implementation will never print logging output.
 76  * 
 77  * Has the same interface (functions etc) as {@link Logger},
 78  * but <code>isXXX</code> functions will always return <code>false</code>
 79  * and the logging-functions will never do anything (i.e. these are
 80  * <em>no-op</em> functions).
 81  * 
 82  * @constructor
 83  * @name DisabledLogger
 84  */
 85 function Logger(){}
 86 
 87 
 88 Logger.prototype = {//public instance members (basically all NO-OPs)
 89     getLevel : function(){
 90     	//always return "disabled" level:
 91     	return 6;
 92     },
 93     setLevel : noop,
 94     log: noop,
 95     verbose: noop,
 96     debug: noop,
 97     info: noop,
 98     warn: noop,
 99     error: noop,
100     critical: noop,
101     iVerbose: deny,
102     isDebug: deny,
103     isInfo: deny,
104     isWarn: deny,
105     isError: deny,
106     isCritical: deny,
107     isDisabled: deny,
108     v: noop,
109     d: noop,
110     i: noop,
111     w: noop,
112     e: noop,
113     c: noop,
114     isv: deny,
115     isd: deny,
116     isi: deny,
117     isw: deny,
118     ise: deny,
119     isc: deny
120 };
121 
122 /**
123  * @memberOf mmir.DisabledLogging.prototype
124  */
125 var _defaultLogger = new Logger();
126 
127 return {//public API
128 	
129 	/**
130 	 * Will alway return the default logger for this logging module
131 	 * @returns {DisabledLogger}
132 	 * 
133 	 * @memberOf mmir.DisabledLogging.prototype
134 	 */
135     create: function(){
136         return _defaultLogger;
137     },
138     get: function(){
139         return _defaultLogger;
140     },
141     setDefaultLogLevel: noop,
142     getDefaultLogLevel: function(){
143     	return 6;
144     },
145     log: noop,
146     verbose: noop,
147     debug: noop,
148     info: noop,
149     warn: noop,
150     error: noop,
151 //    isDebug: deny,
152 //    isInfo: deny,
153 //    isWarn: deny,
154 //    isError: deny
155     isDisabledLogger: true//special property for identifying the disabled logger-factory
156 };
157     
158 });
