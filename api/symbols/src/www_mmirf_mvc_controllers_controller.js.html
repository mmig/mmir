 45 	/**
 46 	 * The Controller Class is a kind of interface-class which gives access to the methods of a controller and its helper. <br>
 47 	 * Also holds information about views and partials associated with the controller.
 48 	 * 
 49 	 * @constructs Controller
 50 	 * @param {String} name Name of the Controller
 51 	 * @param {Object} jsonDef Information about the controllers views and partials 
 52 	 * @category core
 53 	 */
 54 	function Controller(name, jsonDef){
 55 //	    console.log("controller name " + name);
 56 	    /**
 57 	     * The definition of the controller object, containing all properties and functions of the controller.<br>
 58 	     * A method of the controller can be called via:
 59 
 60 		this.script.methodController(parameter);
 61 
 62 	     * 
 63 	     * @property script
 64 	     * @type Object
 65 	     * @public
 66 	     */
 67 		// this can only be invoked, if a function with the name "name" exists
 68 	    this.script = new window[name]();
 69 
 70 	    /**
 71 	     * The json definition of the views and partials associated with the controller. Also contains paths to controller and its views/partials. 
 72 	     * 
 73 	     * @property def
 74 	     * @type Object
 75 	     * @public
 76 	     */
 77 	    this.def = jsonDef;//new JPath(jsonDef);
 78 
 79 	    /**
 80 	     * The name of the controller. 
 81 	     * 
 82 	     * @property name
 83 	     * @type String
 84 	     * @public
 85 	     */
 86 	    this.name = name;
 87 	    
 88 	    var viewDefs = this.def.views;//.query('views');
 89 
 90 	    /**
 91 	     * An array holding the names of all views associated with the controller.  
 92 	     * 
 93 	     * @property views
 94 	     * @type Array
 95 	     * @public
 96 	     */
 97 	    this.views = new Array();
 98 	    this.parseViews(viewDefs);
 99 
100 	    // parsing the partials and saving the names in an array
101 	    var partialDefs = this.def.partials;//.query('partials');
102 	    
103 
104 	    /**
105 	     * An array holding the names of all partials associated with the controller.  
106 	     * 
107 	     * @property partials
108 	     * @type Array
109 	     * @public
110 	     */
111 	    this.partials = new Array();
112 	    this.parsePartials(partialDefs);
113 
114 
115 	    /**
116 	     * The instance of the with the controller associated helper.  
117 	     * 
118 	     * @property helper
119 	     * @type Object
120 	     * @public
121 	     */
122 	    this.helper;
123 	    
124 	    /**
125 	     * The layout for this controller (if undefined, the default layout should be used)
126 	     * @property layout
127 	     * @type Object
128 	     */
129 	    this.layout = this.def.layout;
130 	}

133 	/**
134 	 * This function loads the helper of the controller - if it exists.
135 	 * 
136 	 * @function loadHelper
137 	 * @param {String} name Name of the Helper to be loaded
138 	 * @param {String} helperPath Path to the helper file to load  
139 	 * @public
140 	 */
141 	Controller.prototype.loadHelper = function(name, helperPath){
142 		var self = this;
143 		
144 		//TODO move check of helper existance to Controller.foundControllersCallBack ?
145 
146 		//determine if there is a helper for the controller:
147 		var path = helperPath;
148 		var fileName = path;
149 		var lastPathSeparatorIndex = path.lastIndexOf('/');
150 		if(lastPathSeparatorIndex !== -1){
151 			path = path.substring(0,lastPathSeparatorIndex);
152 			fileName = fileName.substring( lastPathSeparatorIndex + 1 );
153 		}
154 		//get contents of the helper directory:
155 		var dirContents = commonUtils.getDirectoryContents(path);
156 		if(!dirContents){
157 			logger.warn('Could not determine contents for directory "'+path+'"');
158 			return; ////////////////////// EARLY EXIT //////////////////////////////
159 		}
160 		else if(! commonUtils.isArray(dirContents) || dirContents.length < 1){
161 			logger.warn('Invalid information for contents of directory "'+path+'": '+dirContents);
162 			return; ////////////////////// EARLY EXIT //////////////////////////////
163 		}
164 		
165 		//check, if there is an implementation file for this helper:
166 		var helperIsSpecified = false;
167 		for(var i=0, size= dirContents.length; i < size; ++i){
168 			if(dirContents[i] === fileName){
169 				helperIsSpecified = true;
170 				break;
171 			}
172 		}
173 		
174 		if( ! helperIsSpecified){
175 			if(logger.isVerbose()) logger.v("[HELPER] no helper available (not implemented) at '"+ helperPath+"'");
176 			return; ////////////////////// EARLY EXIT //////////////////////////////
177 		}
178 		
179 		//if there is a file: load the helper
180 		commonUtils.getLocalScript(helperPath, function(data, textStatus, jqxhr){
181 			
182 				if(logger.isVerbose()) logger.v("[HELPER] load "+ helperPath);//debug
183 				
184 				self.helper =   new Helper(self, name);//new window["GoogleMapHelper"]();
185 			},
186 			function(exception) {
187 				// print out an error message
188 				logger.error("[WARN] Could not load helper -> '"+ helperPath + "'", exception); //failure
189 			}
190 		);
191 	};

194 	/**
195 	 * This function performs an action of a controller - which is represented by this instance of the Controller <br>
196 	 * class - by calling the method from the corresponding controller, e.g. assets/www/controllers/application.js   
197 	 * 
198 	 * @function perform
199 	 * @param {String} actionName Name of the method to be executed
200 	 * @param {Object} data Data to pass to the method of the controller as argument
201 	 * @returns {Object} The return value of the executed method 
202 	 * @public
203 	 */
204 	Controller.prototype.perform = function(actionName, data){
205 		
206 		if(logger.isVerbose()) logger.v("should perform '" + actionName + "' of '" + this.name + "'"+ ((typeof data !== 'undefined' && data !== null)? " with data: "+JSON.stringify(data): ""));//debug
207 		
208 	    if(arguments.length > 2){
209 	    	return this.script[actionName](data, arguments[2]);
210 		}
211 		else {
212 			return this.script[actionName](data);
213 		}
214 	};
215 
216 	/**
217 	 * 
218 	 * This function performs an action of a controller, but only if an action with this name exists; otherwise nothing is done.
219 	 * 
220 	 * In difference to perform(..), the method does not trigger an ERROR, if the action does not exist / is not implemented.
221 	 * As a consequence, this method refers to "optionally" implemented functions, whereas perform(..) refers to mandatory functions.
222 	 * 
223 	 * @function perform
224 	 * @param {String} actionName Name of the method to be executed
225 	 * @param {Object} data Data to pass to the method of the controller as argument
226 	 * @returns {Object} The return value of the executed method 
227 	 * @public
228 	 */
229 	Controller.prototype.performIfPresent = function(actionName, data){
230 		if(typeof this.script[actionName] === 'function'){
231 		    
232 			if(logger.isVerbose()) logger.v("performing '" + actionName + "' of '" + this.name + "'"+ ((typeof data !== 'undefined' && data !== null)? " with data: "+JSON.stringify(data): ""));//debug
233 		    
234 		    return this.perform.apply(this, arguments);
235 		    
236 		} else if(typeof this.script[actionName] !== 'undefined'){
237 			if(logger.isVerbose()) logger.info("could not perform '" + actionName + "' of '" + this.name + "'"+ ((typeof data !== 'undefined' && data !== null)? " with data: "+JSON.stringify(data): "")+": no function ("+typeof this.script[actionName]+")");//debug
238 		} else {
239 			if(logger.isVerbose()) logger.debug("could not perform '" + actionName + "' of '" + this.name + "'"+ ((typeof data !== 'undefined' && data !== null)? " with data: "+JSON.stringify(data): "")+": not implemented (undefined)");//debug
240 		}
241 	};

244 	/**
245 	 * This function performs a helper action of a controller by calling the appropriate method<br>
246 	 * {@link Helper#perform} of the instance of the helper class associated with the controller.
247 	 * 
248 	 * @function performHelper
249 	 * @param {String} actionName Name of the helper method to be executed
250 	 * @param {Object} data Data to pass to the helper method as argument
251 	 * @returns {Object} The return value of the executed method 
252 	 * @public
253 	 */
254 	Controller.prototype.performHelper = function(actionName, data){
255 		if(arguments.length > 2){
256 			return this.helper.perform(actionName, data, arguments[2]);
257 		}
258 		else {
259 			return this.helper.perform(actionName, data);
260 		}
261 	};

277 	 * Stores all names of the views of the controller by iterating over the array of the views definition.<br>
278 	 * This function is called by the constructor of the {@link mmir.Controller} class.
279 	 * 
280 	 * @function parseViews
281 	 * @param {Array} viewDefs Array of the json-definition of the controllers views - containing name of the views and their corresponding path to the js-files
282 	 * @public
283 	 */
284 	Controller.prototype.parseViews = function(viewDefs){
285 		
286 		for(var i=0, size = viewDefs.length; i < size; ++i){
287 			this.views.push(viewDefs[i].name);
288 		}
289 		
290 	};

298 	 * @param {Array} partialDefs Array of the json-definition of the controllers partials - containing name of the partials and their corresponding path to the js-files
299 	 * @public

302
