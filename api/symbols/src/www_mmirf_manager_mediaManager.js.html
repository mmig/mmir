195     	/**
196     	 * map of listeners: 
197     	 * 		event(String) -> listener(Function)
198     	 * 
199 		 * @private
200     	 * @memberOf MediaManager.prototype
201     	 */
202     	var listener = new Dictionary();
203     	
204     	/**
205     	 * map of listener-observers:
206     	 *  observers get notified if a listener for event X gets added/removed
207     	 * 
208 		 * @private
209     	 * @memberOf MediaManager.prototype
210     	 */
211     	var listenerObserver = new Dictionary();

213 		/** 
214 		 * exported as addListener() and on()
215 		 * 
216 		 * @private
217 		 * @memberOf MediaManager.prototype
218 		 */
219     	var addListenerImpl = function(eventName, eventHandler){
220 			var list = listener.get(eventName);
221 			if(!list){
222 				list = [eventHandler];
223 				listener.put(eventName, list);
224 			}
225 			else {
226 				list.push(eventHandler);
227 			}
228 
229 			//notify listener-observers for this event-type
230 			this._notifyObservers(eventName, 'added', eventHandler);
231 		};
232 		/**
233 		 * exported as removeListener() and off()
234 		 *  
235 		 * @private
236 		 * @memberOf MediaManager.prototype
237 		 */
238     	var removeListenerImpl = function(eventName, eventHandler){
239 			var isRemoved = false;
240 			var list = listener.get(eventName);
241 			if(list){
242 				var size = list.length;
243 				for(var i = size - 1; i >= 0; --i){
244 					if(list[i] ===  eventHandler){
245 						
246 						//move all handlers after i by 1 index-position ahead:
247 						for(var j = size - 1; j > i; --j){
248 							list[j-1] = list[j];
249 						}
250 						//remove last array-element
251 						list.splice(size-1, 1);
252 						
253 						//notify listener-observers for this event-type
254 						this._notifyObservers(eventName, 'removed', eventHandler);
255 						
256 						isRemoved = true;
257 						break;
258 					}
259 				}
260 			}
261 			return isRemoved;
262 		};

524     	    	 * 
525     	    	 * TODO list events that the default media-plugins support
526     	    	 *   * "miclevelchanged": fired by AudioInput plugins that support querying the microphone (audio input) levels
527     	    	 * 
528     	    	 * A plugin can tigger / fire events using the helper {@link #_fireEvent}
529     	    	 * of the MediaManager.
530     	    	 * 
531     	    	 * 
532     	    	 * Media plugins may observe registration / removal of listeners

535     	    	 * 
536     	    	 *

549     	    	 * <code>actionType "removed"</code>, if the handler
550     	    	 * was actually removed.
551     	    	 * 
552     	    	 * @param {String} eventName
553     	    	 * @param {Function} eventHandler
554     	    	 * 
555     	    	 * @returns {Boolean}
556     	    	 * 		<code>true</code> if the handler function was actually 
557     	    	 * 		removed, and <code>false</code> otherwise.
558     	    	 * 
559     	    	 * @function
560     	    	 */

566     			, on:addListenerImpl

572     			/**
573     			 * Get list of registered listeners / handlers for an event.
574     			 * 
575     			 * @returns {Array[Function]} of event-handlers. 
576     			 * 				Empty, if there are no event handlers for eventName
577     			 */
578     			, getListeners: function(eventName){
579     				var list = listener.get(eventName);
580     				if(list && list.length){
581     					//return copy of listener-list
582     					return list.slice(0,list.length);
583     				}
584     				return [];
585     			}
586     			/**
587     			 * Check if at least one listener / handler is  registered for the event.

590     			 * 					  for eventName; otherwise <code>false</code>.
591     			 */
592     			, hasListeners: function(eventName){
593     				var list = listener.get(eventName);
594     				return list && list.length > 0;
595     			}
596     			/**
597     			 * Helper for firing / triggering an event.
598     			 * This should only be used by media plugins (that handle the eventName).
599     			 * 
600     	    	 * @param {String} eventName
601     	    	 * @param {Array} argsArray
602     	    	 * 					the list of arguments with which the event-handlers
603     	    	 * 					will be called.
604     	    	 * @protected
605     			 */
606     			, _fireEvent: function(eventName, argsArray){
607     				var list = listener.get(eventName);
608     				if(list && list.length){
609     					for(var i=0, size = list.length; i < size; ++i){
610     						list[i].apply(this, argsArray);
611     					}
612     				}
613     			}
614     			/** 
615     			 * Helper for notifying listener-observers about changes (adding/removing listeners).
616     			 * This should only be used by media plugins (that handle the eventName).
617     			 * 
618     			 * @param {String} eventName
619     			 * @param {String} actionType
620     			 * 					the change-type that occurred for the event/event-handler:
621     			 * 					one of <code>["added" | "removed"]</code>.
622     	    	 * @param {Function} eventHandler
623     	    	 * 					the event-handler function that has changed.
624     	    	 * 
625     			 * @protected
626     			 */
627     			, _notifyObservers: function(eventName, actionType, eventHandler){//actionType: one of "added" | "removed"
628     				var list = listenerObserver.get(eventName);
629     				if(list && list.length){
630     					for(var i=0, size = list.length; i < size; ++i){
631     						list[i](actionType,eventHandler);
632     					}
633     				}
634     			}
635     			/**
636     			 * Add an observer for registration / removal of event-handler.
637     			 * 
638     			 * The observer gets notified,when handlers are registered / removed for the event.
639     			 * 
640     			 * The observer-callback function will be called with the following
641     			 * arguments
642     			 * 
643     			 * <code>(eventName, ACTION_TYPE, eventHandler)</code>
644     			 * where
645     			 * <ul>
646     			 *  <li>eventName: String the name of the event that should be observed</li>
647     			 *  <li>ACTION_TYPE: the type of action: "added" if the handler was 
648     			 *      registered for the event, "removed" if the the handler was removed
649     			 *  </li>
650     			 *  <li>eventHandler: the handler function that was registered or removed</li>
651     			 * </ul> 
652     			 * 
653     	    	 * @param {String} eventName
654     	    	 * @param {Function} observerCallback
655     			 */
656     			, _addListenerObserver: function(eventName, observerCallback){
657     				var list = listenerObserver.get(eventName);
658     				if(!list){
659     					list = [observerCallback];
660     					listenerObserver.put(eventName, list);
661     				}
662     				else {
663     					list.push(observerCallback);
664     				}
665     			}
666     			
667     			, _removeListenerObserver: function(eventName, observerCallback){
668     				var isRemoved = false;
669     				var list = listenerObserver.get(eventName);
670     				if(list){
671     					var size = list.length;
672     					for(var i = size - 1; i >= 0; --i){
673     						if(list[i] ===  observerCallback){
674     							
675     							//move all handlers after i by 1 index-position ahead:
676     							for(var j = size - 1; j > i; --j){
677     								list[j-1] = list[j];
678     							}
679     							//remove last array-element
680     							list.splice(size-1, 1);
681     							
682     							isRemoved = true;
683     							break;
684     						}
685     					}
686     				}
687     				return isRemoved;
688     			}
