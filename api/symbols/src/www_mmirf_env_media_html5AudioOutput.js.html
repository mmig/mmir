 28 newMediaPlugin = {
 29 
 30 		/**  @memberOf Html5AudioOutput# */
 31 		initialize: function(callBack, mediaManagerInstance){
 32 			
 33 			/**  @memberOf Html5AudioOutput# */
 34 			var _pluginName = 'html5AudioOutput';
 35 
 36 			//invoke the passed-in initializer-callback and export the public functions:
 37 			callBack({
 38 				/**
 39 				 * @public
 40 				 * @memberOf Html5AudioOutput.prototype
 41 				 * @see mmir.MediaManager#playWAV
 42 				 */
 43 				playWAV: function(blob, successCallback, failureCallback){
 44 					try {
 45 						blobURL = window.webkitURL.createObjectURL(blob);
 46 						var my_audio = new Audio(blobURL,null,failureCallback);
 47 						if(successCallback){
 48 							my_audio.addEventListener('ended', successCallack, false);
 49 						}
 50 						my_audio.play();
 51 					} catch (e){
 52 						if(failureCallBack){
 53 							failureCallBack(e);
 54 						}
 55 					}
 56 				},
 57 				/**
 58 				 * @public
 59 				 * @memberOf Html5AudioOutput.prototype
 60 				 * @see mmir.MediaManager#playURL
 61 				 */
 62 				playURL: function(url, onEnd, failureCallBack, successCallback){
 63 					try {
 64 
 65 						var my_media = new Audio(url,null,failureCallback);
 66 
 67 						if(successCallback){
 68 							my_media.addEventListener('ended', onEnd, false);
 69 							my_media.addEventListener('canplay', successCallback, false);
 70 						}
 71 						my_media.play();
 72 					} catch (e){
 73 						if(failureCallback){
 74 							failureCallback(e);
 75 						}
 76 					}
 77 				},
 78 				/**
 79 				 * @public
 80 				 * @memberOf Html5AudioOutput.prototype
 81 				 * @see mmir.MediaManager#getURLAsAudio
 82 				 */
 83 				getURLAsAudio: function(url, onEnd,  failureCallback, successCallback){
 84 					try {
 85 
 86 						/**
 87 						 * @private
 88 						 * @memberOf AudioHtml5Impl#
 89 						 */
 90 						var enabled = true;
 91 						/**
 92 						 * @private
 93 						 * @memberOf AudioHtml5Impl#
 94 						 */
 95 						var ready = false;
 96 						/**
 97 						 * @private
 98 						 * @memberOf AudioHtml5Impl#
 99 						 */
100 						var my_media = new Audio(url,null,failureCallback);
101 
102 						my_media.addEventListener('ended', function(){
103 
104 							//only proceed if we have a media-object (may have already been released)
105 							if(enabled & mediaImpl){
106 								mediaImpl.stop();
107 							}
108 							if (onEnd){
109 								onEnd.apply(mediaImpl, arguments);
110 							}
111 						}, false);
112 						
113 						/**
114 						 * @private
115 						 * @memberOf AudioHtml5Impl#
116 						 */
117 						var canPlayCallback = function(){
118 							ready = true;
119 //							console.log("sound is ready!");
120 
121 							//FIX: remove this listener after first invocation 
122 							//     (this is meant as "on-init" listener, but "canplay" 
123 							//      may be triggered multiple times during the lifetime of the audio object).
124 							this.removeEventListener('canplay', canPlayCallback);
125 
126 							if (enabled && successCallback){
127 								successCallback.apply(mediaImpl, arguments);
128 							}
129 						};
130 						my_media.addEventListener('canplay', canPlayCallback, false);
131 
132 						/**
133 						 * The Audio abstraction that is returned by {@link mmir.MediaManager#getURLAsAudio}.
134 						 * 
135 						 * <p>
136 						 * NOTE: when an audio object is not used anymore, its {@link #release} method should
137 						 * 		 be called.
138 						 * 
139 						 * <p>
140 						 * This is the same interface as {@link AudioCordovaImpl}.
141 						 * 
142 						 * @class
143 						 * @name AudioHtml5Impl
144 						 * @public
145 						 */
146 						var mediaImpl = {
147 								/**
148 								 * Play audio.
149 								 * 
150 								 * @public
151 								 * @name play
152 								 * @function
153 								 * @memberOf AudioHtml5Impl.prototype
154 								 */
155 								play: function(){
156 									if (enabled){
157 										// if (ready){
158 										my_media.play();
159 										// } else {
160 										//	 my_media.addEventListener('canplay', my_media.play, false);
161 										// }
162 
163 									};
164 								},
165 								/**
166 								 * Stop playing audio.
167 								 * 
168 								 * @public
169 								 * @name stop
170 								 * @function
171 								 * @memberOf AudioHtml5Impl.prototype
172 								 */
173 								stop: function(){
174 									if(enabled){
175 										if(my_media.stop){
176 											//TODO really we should check first, if the audio is playing...
177 											my_media.stop();
178 										}
179 										else {
180 											my_media.pause();
181 											//apparently, browser treat pause() differently: Chrome pauses, Firefox seems to stop... -> add try-catch-block in case, pause was really stop...
182 											try{
183 												my_media.currentTime=0;
184 
185 												//HACK: for non-seekable audio in Chrome
186 												//      -> if currentTime cannot be set, we need to re-load the data
187 												//         (otherwise, the audio cannot be re-played!) 
188 												if(my_media.currentTime != 0){
189 													my_media.load();
190 												}
191 											}catch(e){};
192 										}
193 									}
194 								},
195 								/**
196 								 * Enable audio (should only be used internally).
197 								 * 
198 								 * @public
199 								 * @name enable
200 								 * @function
201 								 * @memberOf AudioHtml5Impl.prototype
202 								 */
203 								enable: function(){
204 									if(my_media != null){
205 										enabled = true;
206 									}
207 									return enabled;
208 								},
209 								/**
210 								 * Disable audio (should only be used internally).
211 								 * 
212 								 * @public
213 								 * @name disable
214 								 * @function
215 								 * @memberOf AudioHtml5Impl.prototype
216 								 */
217 								disable: function(){
218 									if(enabled){
219 										this.stop();
220 										enabled = false;
221 									}
222 								},
223 								/**
224 								 * Release audio: should be called when the audio
225 								 * file is not used any more.
226 								 * 
227 								 * @public
228 								 * @name release
229 								 * @function
230 								 * @memberOf AudioHtml5Impl.prototype
231 								 */
232 								release: function(){
233 									if(enabled && ! this.isPaused()){
234 										this.stop();
235 									}
236 									enabled= false;
237 									my_media=null;
238 								},
239 								/**
240 								 * Set the volume of this audio file
241 								 * 
242 								 * @param {Number} value
243 								 * 			the new value for the volume:
244 								 * 			a number between [0.0, 1.0]
245 								 * 
246 								 * @public
247 								 * @name setVolume
248 								 * @function
249 								 * @memberOf AudioHtml5Impl.prototype
250 								 */
251 								setVolume: function(value){
252 									if(my_media){
253 										my_media.volume = value;
254 									}
255 								},
256 								/**
257 								 * Get the duration of the audio file
258 								 * 
259 								 * @returns {Number} the duration in MS (or -1 if unknown)
260 								 * 
261 								 * @public
262 								 * @name getDuration
263 								 * @function
264 								 * @memberOf AudioHtml5Impl.prototype
265 								 */
266 								getDuration: function(){
267 									if(my_media){
268 										return my_media.duration;
269 									}
270 									return -1;
271 								},
272 								/**
273 								 * Check if audio is currently paused.
274 								 * 
275 								 * NOTE: "paused" is a different status than "stopped".
276 								 * 
277 								 * @returns {Boolean} TRUE if paused, FALSE otherwise
278 								 * 
279 								 * @public
280 								 * @name isPaused
281 								 * @function
282 								 * @memberOf AudioHtml5Impl.prototype
283 								 */
284 								isPaused: function(){
285 									if(my_media){
286 										return my_media.paused;
287 									}
288 									return false;
289 								},
290 								/**
291 								 * Check if audio is currently enabled
292 								 * 
293 								 * @returns {Boolean} TRUE if enabled
294 								 * 
295 								 * @public
296 								 * @name isEnabled
297 								 * @function
298 								 * @memberOf AudioHtml5Impl.prototype
299 								 */
300 								isEnabled: function(){
301 									return enabled;
302 								}
303 						};
304 
305 						return mediaImpl;
306 
307 					} catch (e){
308 						if(failureCallback){
309 							failureCallback(e);
310 						}
311 					}
312 				}
313 			});
314 		}
315 };
