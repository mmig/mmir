1 (
  2 /**
  3  * Main module / namespace for the MMIR framework.
  4  * 
  5  * On initialization, a global module <code>window.mmir</code> is created.
  6  * 
  7  * If called multiple times, the existing module instance is returned.
  8  * 
  9  * If a function <code>require</code> exists, the module tries to registers itself
 10  * according to the <em>RequireJS</em> interface (using the default as its module name, i.e. "core").
 11  * 
 12  * @name mmir
 13  * @export initMmir as mmir
 14  * @class
 15  * 
 16  * @returns the module instance <code>mmir</code>
 17  * 
 18  */
 19 function initMmir() {
 20     
 21 	if(window.mmir){
 22 		if(typeof define === 'function'){
 23 			define(function(){ return window.mmir; });
 24 		}
 25 		return window.mmir;
 26 	}
 27 	
 28     
 29     //STATE: state variable for indicating "doc is already loaded" (this needs to be set/reset manually)
 30 	var _isReady = false;
 31 	var _funcList = [];
 32 	function dequeue () { return _funcList.shift(); };
 33 	function isEmpty () { return _funcList.length === 0; };
 34 	//param func OPTIONAL
 35 	//			if func is present, func will be used instead of dequeueing a callback from the queue
 36 	function deqExex (func) {
 37 		if(!func){
 38 			func = dequeue();
 39 		}
 40 
 41 		//run function in context of the document (whith jQuery as argument)
 42 		func.call(mmir);
 43 	};
 44 	
 45 	//STATE: state variable for indicating "configs for requirejs are already applied"
 46 	var _isApplied = false;
 47 	var _configList = [];
 48 	function applyConfigs(){
 49 		
 50 		if(typeof require === 'undefined'){
 51 			return;
 52 		}
 53 		
 54 		
 55 		_isApplied = true;
 56 		while(_configList.length > 0){
 57 			require.config( _configList.shift() );
 58 		}
 59 	}
 60 	
 61 	var mmir = {
 62 			
 63 			setInitialized : function() {
 64 				
 65 				_isReady = true;
 66 
 67 				//apply configurations to requirejs instance:
 68 				applyConfigs();
 69 				
 70 				//execute all callbacks in queue
 71 				while(!isEmpty()){
 72 					deqExex();
 73 				}
 74 			},
 75 			
 76 			/**
 77 			 * Register callbacks for initialization-event of framework.
 78 			 * 
 79 			 * If used after framework has been initialized, the callback is invoked immediately.
 80 			 * 
 81 			 * @memberOf mmir
 82 			 * @name ready
 83 			 * @function
 84 			 * @public
 85 			 * 
 86 			 * @param {Function} func
 87 			 * 				callback Function that will be triggered when the framework has been initialized 
 88 			 */
 89 			ready: function(func) {
 90 		
 91 				//SPECIAL MODE: if already active, execute the callback 
 92 				//				(if queue is not empty yet: queue function call in order to preserve the execution ordering)
 93 				if(_isReady && ! isEmpty()){
 94 					deqExec(func);
 95 				}
 96 				else {
 97 					_funcList.push(func);
 98 				}
 99 			},
100 			
101 			/**
102 			 * Set options / settings for overwriting the default
103 			 * configuration for RequireJS:
104 			 * 
105 			 * <br>
106 			 * Options / configurations that are added by this
107 			 * method will overwrite settings specified in
108 			 * <code>mainConfig.js</code>.
109 			 * 
110 			 * <p>
111 			 * NOTE: the options added here will be applied in the order
112 			 *       the were added, i.e. if a later option specifies
113 			 *       settings that were already set by a previous call,
114 			 *       then these later options will overwrite the earlier
115 			 *       ones.
116 			 * 
117 			 * @memberOf mmir
118 			 * @name config
119 			 * @function
120 			 * @param {PlainObject} options
121 			 * 			options for RequireJS
122 			 * @public
123 			 */
124 			config: function(options){
125 				if(_isApplied && typeof require !== 'undefined'){
126 					require.config(options);
127 				}
128 				else {
129 					_configList.push(options);
130 				}
131 			},
132 			
133 			/**
134 			 * Applies settings that were added via
135 			 * {@link #config}.
136 			 * 
137 			 * <p>
138 			 * WARNING: use this only, if you know what
139 			 *          you are doing -- normally this
140 			 *          functions is only called once
141 			 *          during initialization by the
142 			 *          framework, after the default
143 			 *          configuration settings for 
144 			 *          RequireJS in <code>mainConfig.js</code>
145 			 *          were applied.
146 			 * 
147 			 * @memberOf mmir
148 			 * @name applyConfigs
149 			 * @function
150 			 * @private this is a semi-private function that 
151 			 *          should only be used by the initialization
152 			 *          process.
153 			 */
154 			applyConfig: applyConfigs,
155 			
156 			/**
157 			 * The name / ID of the RequireJS module that will
158 			 * be loaded, after the configuration in
159 			 * <code>mainConfig.js</code> was applied.
160 			 * 
161 			 * <p>
162 			 * This module should first start-up the framework and
163 			 * then signal the application (via {@link #setInitialized}
164 			 * that it is ready to be used, i.e. fully initialized now.
165 			 * 
166 			 * <p>
167 			 * NOTE: If set to <code>undefined</code>, no module will be 
168 			 * loaded after configuration in <code>mainConfig.js</code>
169 			 * was applied.
170 			 * 
171 			 * @memberOf mmir
172 			 * @name startModule
173 			 * @property
174 			 * @type String
175 			 * @default {String} "main" will load the module specified in main.js
176 			 * @public
177 			 */
178 			startModule: 'main',
179 
180 			/**
181 			 * Name / ID / load-path (requirejs) for the module
182 			 * that handles the views (i.e. "rendering" that is
183 			 * change from one view to the next).
184 			 * 			 
185 			 * @memberOf mmir
186 			 * @name viewEngine
187 			 * @property
188 			 * @type String
189 			 * @default "jqmViewEngine" will load the default view-engine that uses jQuery Mobile
190 			 * @public
191 			 */
192 			viewEngine: 'jqmViewEngine',
193 			
194 			
195 			/**
196 			 * Property for enabling / disabling logging:
197 			 * if set to <code>true</code> (or omitted), the default Logger implementation <code>tools/logger.js</code>
198 			 * will be loaded as "logger" module.
199 			 * 
200 			 * If set to <code>false</code> the "dummy" Logger implementation <code>tools/loggerDisabled.js</code> will
201 			 * be loaded as "logger" module which essentially will create no logging output.
202 			 * 			 
203 			 * @memberOf mmir
204 			 * @name debug
205 			 * @property
206 			 * @type Boolean
207 			 * @default true
208 			 * @public
209 			 * 
210 			 * @see #logLevel
211 			 */
212 			debug: true,
213 			
214 			/**
215 			 * Property for the log-level of the Logger module:
216 			 * if set, and property <code>debug</code> is <code>true</code>, then the logger module
217 			 * will use the log-level as default log-level.
218 			 * 
219 			 * If omitted, the Logger's implementation defaults will be used.
220 			 * 
221 			 * If set, the property must be either a Number or a String with one of the following values:
222 			 * 
223 			 * 0: "verbose"
224 			 * 1: "debug"
225 			 * 2: "info"
226 			 * 3: "warn"
227 			 * 4: "error"
228 			 * 5: "critical"
229 			 * 6: "disabled"
230 			 * 
231 			 * NOTE: if you want to disable logging completely, use {@link #debug}.
232 			 *       Setting the logLevel to "disabled" will still allow specific module's to create logging output
233 			 *       (if their log-level is set appropriately)
234 			 * 			 
235 			 * @memberOf mmir
236 			 * @name logLevel
237 			 * @property
238 			 * @type Integer | String
239 			 * @default "debug"
240 			 * @public
241 			 * 
242 			 * @see #debug
243 			 */
244 			logLevel: 'debug',
245 			
246 			/**
247 			 * Property for enabling / disabling trace output in the Logger module:
248 			 * if set to <code>true</code>, and property <code>debug</code> is <code>true</code>, then 
249 			 * the logger module will print a stack-trace for each log-message.
250 			 * 
251 			 * If set to a configuration object:
252 			 * <pre>
253 			 * {
254 			 * 		"trace": [true | false],	//same as the Boolean primitive for logTrace
255 			 * 		"depth": ["full" | other]	//OPTIONAL: if "full" then the complete stack trace is printed,
256 			 * 									// otherwise only the first stack-entry (i.e. the calling function)
257 			 * 									// is printed.
258 			 * 									//DEFAULT: other
259 			 * }
260 			 * </pre>
261 			 * 
262 			 * i.e. <code>{trace: true}</code> would be the same as using <code>true</code> (or omitting this property).
263 			 * 
264 			 * 
265 			 * The default value (also if omitted!) is <code>true</code>.
266 			 * 			 
267 			 * @memberOf mmir
268 			 * @name logTrace
269 			 * @property
270 			 * @type Boolean | PlainObject
271 			 * @default true
272 			 * @public
273 			 * 
274 			 * @see #debug
275 			 * @see #logLevel
276 			 */
277 			logTrace: true	//{trace: true, depth: 'full'}
278 	};
279 	
280 	if(typeof define === 'function'){
281 		define(function(){ return mmir; });
282 	}
283 	
284 	//export as global namespace:
285 	window.mmir = mmir;
286 	
287 	return mmir;
288 }());
