 43 /**
 44  * The ContentElement represents "content" parts of a view; it may itself contain one or more ContentElements.
 45  * 
 46  * This class holds the name of the content-field (used via the yield-tag in the layouts: content, header, footer, dialogs, ...)
 47  * and its definition as HTML-String.
 48  * 
 49  * @constructs ContentElement
 50  * @public
 51  * 
 52  * @param {Array|Object} group
 53  * 				 an array or object with properties <code>name</code> {String}, and <code>content</code> {String}
 54  * @param {Object} view 
 55  * 				the view that owns this ContentElement-element 
 56  * @param {mmir.parser.ParserUtils} parser 
 57  * 				for the the content (optional) if supplied this object must have a function <code>parse({String})</code> (see templateParseUtil)
 58  * @param {mmir.parser.RenderUtils} renderer
 59  * 				 for the the content (optional) if supplied, a <code>parser</code> must also be supplied; the renderer must have a function <code>parse({String})</code> (see templateRenderUtil)
 60  * 
 61  */

196 		
197 		var err = Error.apply(this, arguments);
198 		err.name = this.name = 'ScriptEvalError';

200 		this.stack = err.stack;
201 		this.message = err.message;
202 
203 		if(typeof this.stack === 'string'){
204 			//remove first line of stack (which would only contain a reference to this constructor)
205 			this.stack = this.stack.replace(/^.*?\r?\n/, this.name + ': ');
206 		}
207 		
208 		var offset = 0;
209 //		if(parsingElement.contentOffset){
210 //			console.error('elem.offset: '+parsingElement.contentOffset);
211 ////			offset = parsingElement.contentOffset;
212 //		}
213 		offset += contentElement.getOffset();
214 		
215 		this.offset = offset;
216 		
217 		this.start = this.offset + parsingElement.getStart();
218 		this.end = this.offset + parsingElement.getEnd();
219 		
220 		this.errorDetails = parser_context.parserCreatePrintMessage(
221 				'ContentElement.ScriptEvalError: Error evaluating script '
222 					+JSON.stringify(strScript)
223 					+' for ' + parsingElement.getTypeName()
224 					+' element:\n', 
225 				this.message,
226 				this
227 		);
228 		
229 		this.getDetails = function(){
230 			return this.errorDetails;
231 		};
232 		
233 		return this;

527 			} catch (err){
528 				var error = new ScriptEvalError(err, preparedJSCode,  this, forElement);
529 				//attach a dummy function that prints the error each time it is invoked:
530 				forElement.forIncrementEval = function(){ console.error(error.getDetails()); };
531 				//... and print the error now, too:
532 				console.error(error.getDetails());
533 			}
534 		}
535 	}
536 	
537 	//recursively parse content-fields:
538 	for(i=0, size = all.length; i < size; ++i){
539 		subContentElement = all[i];
540 		
541 		if(typeof subContentElement.scriptContent === 'string'){
542 			
543 			var isScriptStatement = subContentElement.isScriptStatement();
544 			
545 			var parsedJS; 
546 			if(isScriptStatement===true){
547 				parsedJS = parser.parseJS(
548 						subContentElement.scriptContent, 
549 						'embeddedStatementTail',
550 						this//TODO supply/implement more accurate error-localization: this is indeed wrong, since it is not the view-defintion, but: this.definition=<view's contentFor>, then helperElement.rawResult and .dataPos contain the information, where exactly this element is located...
551 						, subContentElement.getStart() + this.getOffset() + '@('.length
552 				);
553 			}
554 			else {
555 				parsedJS = parser.parseJS(
556 						subContentElement.scriptContent,
557 						this//TODO supply/implement more accurate error-localization: this is indeed wrong, since it is not the view-defintion, but: this.definition=<view's contentFor>, then helperElement.rawResult and .dataPos contain the information, where exactly this element is located...
558 						, subContentElement.getStart() + this.getOffset() + '@{'.length
559 				);
560 			}
561 			
562 			subContentElement.scriptContent = parsedJS;
563 			
564 			preparedJSCode = renderer.renderJS(parsedJS.rawTemplateText, parsedJS.varReferences); 
565 			
566 			if(isScriptStatement===true){
567 				preparedJSCode = 'return ('+preparedJSCode+');';
568 			}
569 			
570 			try{
571 				subContentElement.scriptEval = createJSEvalFunction(preparedJSCode, 'scriptEval');
572 			} catch (err){
573 				var error = new ScriptEvalError(err, preparedJSCode,  this, subContentElement);
574 				//attach a dummy function that prints the error each time it is invoked:
575 				subContentElement.scriptEval = function(){ console.error(error.getDetails()); };
576 				//... and print the error now, too:
577 				console.error(error.getDetails());
578 			}
579 			
580 			this.internalHasDynamicContent = true;
581 		}
582 		
583 		if(typeof subContentElement.content === 'string'){
584 			
585 			subContentElement.content = new ContentElement({
586 					name: SUB_ELEMENT_NAME, 
587 					content: subContentElement.content,
588 					offset: this.getOffset() + subContentElement.contentOffset
589 				}, view, parser, renderer
590 			);
591 			
592 			this.internalHasDynamicContent = this.internalHasDynamicContent || subContentElement.content.hasDynamicContent();
593 		}
594 		
595 		//IF-elements can have an additional ELSE-content field:
596 		if(subContentElement.hasElse() && typeof subContentElement.elseContent.content === 'string'){
597 			
598 			subContentElement.elseContent.content = new ContentElement({
599 					name: SUB_ELEMENT_NAME,
600 					content: subContentElement.elseContent.content,
601 					offset: this.getOffset() + subContentElement.elseContent.contentOffset
602 				}, view, parser, renderer
603 			);
604 
605 			this.internalHasDynamicContent = this.internalHasDynamicContent || subContentElement.elseContent.content.hasDynamicContent();
606 		}
607 	}
608 	
609     return this;
610 }
