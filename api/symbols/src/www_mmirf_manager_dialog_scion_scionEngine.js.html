 10     /**
 11      * An array containing all states active.
 12      * 
 13      * @property statesActive
 14      * @type Array
 15      * @private
 16      */
 17     var statesActive = new Array();
 18 
 19     var newInstance = function( scion, scionUtil, instanceContext ) {
 20     	
 21         var _interpreter = null;
 22 
 23         var load = function(){
 24 
 25             var _url = this.arguments || this.doc
 26             	, _defer = $.Deferred();
 27             
 28             if (typeof _url === 'undefined') {
 29             	instanceContext._log.error('URL is missing!');
 30                 return;
 31             }
 32 
 33             var self = this;
 34             
 35             scion.urlToModel(_url, function(err, model) {
 36 
 37                 if (err) {
 38                 	
 39                 	var url = '';
 40                 	var printError = function(){
 41                 		console.error('error for SCXML model at ',_url, ': ',
 42                 				(url? 'could not load "' + url + '" ' : ''),
 43                 				(err.statusText? ': ' + err.statusText : ''),
 44                 				err,
 45                 				model
 46                 		);
 47                 	};
 48                 	if(err.always) err.always(function(){url = this.url; printError();});
 49                 	else printError();
 50                 	
 51 //                    alert('SCXML is not valid!');
 52                 	_defer.fail(err);
 53                     return;
 54                 }
 55 
 56                 // instantiate the interpreter
 57                 _interpreter = new scion.SCXML(model);
 58 
 59                 // listener for transitions / state-changes:
 60                 var listener = {
 61                     onEntry : function(stateName) {
 62                         statesActive.push(stateName);
 63                         if (instanceContext._log.isDebug()) instanceContext._log.debug('SCXML State Entry: "' + stateName + '"');// debug
 64                     },
 65                     onExit : function(stateName) {
 66                         statesActive.pop();
 67 
 68                         if (instanceContext._log.isDebug()) instanceContext._log.debug('SCXML State Exit: "' + stateName + '"');// debug
 69                     },
 70                     onTransition : function(sourceState, targetStatesArray) {
 71                         
 72                     	if (instanceContext._log.isDebug()) instanceContext._log.debug('SCXML State Transition: "' + sourceState + '"->"' + targetStatesArray + '"');// debug
 73 
 74                         if (targetStatesArray && targetStatesArray.length > 1) {
 75                         	instanceContext._log.warn('SCXML State Transition: multiple target states!');
 76                         }
 77                     }
 78                 };
 79 
 80                 _interpreter.registerListener(listener);
 81 
 82                 if (self.onload) {
 83                 	var _scion = scionUtil( _interpreter );
 84                 	if(!self.evalScript) self.scion.ignoreScript();
 85                 	self.onload( _scion, _defer );
 86                 } else {
 87                 	deferred.resolve(instanceContext);
 88                 }
 89                 
 90             });//END: scion.urlToModel(...
 91 
 92             
 93             return _defer.promise();
 94             
 95         };//END: load = function(){...
 96 
 97         /**
 98     	 * @deprecated instead use the object directly
 99     	 */
100         instanceContext.getInstance= function () {
101         	return this;
102         };
103         instanceContext.load = load;
104         instanceContext.onload = null;
105         instanceContext.doc = null;
106         instanceContext.raise = null;
107         
108         return instanceContext;
109         
110     };//END: newInstance(){...

115     /**
116      * Creates a new SCION engine.
117      * 
118      * @param {Object} configuration

120      * 				all properties and functions from this object will be attached

123      * 				The context object: the SCION engine will be attached to this object.
124      * 				The context object must have a property <code>_log</code> with the
125      * 				following attributes (may be empty non-function):
126      * 				<code>_log.isDebug() : Boolean</code>  (this SHOULD NOT print / show any message)
127      * 				<code>_log.debug(String) : void</code> (this SHOULD NOT print / show any message)
128      * 				<code>_log.warn(String) : void</code>  (this MAY print / show a warning message)
129      * 				<code>_log.error(String) : void</code> (this SHOULD print / show an error message)
130      * 				or use a Logger instance (see /tools/logger.js) that is setup for the module
131      * 				that calls this function (see e.g. /manager/dialog/dialogManager::init).
132      * @returns {Object} the created SCION engine object
133      */
134 	return function (configuration , context){
135 		
136         var _instance = newInstance( scion, scionUtil , context);
137         
138         for (var key in configuration) {
139             _instance[key] = configuration[key];
140         }
141         
142         return _instance;
143
