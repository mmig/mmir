127         /**
128          * "Normalizes" a string or an array into a path:
129          * the result is a single, flat array where each string has
130          * been separated at dots (i.e. each path component is a separate entry).
131          * 
132          * @example 
133          *   //result is ['dot', 'separated', 'list']
134          *   _getAsPath('dot.separated.list');
135          *   _getAsPath(['dot', 'separated.list']);
136          *   _getAsPath(['dot', 'separated', 'list']);
137          * 
138          * @private
139          * @param {String|Array<String>} propertyName
140          * 				resolves a dot-separated property-name into an array.
141          * 				If <code>propertyName</code> is an Array, all contained
142          * 				String entries will be resolved, if necessary
143          * 				
144 		 * @memberOf mmir.ConfigurationManager#
145          */
146         function _getAsPath(propertyName){
147         	
148         	var path = propertyName;
149         	if( ! $.isArray(path)){
150         		path = propertyName.split('.');
151         	}
152         	else {
153         		path = _toPathArray(propertyName);
154         	}
155         	
156         	return path;
157         }
158         
159         /**
160          * "Normalizes" an array of Strings by separating
161          * each String at dots and creating one single (flat) array where
162          * each path-component is a single entry.
163          * 
164          * @private
165          * @param {Array<String>} pathList
166          * 				resolves an array with paths, i.e. dot-separated property-names
167          * 				into a single, flat array where each path component is a separate Strings
168          * 
169 		 * @memberOf mmir.ConfigurationManager#
170          */
171         function _toPathArray(pathList){
172         		
173     		var entry;
174     		var increase = 0;
175     		var size = pathList.length;
176     		var tempPath;
177     		
178     		for(var i=0; i < size; ++i){
179     			
180     			entry = pathList[i];
181     			tempPath = entry.split('.');
182     			
183     			//if entry contained dot-separated path:
184     			// replace original entry with the new sub-path
185     			if(tempPath.length > 1){
186     				pathList[i] = tempPath;
187     				increase += (tempPath.length - 1);
188     			}
189     		}
190     		
191     		//if sup-paths were inserted: flatten the array
192     		if(increase > 0){
193     			
194     			//create new array that can hold all entries
195     			var newPath = new Array(size + increase);
196     			var index = 0;
197     			for(var i=0; i < size; ++i){
198 
199         			entry = pathList[i];
200     				
201         			//flatten sub-paths into the new array:
202     				if( $.isArray(entry) ){
203     					
204     	        		for(var j=0, len=entry.length; j < len; ++j){
205     	        			newPath[index++] = entry[j];
206     					}
207     				}
208     				else {
209     					//for normal entries: just insert 
210     					newPath[index++] = entry;
211     				}
212     			}
213     			
214     			pathList = newPath;
215     		}
216         	
217     		return pathList;
218         }

263 			/**
264 			 * Returns the value of a property.
265 			 *  
266 			 * @function get
267 			 * @param {String} propertyName
268 			 * 					The name of the property.
269 			 * 					NOTE: If the property does not exists at the root-level,
270 			 * 						  dot-separated names will be resolved into
271 			 * 						  object-structures, e.g.
272 			 * 						  <code>some.property</code> will be resolved
273 			 * 						  so that the <code>value</code> at:
274 			 * 						  <code>some: {property: <value>}</code>
275 			 * 						  will be returned
276 			 * @param {Boolean} [useSafeAccess] OPTIONAL
277 			 * 					if <code>true</code>, resolution of dot-separated paths
278 			 * 					will be done "safely", i.e. if a path-element does not
279 			 * 					exists, no <code>error</code> will be thrown, but instead
280 			 * 					the function will return the <code>defaultValue</code>
281 			 * 					(which will be <code>undefined</code> if the argument is not given).
282 			 * @param {any} [defaultValue] OPTIONAL
283 			 * 					a default value that will be returned, in case there is no property
284 			 * 					<code>propertyName</code>.
285 			 * 
286              * 					NOTE: if this argument is used, <code>useSafeAccess</code> must also be given!
287 			 * 
288 			 * @returns {any} 
289 			 * 					The value of the property
290 			 * @public
291 			 */
292             get: function(propertyName, useSafeAccess, defaultValue){
293             	
294             	if(configData){
295             		
296             		if(typeof configData[propertyName] !== 'undefined'){
297             			return configData[propertyName];//////////// EARLY EXIT ///////////////////
298             		}
299             		
300             		var path = _getAsPath(propertyName);
301             		
302             		//ASSERT path.length == 1: already handled by if(configData[propertyName]...
303             		
304             		if(path.length > 1){
305             			
306             			if(useSafeAccess && typeof configData[ path[0] ] === 'undefined'){
307             				return defaultValue;///////////// EARLY EXIT /////////////////////////
308             			}
309             			
310                 		var obj = configData;
311                 		var prop;
312                 		while(path.length > 1){
313                 			prop = path.shift();
314                 			obj = obj[prop];
315                 			
316                 			if(useSafeAccess && typeof obj === 'undefined'){
317                 				return defaultValue;///////////// EARLY EXIT /////////////////////
318                 			}
319                 		}
320                 		
321                 		//ASSERT now: path.length === 1
322                 		
323                 		if(typeof obj[path[0]] === 'undefined'){
324                 			return defaultValue;///////////// EARLY EXIT /////////////////////
325                 		}
326                 		return obj[path[0]];
327                 		
328                 	}
329             	}
330             	
331             	return defaultValue;
332             },
333 			/**
334 			 * Sets a property to a given value.
335 			 *  
336 			 * @function set
337 			 * @param {String|Array<String>} propertyName
338 			 * 				
339 			 * 				The name of the property.
340 			 * 				
341 			 * 				If <code>propertyName</code> is an Array, it
342 			 * 				will be treated as if its entries were path-elements
343 			 * 				analogous to a dot-separated String propertyName.
344 			 * 
345 			 * 				NOTE: dot-separated names will be resolved into
346 			 * 					  object-structures, e.g.
347 			 * 					  <code>some.property</code> will be resolved
348 			 * 					  so that the <code>value</code> will set to:
349 			 * 					  <code>some: {property: <value>}</code>
350 			 * 					
351 			 * @param {any} value
352 			 * 				The value of the property
353 			 * 
354 			 * @throws {Error} if the propertyName is dot-separated AND
355 			 * 					one of its path-elements (except for the last)
356 			 * 					already exists AND its type is not 'object' 
357 			 * 
358 			 * @public
359 			 */
360             set: function(propertyName, value){
361             	if(!configData){
362             		configData = {};
363             	}
364             	
365             	var path = _getAsPath(propertyName);
366             	
367             	if(path.length > 1){
368             		var obj = configData;
369             		var prop;
370             		while(path.length > 1){
371             			
372             			prop = path.shift();
373             			
374             			if(typeof obj[prop] === 'undfined' || obj[prop] === null){
375             				obj[prop] = {};
376             			}
377             			else if(typeof obj[prop] !== 'object'){
378             				throw new Error('Cannot expand path "'+propertyName+'": path-element "'+prop+'" already exist and has type "'+(typeof obj[prop])+'"');
379             			}
380             			
381             			obj = obj[prop];
382             		}
383             		
384             		//ASSERT path.length == 1
385             		
386             		obj[path[0]] = value;
387             	}
388             	else {            	
389             		configData[propertyName] = value;
390             	}
391             },

393             /**
394              * Uses {@link #get}.
395              * 
396              * If the propertyName does not exists, returns <code>undefined</code>,
397              * otherwise values will be converted into Boolean values.
398              * 
399              * Special case for Strings:
400              * the String <code>"false"</code> will be converted to
401              * Boolean value <code>false</code>.
402              * 
403              * @param {any} [defaultValue] OPTIONAL
404              * 
405              * 			if a default value is specified and there exists
406              * 			no property <code>propertyName</code>, the
407              * 			specified default value will be returned.
408              * 
409              * 			NOTE: if this argument is used, <code>useSafeAccess</code> must also be given!
410              * 			
411              * 			NOTE: the default value will also be converted
412              * 				  to a Boolean value, if necessary. 
413              * 
414              * @see {@link #get}
415              */
416             getBoolean: function(propertyName, useSafeAccess, defaultValue){
417             	
418             	var val = this.get(propertyName, useSafeAccess);
419             	
420             	if(typeof val === 'undefined' && typeof defaultValue !== 'undefined' ){
421             		val = defaultValue;
422             	}
423             	
424             	if(typeof val !== 'undefined'){
425             		
426             		if( val === 'false'){
427             			return false;
428             		}
429             		else {
430             			return val? true : false;
431             		}
432             	}
433             	
434             },
435             
436             /**
437              * Uses {@link #get}.
438              * 
439              * If the property does not exists, returns <code>undefined</code>,
440              * otherwise values will be converted into String values.
441              * 
442              * If the value has not the type <code>"string"</code>, it will
443              * be converted by <code>JSON.stringify</code>.
444              * 
445              * @param {any} [defaultValue] OPTIONAL
446              * 			if a default value is specified and there exists
447              * 			no property <code>propertyName</code>, the
448              * 			specified default value will be returned.
449              * 
450              * 			NOTE: if this argument is used, <code>useSafeAccess</code> must also be given!
451              * 			
452              * 			NOTE: the default value will also be converted
453              * 				  to a String value, if necessary. 
454              * 
455              * @see {@link #get}
456              */
457             getString: function(propertyName, useSafeAccess, defaultValue){
458             	
459             	var val = this.get(propertyName, useSafeAccess);
460             	
461             	if(typeof val === 'undefined' && typeof defaultValue !== 'undefined' ){
462             		val = defaultValue;
463             	}
464             	
465             	if(typeof val !== 'undefined'){
466             		
467             		if(typeof val === 'string'){
468             			return val;
469             		}
470             		else {
471             			return JSON.stringify(val);
472             		}
473             	}
474             	
475             }
