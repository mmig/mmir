 28 define(
 29 	/**
 30 	 * @name Helper
 31 	 * @class
 32 	 */
 33 	function(
 34 ){
 35 	/** @scope Helper.prototype */
 36 	/**
 37 	 * #@+
 38 	 * @memberOf Helper.prototype
 39 	 */
 40 		
 41 	/**
 42 	 * The Helper Class is a kind of interface-class which gives access to the methods and data of a helper (which itself belongs to a controller)<br>
 43 	 * 
 44 	 * @constructs Helper
 45 	 * @param {Object} ctrl Controller instance / object
 46 	 * @param {String} name Name of the Helper 
 47 	 * @category core
 48 	 */
 49 	function Helper(ctrl, name){
 50 	    /**
 51 	     * The definition of the helper object, containing all properties and functions of the controller.<br>
 52 	     * A method of the controller can be called via:
 53 
 54 		this.script.methodController(parameter);
 55 
 56 	     * 
 57 	     * @property script
 58 	     * @type Object
 59 	     * @public
 60 	     */
 61 		// this can only be invoked, if a function with the name "name" exists
 62 		this.script = new window[name]();
 63 
 64 		/**
 65 	     * The name of the helper. 
 66 	     * 
 67 	     * @property name
 68 	     * @type String
 69 	     * @public
 70 	     */
 71 		this.name=name;
 72 
 73 		/**
 74 	     * The controller to which this helper belongs.
 75 	     * 
 76 	     * @property controller
 77 	     * @type Object
 78 	     * @public
 79 	     */
 80 		this.controller = ctrl;
 81 
 82 	}
 83 
 84 
 85 	/**
 86 	 * This function performs an action of a helper.<br>
 87 	 * 
 88 	 * @function perform
 89 	 * @param {String} actionName Name of the method to be executed
 90 	 * @param {Object} data Data to pass to the method of the helper as argument
 91 	 * @returns {Object} The return value of the executed method 
 92 	 * @public
 93 	 */
 94 	Helper.prototype.perform = function(actionName, data){
 95 		
 96 //		if(logger.isv()) logger.v("should perform '" + actionName + "' of '" + this.name + "'" + ((typeof data !== 'undefined' && data !== null)? " with data: "+JSON.stringify(data): ""));//debug
 97 		
 98 		if(arguments.length > 2){
 99 		    return this.script[actionName](this.controller, data, arguments[2]);
100 		}
101 		else {
102 		    return this.script[actionName](this.controller, data);
103 		}
104 	};
105 
106 
107 	return Helper;
108
