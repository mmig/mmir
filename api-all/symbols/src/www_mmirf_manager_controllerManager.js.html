118 	    /**
119 		 * HELPER FUNC: add file path for generated / compiled view-element
120 		 *              if it exists.
121 		 * 
122 		 * @param {String} genDirPath
123 		 * 				path the the directory, where the file for the generated view-element
124 		 * 				is potentially located (generated file may not exists)
125 		 * 
126 		 * @param {PlainObject} infoObj
127 		 * 				the info-object for the view-element. MUST HAVE property <code>name</code>!
128 		 * 
129 		 * @param {String} [fileNamePrefix] OPTIONAL
130 		 * 				prefix for the file-name, e.g. in case of Partials, the file-name would be:
131 		 * 				<code>fileNamePrefix + infoObj.name</code> (+ file-extension)
132 		 * 
133 		 * @returns {PlainObject}
134 		 * 				the info-object: if a path for the generated file exists,
135 		 * 				a property <code>genPath</code> (String) with the path as value is added.
136 		 * @private
137 		 * 
138 		 * @memberOf mmir.ControllerManager#
139 		 */
140 	    function addGenPath(genDirPath, infoObj, fileNamePrefix){
141 	    	
142 	    	var prefix = fileNamePrefix? fileNamePrefix : '';
143 	    	var genPath = commonUtils.getDirectoryContentsWithFilter(genDirPath, prefix + infoObj.name+".js");
144     		if(genPath && genPath.length > 0){
145     			infoObj.genPath = genDirPath + '/' + genPath[0];
146     		}
147     		
148     		return infoObj;
149 	    }

151 		/**
152 		 * This function gets the controller file names and builds a JSON object containing information about
153 		 * the location, file name etc. for the controller itself, its views, partials, layout, and helper.
154 		 * 
155 		 * @function getControllerResources
156 		 * @param {String} controllerName
157 		 * 					the name of the Controller (must start with an upper case letter).
158 		 * @param {String} controllerPath
159 		 * 					the path (URL) where the file with the Controller's implementation
160 		 * 					is located (according to information in file <code>/config/directories.json</code>,
161 		 * 					i.e. {@link mmir.CommonUtils#getDirectoryStructure})
162 		 * @returns {JSON} JSON-Object containing information about the controller,
163 		 * 				its views, partials, and paths etc. 
164 		 * @private
165 		 * 
166 		 * @example
167 		 * //EXAMPLE for returned object:
168 		 * {
169 		 *   "fileName": "application",
170 		 *   "name": "Application",
171 		 *   "path": "controllers/application.js",
172 		 *   "views": [
173 		 *     {
174 		 *       "name": "login",
175 		 *       "path": "views/application/login.ehtml",
176 		 *       "genPath": "gen/views/application/login.js"
177 		 *     },
178 		 *     {
179 		 *       "name": "registration",
180 		 *       "path": "views/application/registration.ehtml",
181 		 *       "genPath": "gen/views/application/registration.js"
182 		 *     },
183 		 *     {
184 		 *       "name": "welcome",
185 		 *       "path": "views/application/welcome.ehtml",
186 		 *       "genPath": "gen/views/application/welcome.js"
187 		 *     }
188 		 *   ],
189 		 *   "partials": [
190 		 *     {
191 		 *       "name": "languageMenu",
192 		 *       "path": "views/application/~languageMenu.ehtml",
193 		 *       "genPath": "gen/views/application/~languageMenu.js"
194 		 *     }
195 		 *   ],
196 		 *   "helper": {
197 		 *     "fileName": "applicationHelper",
198 		 *     "name": "ApplicationHelper",
199 		 *     "path": "helpers/applicationHelper.js"
200 		 *   },
201 		 *   "layout": {
202 		 *     "fileName": "application",
203 		 *     "name": "application",
204 		 *     "path": "views/layouts/application.ehtml",
205 		 *     "genPath": "gen/views/layouts/application.js"
206 		 *   }
207 		 * }
208 		 * //NOTE 1: genPath is an optional field, i.e. it will only be added
209 		 *           if the corresponding file exists
210 		 * //NOTE 2: layout may be NULL
211 		 * 
212 		 * @requires mmir.CommonUtils
213 		 * @requires mmir.Constants
214 		 * 
215 		 * @memberOf mmir.ControllerManager#
216 		 */
217 	    function getControllerResources(controllerName, controllerPath){
218 	    	
219 	    	var partialsPrefix = commonUtils.getPartialsPrefix();
220 	    	var controllerFilePath = controllerPath + controllerName;
221 	    	
222 	    	var rawControllerName= removeFileExt(controllerName);
223 	    	controllerName = rawControllerName;
224 	    	
225 	    	
226 	    	var viewsPath = constants.getViewPath() + controllerName;
227 	    	var genViewsPath = constants.getCompiledViewPath() + controllerName;
228 	    	
229 	    	controllerName = firstToUpperCase(controllerName);
230 	    	
231 	    	var viewsFileList = commonUtils.getDirectoryContentsWithFilter(viewsPath, "(?!"+partialsPrefix+")*.ehtml");
232 
233 	    	var i, size;
234 	    	var viewsList = [];
235 	    	if(viewsFileList != null){
236 	    		for (i=0, size = viewsFileList.length; i < size; ++i){
237 	    			
238 	    			viewsList.push(addGenPath( genViewsPath, {
239 		    			name: removeFileExt(viewsFileList[i]),
240 		    			path: viewsPath+"/"+viewsFileList[i]
241 		    		}));
242 		    	}
243 	    	}
244 
245 	    	var partialsFileList = commonUtils.getDirectoryContentsWithFilter(viewsPath, partialsPrefix+"*.ehtml");
246 
247 	    	var partialsInfoList = [];
248 	    	if(partialsFileList != null) {
249 	    		for (i=0, size = partialsFileList.length; i < size; ++i){
250 	    		
251 		    		partialsInfoList.push(addGenPath(genViewsPath, {
252 				    		// remove leading "~" indicating it is a partial
253 				    		name: removeFileExt( partialsFileList[i].replace(partialsPrefix,'') ),
254 				        	path: viewsPath+"/"+partialsFileList[i]
255 		    		
256 					}, partialsPrefix));
257 		        }
258 	    	}
259 
260 	    	var helpersPath = constants.getHelperPath();
261 	    	helpersPath = helpersPath.substring(0, helpersPath.length-1);//remove trailing slash
262 	    	var helpersFileList = commonUtils.getDirectoryContentsWithFilter(helpersPath, "(?!"+partialsPrefix+")*.js");
263 
264 	    	var helperSuffix = constants.getHelperSuffix();
265 	    	var helperInfo = null;
266 	    	if(helpersFileList != null){
267 	    		
268 	    		for(i=0, size = helpersFileList.length; i < size; ++i){
269 		    		if(helpersFileList[i].startsWith(controllerName, true) && helpersFileList[i].endsWith(helperSuffix+'.js', true)){
270 		    	    	
271 		    			var name = removeFileExt(helpersFileList[i]);
272 		    			helperInfo = {
273 		    	    			fileName: name,
274 		    	    			name: firstToUpperCase(name),
275 		    	    			path: helpersPath+"/"+helpersFileList[i]
276 		    	    	};
277 		    		}
278 		    	}
279 	    		
280 	        }
281 	    	
282 	    	var layoutsPath = constants.getLayoutPath();
283 	    	layoutsPath = layoutsPath.substring(0, layoutsPath.length-1);//remove trailing slash
284 	    	var layoutsFileList = commonUtils.getDirectoryContentsWithFilter(layoutsPath, "(?!"+partialsPrefix+")*.ehtml");
285 	    	
286 	    	var layoutInfo = null, layoutGenPath;
287 	    	for(i=0, size = layoutsFileList.length; i < size; ++i){
288 	    		
289 	    		if( layoutsFileList[i].startsWith(controllerName, true) ){
290 	    			
291 	    			var layoutName = removeFileExt(layoutsFileList[i]);
292 	    	    	layoutInfo = {
293 			    		fileName: layoutName,
294 			    		name: firstToUpperCase(layoutName),
295 			        	path: layoutsPath+"/"+layoutsFileList[i],
296 	    	    	};
297 	    	    	
298 	    	    	layoutGenPath = constants.getCompiledLayoutPath();
299 	    	    	addGenPath(layoutGenPath.substring(0, layoutGenPath.length-1), layoutInfo);
300 		        	
301 		        	break;
302 	    		}
303 	        }
304 	    	
305 	    	var ctrlInfo = {
306 	    		fileName: rawControllerName,
307 	    		name:     controllerName,
308 	    		path:     controllerFilePath,
309 	    		
310 	    		views:    viewsList,
311 	    		partials: partialsInfoList,
312 	    		helper:   helperInfo,
313 	    		layout:   layoutInfo
314 	    	};
315 	    	
316 	    	//TEST compare info with "reference" result from original impl.:
317 //	    	var test ={
318 //	    			application: '{"fileName":"application","name":"Application","path":"controllers/application.js","views":[{"name":"login","path":"views/application/login.ehtml"},{"name":"registration","path":"views/application/registration.ehtml"},{"name":"welcome","path":"views/application/welcome.ehtml"}],"partials":[{"name":"languageMenu","path":"views/application/~languageMenu.ehtml"}],"helper":{"fileName":"applicationHelper","name":"ApplicationHelper","path":"helpers/applicationHelper.js"},"layout":{"fileName":"application","name":"application","path":"views/layouts/application.ehtml"}}',
319 //	    			calendar: '{"fileName":"calendar","name":"Calendar","path":"controllers/calendar.js","views":[{"name":"create_appointment","path":"views/calendar/create_appointment.ehtml"}],"partials":[],"helper":null,"layout":null}'
320 //	    	};
321 //	    	
322 //	    	var isEqual = (JSON.stringify(ctrlInfo) === test[ctrlInfo.fileName]);
323 //	    	console[isEqual? 'info':'error']('compliance-test: isEual? '+  isEqual);
324 	        
325 	        return ctrlInfo;
326 		};

346 				function callbackStatus(status, fileName, msg) {
347 					if(status==='info'){
348 						
349 						console.info('[loadController] "'+fileName);
350 
351 						var ctrlInfo = getControllerResources(fileName, constants.getControllerPath());
352 
353 						var controller = new Controller(ctrlInfo.name, ctrlInfo);
354 
355 						if(ctrlInfo.helper){
356 							var helperPath = ctrlInfo.helper.path;
357 							var helperName = ctrlInfo.helper.name;
358 							controller.loadHelper(helperName,helperPath);
359 						}
360 
361 						controllers.put(controller.getName(), controller);
362 					}
363 					else if(status==='warning'){
364 						console.warn('[loadController] "'+fileName+'": '+msg);
365 					}
366 					else if(status==='error'){
367 						console.error('[loadController] "'+fileName+'": '+msg);
368 					}
369 					else{
370 						console.error('[loadController] '+status+' (UNKNOWN STATUS) -> "'+fileName+'": '+msg);
371 					}               
372 				}
373 
374 		);		
375 
376 		return deferred.promise(_instance);

409 			 * This function gets the controller by name.

413 			 * @returns {Object} controller if found, null else

438 			/**
439 			 * This function performs an action of a controller. 
440 			 * 
441 			 * @function perform
442 			 * @param {String} ctrlName Name of the controller to which the action belongs
443 			 * @param {String} actionName Name of the action that should be performed
444 			 * @param {Object} data optional data that can be submitted to the action
445 			 * @returns {Object} the return object of the performed action
446 			 * @public
447 			 */
448 			perform: function(ctrlName, actionName, data){
449 				var ctrl = this.getController(ctrlName);
450 				if (ctrl != null) {
451 					return ctrl.perform(actionName, data);
452 				}
453 				else {
454 					console.error('ControllerManager.perform: the controller could not be found "'+ctrlName+'"');
455 				}
456 			},

458 
459 			/**
460 			 * This function performs an action of a helper-class for a controller. 
461 			 * 
462 			 * @function performHelper
463 			 * @param {String} ctrlName Name of the controller to which the helper action belongs
464 			 * @param {String} actionName Name of the action that should be performed by the helper
465 			 * @param {Object} data optional data that can be submitted to the action
466 			 * @returns {Object} the return object of the performed action
467 			 * @public
468 			 */
469 			performHelper: function(ctrlName, actionName, data) {
470 
471 				var ctrl = this.getController(ctrlName);
472 				if (ctrl != null) {
473 					if(arguments.length > 3){
474 						return ctrl.performHelper(actionName, data, arguments[3]);
475 					}
476 					else {
477 						return ctrl.performHelper(actionName, data);
478 					}
479 				}
480 				else {
481 					console.error('ControllerManager.performHelper: the controller could not be found "'+ctrlName+'"');
482 				}
483 			},
484 			/**
485 			 * This function must be called before using the {@link mmir.ControllerManager}. The Initialization process is asynchronous, 
486 			 * because javascript-files must be loaded (the controllers).
487 			 * To ensure that the ControllerManager is initialized, a callback can be used, or the returned
488 			 * <em>Promise</em> (e.g. see documentation of jQuery.Deferred) for code, that relies
489 			 * on the presence of the loaded controllers.<br>   
490 			 * 
491 			 * 
492 			 * <div class="box important">
493 			 * <b>Note:</b>
494 			 * The callback function should be used for code, that requires the prior loading of the controllers.<br> 
495 			 * The callback mechanism is necessary, because loading the controllers is asynchronous.<br><br>
496 			 * If provided, the callback function is invoked with 1 argument, the ControllerManager instance:<br>
497 			 * <code> callbackFunction(controllerManagerInstance) </code>
498 			 * </div>
499 			 *

501 			 * 
502 			 * @param {Function} [callback] OPTIONAL
503 			 * 				an optional callback that will be triggered after the controllers where loaded
504 			 * @returns {Promise}
505 			 * 				a Deferred.promise that will get fulfilled when controllers are loaded
506 			 * @example
507 			 *  //recommended style:
508 			 *  require(['controllerManager', ...], function(controllerManager, ...) {
509 			 *  	controllerManager.init().then(function(theInitializedControllerInstance){
510 			 *  		...
511 			 *  	});
512 			 *  })
513 			 *  
514 			 *  //old style:
515 			 * 	function afterLoadingControllers(controllerManagerInstance){
516 			 * 		var appCtrl = controllerManagerInstance.getController('Application');
517 			 * 		//do something...
518 			 * 	} 
519 			 * 	mmir.ControllerManager.init(afterLoadingControllers);
520 			 * @public
521 			 */
522 			init: _init
