217 	function setBasePath(isBrowserEnvParam){
218 		// if not on browser: basepath must be different
219 		if(typeof isBrowserEnvParam === 'string'){
220 			basePath = isBrowserEnvParam;
221 		}
222 		else if (isBrowserEnvParam && isBrowserEnvParam.env){
223 
224 			switch(isBrowserEnvParam.env){
225 				case 'cordova':
226 				case 'android':
227 					basePath = "file:///android_asset/www/";
228 					break;
229 				case 'ios':
230 				case 'default':
231 					basePath = "";
232 			}
233 			
234 		}
235 		else if (isBrowserEnvParam){
236 			basePath = "";
237 		}
238 		else {
239 			//TODO this is only for ANDROID -- need to set appropriate path path according to ENV setting...
240 			basePath = "file:///android_asset/www/";
241 		}
242 	}
