226 			/**
227 			 * HELPER invoke current callback function with last recognition results.
228 			 * @private
229 			 * @memberOf NuanceAndroidAudioInput#
230 			 */
231 			var call_callback_with_last_result = function(){
232 				if(typeof last_result !== "undefined") {
233 					if (currentSuccessCallback){
234 						currentSuccessCallback.apply(mediaManager, last_result);
235 						last_result = void(0);
236 					} else {
237 						console.error("nuanceAudioInput Error: No callback function defined for success.");
238 					}
239 				} else {
240 					console.warn("nuanceAudioInput Warning: last_result is undefined.");
241 				}
242 			};

244 			/**
245 			 * Creates the wrapper for the success-back:
246 			 * 
247 			 * successcallback(asr_result, asr_score, asr_type, asr_alternatives, asr_unstable) OR in case of error:
248 			 * successcallback(asr_result, asr_score, asr_type, asr_error_code, asr_error_suggestion)
249 			 * 
250 			 * @private
251 			 * @memberOf NuanceAndroidAudioInput#
252 			 */
253 			var successCallbackWrapper = function successCallbackWrapper (cb){
254 				return (function (res){
255 					
256 					console.log("nuanceAudioInput: " + JSON.stringify(res));//FIXME DEBUG
257 
258 					var asr_result = null;
259 					var asr_score = -1;
260 					var asr_type = -1;
261 					var asr_alternatives = [];
262 
263 					if(res) {
264 						
265 						if(typeof res['result'] !== "undefined"){
266 							asr_result = res['result'];
267 						}
268 						if  (typeof res["score"] !== "undefined"){
269 							asr_score = res["score"];
270 						}
271 						if  (typeof res["type"] !== "undefined"){
272 							asr_type = res["type"];
273 						}
274 						if  (typeof res["alternatives"] !== "undefined"){
275 							asr_alternatives = res["alternatives"];
276 						}
277 					}
278 
279 					//call voice recognition again, if repeat is set to true
280 					if (repeat === true) {
281 						if (asr_type === result_types.RECORDING_BEGIN){
282 							// only call success-callback, if we started recording again.
283 							_wait.ready();
284 							call_callback_with_last_result();
285 						} else if (asr_type === result_types.RECORDING_DONE){
286 							// Do nothing right now at the recording done event
287 						} else if (asr_type === result_types.FINAL){
288 							// its the final result
289 							// post last result
290 							call_callback_with_last_result();
291 							last_result = [asr_result, asr_score, asr_type, asr_alternatives];
292 							// post current result
293 							call_callback_with_last_result();
294 						} else if (asr_type === result_types.INTERMEDIATE){
295 							_wait.preparing();
296 							// save the last result and start recognition again
297 							last_result = [asr_result, asr_score, asr_type, asr_alternatives];
298 
299 							window.plugins.nuancePlugin.recognizeNoEOS(
300 									languageManager.getLanguageConfig(_pluginName),
301 									successCallbackWrapper(currentSuccessCallback),
302 									failureCallbackWrapper(currentFailureCallback),
303 									intermediate_results
304 							);
305 
306 						} else {
307 							// save the last result and start recognition again
308 //							last_result = [asr_result, asr_score, asr_type, asr_alternatives];
309 
310 //							window.plugins.nuancePlugin.recognizeNoEOS(
311 //							languageManager.getLanguageConfig(_pluginName),
312 //							successCallbackWrapper(currentSuccessCallback),
313 //							failureCallbackWrapper(currentFailureCallback),
314 //							intermediate_results
315 //							);
316 //							console.warn("[NuanceAudioInput] Success - Repeat - Else\nType: " + asr_type+"\n"+JSON.stringify(res));
317 						}
318 
319 					} else {
320 						// no repeat, there won't be another recording, so call callback right now with previous and current result
321 
322 						_wait.ready();
323 
324 						if (asr_type === result_types.RECORDING_DONE){
325 						} else if (asr_type === result_types.FINAL){
326 						}
327 						call_callback_with_last_result();
328 
329 						if(typeof res !== "undefined") {
330 							if (cb){
331 								cb(asr_result, asr_score, asr_type, asr_alternatives);
332 							} else {
333 								console.error("nuanceAudioInput Error: No callback function defined for success.");
334 							}
335 						} else {
336 							console.warn("nuanceAudioInput Warning: result parameter is undefined.");
337 						}
338 					}
339 				});
340 			};

343 			/**
344 			 * creates the wrapper for the failure callback.
345 			 * 
346 			 * NOTE: error code 2 - there is no difference between a long silence and unintelligible words
347 			 * 
348 			 * @private
349 			 * @memberOf NuanceAndroidAudioInput#
350 			 */
351 			var failureCallbackWrapper = function failureCallbackWrapper (cb){
352 				return (function (res){
353 					var error_code = -1;
354 					var error_msg = "";
355 					var error_suggestion = "";
356 					var error_type = -1;
357 
358 					if (typeof res !== "undefined"){
359 						if(typeof res['error_code'] !== "undefined") {
360 
361 							error_code = res['error_code'];
362 						}
363 						if  (typeof res["msg"] !== "undefined"){
364 							error_msg = res["msg"];
365 						}
366 
367 						if (typeof res["suggestion"] !== "undefined"){
368 							error_suggestion = res["suggestion"];
369 						}
370 
371 						if  (typeof res["type"] !== "undefined"){
372 							error_type = res["type"];
373 						}
374 					}
375 
376 					_wait.ready();
377 
378 					// TEST: if there is still a pending last result, call successcallback first.
379 					call_callback_with_last_result();
380 					if (repeat === true){
381 						// only call error callback on "severe" errors
382 						// TODO: on SERVER_RETRY => message to user
383 						if (		(error_code == error_codes_enum.UNKNOWN)
384 								// if asr is active and the internet connection is lost, it throws error code 1 (SERVER_CONNECTION) once and thereafter error code 3 (RECOGNIZER)
385 								||	(error_code == error_codes_enum.SERVER_CONNECTION)
386 								// if asr is started when the internet connection is disabled, error code 1 (SERVER_CONNECTION) is not thrown, only error code 3 (RECOGNIZER) - so also stop at recognizer error.
387 								||	(error_code == error_codes_enum.RECOGNIZER)
388 								||	(error_code == error_codes_enum.CANCEL)
389 						){
390 							if (cb){
391 								console.warn("nuanceAudioInput: Calling error callback (" + error_code + ": " + error_msg + ").");
392 								cb(error_msg, error_code, error_suggestion);
393 							} else {
394 								console.error("nuanceAudioInput Error: No callback function defined for failure.");
395 							}
396 						} else {
397 							
398 							// this is a minor error, call success-callback with empty result-string
399 							currentSuccessCallback("", -1, result_types.RECOGNITION_ERROR, error_code, error_suggestion);
400 							
401 							//show loader so that the user knows it may take a while before he can start talking again
402 							if (error_type !== result_types.FINAL){
403 								_wait.preparing();
404 							}
405 							
406 							//no (serious) error, call voice recognition again
407 							return window.plugins.nuancePlugin.recognizeNoEOS(
408 									languageManager.getLanguageConfig(_pluginName),
409 									successCallbackWrapper(currentSuccessCallback),
410 									failureCallbackWrapper(currentFailureCallback),
411 									intermediate_results
412 							);
413 						}
414 						
415 					} else {
416 						
417 						// do no repeat, just call errorCallback
418 						if (cb){
419 							console.debug("nuanceAudioInput: Calling error callback (" + error_code + ").");
420 							cb(error_msg, error_code, error_suggestion);
421 						} else {
422 							console.error("nuanceAudioInput Error: No callback function defined for failure.");
423 						}
424 					}
425 				});
426 			};

447 							successCallbackWrapper(successCallback),
