230     		 /**
231     		  * The function that actually renders the View.<br>
232     		  * 
233     		  * The function will be invoked in context of the PresentationManager instance
234     		  * (i.e. the manager will be the <em>this</em> context).
235     		  * 
236     		  * Implementations of this function should adhere to the following procedure:
237     		  * 
238     		  * <br>
239     		  * 
240     		  * First this function fetches the <em>layout</em> for the <em>controller</em>
241     		  * (or uses the <code>dialogManager.DEFAULT_LAYOUT<code>).
242     		  * 
243     		  * Then the <code>before_page_prepare</code> of the <em>controller</em> is invoked (if it exists).
244     		  * 
245     		  * Then renders the <em>view</em> into the
246     		  * layout-template; <em>partials</em>, <em>helpers</em> etc.
247     		  * that are referenced in the <em>view</em> will be processed,
248     		  * executed etc.; during this, localized Strings should be processed and rendered by
249     		  * {@link mmir.LanguageManager#getText}. 
250     		  * 
251     		  * Then <em>dialogs</em> are created and the <code>dialogManager.pageIndex</code> is updated.
252     		  * 
253     		  * The new content is inserted into the document/page (invisibly).
254     		  * 
255     		  * Then the <code>before_page_load</code> of the <em>controller</em> is invoked (if it exists).
256     		  * 
257     		  * The new content/page is made visible, and the old one invisible and / or is removed.
258     		  * 
259     		  * At the end the <b>on_page_load</b> action of the <em>controller</em> is performed.
260     		  * 
261     		  * @function render
262     		  * @memberOf mmir.PresentationManager._renderingEngine
263     		  * 
264     		  * @param {String}
265     		  *            ctrlName Name of the controller
266     		  * @param {String}
267     		  *            viewName Name of the view to render
268     		  * @param {Object}
269     		  *            view View object that is to be rendered
270     		  * @param {Object}
271     		  *            ctrl Controller object of the view to render
272     		  * @param {Object}
273     		  *            [data] optional data for the view.

476 			/**
477 			 * Gets the view for a controller, then executes helper methods on
478 			 * the view data. The Rendering of the view is done by the
479 			 * {@link #doRenderView} method. Also
480 			 * stores the previous and current view with parameters.<br>
481 			 * 
482 			 * @function renderView
483 			 * @param {String}
484 			 *            ctrlName Name of the controller
485 			 * @param {String}
486 			 *            viewName Name of the view to render
487 			 * @param {Object}
488 			 *            [data] optional data for the view.
489 			 *            Currently same jQuery Mobile specific properties are supported: <br>
490 			 *            When these are present, they will be used for animating the 
491 			 *            page transition upon rendering.
492 			 *            
493 			 *            <pre>{transition: STRING, reverse: BOOLEAN}</pre>
494 			 *            where<br>
495 			 *            <code>transition</code>: the name for the transition (see jQuery Mobile Doc for possible values)
496 			 *            							DEFAULT: "none".
497 			 *            <code>reverse</code>: whether the animation should in "forward" (FALSE) direction, or "backwards" (TRUE)
498 			 *            						DEFAULT: FALSE
499 			 * @public
500 			 */
501             renderView : function(ctrlName, viewName, data) {
502             	
503                 var ctrl = controllerManager.getController(ctrlName);
504 
505                 if (ctrl != null) {
506                     var view = this.getView(ctrlName, viewName);
507                     
508                     _renderEngine.render.call(this, ctrlName, viewName, view, ctrl, data);
509 
510                     //TODO russa: _previousView is deprecated (should use a history instead, i.e. application level)
511                     // Only overwrite previous state if and only if the view is not re-rendered!
512 					if (ctrlName != _currentView.ctrlName || viewName != _currentView.viewName || data != _currentView.data){
513 						_previousView.ctrlName=_currentView.ctrlName;
514 						_previousView.viewName=_currentView.viewName;
515 						_previousView.data=_currentView.data;
516 					}
517 					
518 					_currentView.ctrlName=ctrlName; 
519 					_currentView.viewName=viewName; 
520 					_currentView.data=data;
521                 }
522                 else {
523                 	console.warn('PresentationManager.renderView: could not retrieve controller "'+ctrlName+'"');
524                 }
525             },

622             /**
623              * This function allows to call custom functions of the rendering-engine
624              * that was set via {@link #setRenderEngine}.
625              * 
626              * IMPORTANT:
627              * note that the function will be invoked in context of rendering-engine
628              * (i.e. <code>this</code> references will refer to rendering-engine
629              * and not to the PresentationManager instance.
630              * For example, when <code>mmir.PresentationManager.callRenderEngine('hideWaitDialog')</code>
631              * is called, any <code>this</code> references within the <code>hideWaitDialog</code>
632              * implementation would refer to object, that was set in <code>setRenderEngine(object)</code>.
633              * In comparison, when called as <code>mmir.PresentationManager.hideWaitDialog()</code> the
634              * <code>this</code> references refer to the mmir.PresentationManager instance.
635              * 
636              * <br>
637              * NOTE that calling non-existing functions on the rendering-engine 
638              *      will cause an error.
639              * 
640              * @param {String} funcName
641              * 			the name of the function, that should be invoked on the rendering
642              * 			engine.
643              * @param {Array<any>} [args] OPTIONAL
644              * 			the arguments for <code>funcName</code> invoked
645              *          via <code>Function.apply(renderingEngine, args)</code>, e.g.
646              * 			for <code>args = [param1, param2, param3]</code> the
647              * 			function will be called with
648              * 			<code>funcName(param1, param2, param3)</code>
649              * 			(note that the function receives 3 arguments, and
650              *           not 1 Array-argument).
651              * 
652              * @function callRenderEngine
653              * @public
