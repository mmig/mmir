 27 define(
 28 	/**
 29 	 * @name Dictionary
 30 	 * @class
 31 	 * 
 32 	 */
 33 	function(
 34 ){
 35 
 36 /**
 37  * A dictionary (or map) for key-value storage and access.
 38  * 
 39  * @constructs Dictionary
 40  */
 41 function Dictionary() {
 42 
 43 	/**
 44 	 * "map" for the dictionary
 45 	 * 
 46 	 * @property map
 47 	 * @private
 48 	 * 
 49 	 * @memberOf Dictionary#
 50 	 */
 51 	var map = {};
 52 
 53 	/**
 54 	 * This list contains the "keys" of all current entries in <tt>map</tt>.
 55 	 * 
 56 	 * @property keyList
 57 	 * @private
 58 	 * 
 59 	 * @memberOf Dictionary#
 60 	 */
 61 	var keyList = [];
 62 
 63 	/**
 64 	 * Prefix for keys in internal MAP object, for avoiding overwrite of
 65 	 * existing Object properties/functions
 66 	 * 
 67 	 * @property KEY_PREFIX
 68 	 * @static
 69 	 * @private
 70 	 * 
 71 	 * @memberOf Dictionary#
 72 	 */
 73 	var KEY_PREFIX = '$$';
 74 
 75 	/**
 76 	 * Helper function that creates the actual lookup key.
 77 	 * 
 78 	 * The "lookup key" is original key with applied key-prefix.
 79 	 * 
 80 	 * @function lookupKey
 81 	 * @private
 82 	 * 
 83 	 * @memberOf Dictionary#
 84 	 */
 85 	var lookupKey = function(key) {
 86 		return KEY_PREFIX + key;
 87 	};
 88 
 89 	/** @lends Dictionary.prototype */
 90 	return {
 91 		/**
 92 		 * Put / add an entry to the dictionary.
 93 		 * 
 94 		 * @function put
 95 		 * 
 96 		 * @param {String}
 97 		 *            key the lookup key for the value
 98 		 * @param {any}
 99 		 *            value the value to store
100 		 * 
101 		 * @public
102 		 * 
103 		 * @memberOf Dictionary.prototype
104 		 */
105 		put : function(key, value) {
106 
107 			var isAlreadyPresent = this.containsKey(key);
108 
109 			var lKey = lookupKey(key);
110 			map[lKey] = value;
111 
112 			if (!isAlreadyPresent) {
113 				keyList.push(lKey);
114 			}
115 		},
116 		/**
117 		 * Check if the dictionary contains an entry for a key.
118 		 * 
119 		 * @function containsKey
120 		 * 
121 		 * @param {String}
122 		 *            key the lookup key to check
123 		 * @returns {Boolean} <code>true</code> if an entry exists, otherwise
124 		 *          <code>false</code>
125 		 * 
126 		 * @public
127 		 */
128 		containsKey : function(key) {
129 			return typeof map[lookupKey(key)] !== 'undefined';
130 		},
131 		/**
132 		 * Check if the dictionary contains an entry with the value.
133 		 * 
134 		 * <p>
135 		 * NOTE that this function may execute rather slowly, with O(n).
136 		 * 
137 		 * @function containsValue
138 		 * 
139 		 * @param {any}
140 		 *            value the value to check
141 		 * @param {Boolean}
142 		 *            [useStrict] if <code>true</code> entry-values are
143 		 *            checked against param <tt>value</tt> with
144 		 *            <code>===</code>. If <code>false</code> or omitted,
145 		 *            values are compared with each other using <code>==</code>.
146 		 * @returns {Boolean} <code>true</code> if an entry exists, otherwise
147 		 *          <code>false</code>
148 		 * 
149 		 * @public
150 		 */
151 		containsValue : function(value, useStrict) {
152 			for (var i = 0, size = keyList.length; i < size; ++i) {
153 				if (useStrict) {
154 					if (map[keyList[i]] === value) {
155 						return true;
156 					}
157 				} else {
158 					if (map[keyList[i]] == value) {
159 						return true;
160 					}
161 				}
162 			}
163 			return false;
164 		},
165 		/**
166 		 * Get the value for a key.
167 		 * 
168 		 * @function get
169 		 * 
170 		 * @param {String}
171 		 *            key the lookup key with was used to store the entry/value.
172 		 * @returns {any} the value for the <tt>key</tt>, or
173 		 *          <code>undefined</code> if the dictionary has no entry for
174 		 *          the <tt>key</tt>.
175 		 * 
176 		 * @public
177 		 */
178 		get : function(key) {
179 			return map[lookupKey(key)];
180 		},
181 		/**
182 		 * Remove an entry from the dictionary.
183 		 * 
184 		 * <p>
185 		 * NOTE that this may execute rather slowly, with O(n).
186 		 * 
187 		 * @function remove
188 		 * 
189 		 * @param {String}
190 		 *            key the lookup key for the entry to remove
191 		 * @returns {Boolean} <code>true</code> if the entry was removed. If
192 		 *          there was no entry for the <tt>key</tt> and nothing was
193 		 *          removed, <code>false</code> is returned.
194 		 * 
195 		 * @public
196 		 */
197 		remove : function(key) {
198 
199 			if (this.containsKey(key)) {
200 
201 				var lKey = lookupKey(key);
202 
203 				// remove from map:
204 				delete map[lKey];
205 
206 				// remove from key-list
207 				for (var i = 0, size = keyList.length; i < size; ++i) {
208 					if (keyList[i] == lKey) {
209 						keyList.splice(i, 1);
210 						break;
211 					}
212 				}
213 				return true;
214 			}
215 
216 			return false;
217 		},
218 		/**
219 		 * Get a list of the keys for all entries in the dictionary.
220 		 * 
221 		 * <p>
222 		 * The returned list has no specific ordering.
223 		 * 
224 		 * <p>
225 		 * NOTE that this may execute rather slowly, with O(n).
226 		 * 
227 		 * <p>
228 		 * NOTE that the returned list is no "view" for the keys, i.e. changes
229 		 * on this list will not be reflected by the dictionary's key-list.
230 		 * 
231 		 * @returns {Array<String>} a list of all keys
232 		 * 
233 		 * @public
234 		 */
235 		getKeys : function() {
236 			var prefixLen = KEY_PREFIX.length;
237 			var size = keyList.length;
238 			var list = new Array(size);
239 			// create copy of keyList with removed key-prefixes:
240 			for (var i = 0; i < size; ++i) {
241 				list[i] = keyList[i].substring(prefixLen);
242 			}
243 			return list;
244 		},
245 		/**
246 		 * Get the size of the dictionary.
247 		 * 
248 		 * @function size
249 		 * @returns {Number} the count of entries in the dictionary
250 		 * @public
251 		 */
252 		size : function() {
253 			return keyList.length;
254 		},
255 		/**
256 		 * Remove all entries from the dictionary.
257 		 * 
258 		 * <p>
259 		 * NOTE that this may execute rather slowly, with O(n).
260 		 * 
261 		 * @function clear
262 		 * @public
263 		 */
264 		clear : function() {
265 			// var size = keyList.length;
266 			// for(var i=0; i < size; ++i){
267 			// delete map[keyList[i]];
268 			// }
269 			// keyList.splice(0, size);
270 			delete map;
271 			map = {};
272 			keyList.splice(0, keyList.length);
273 		}
274 	};
275 };
276 
277 return Dictionary;
278 });
