258 //        	  'define('+moduleNameString+',["semanticInterpreter"],function(semanticInterpreter){\n'

334 /**
335  * Initializes the grammar-engine:
336  * 
337  * loads the template and resolves the engine as initialzed.
338  * 
339  * @param {String} url
340  * 			URL to the template file
341  * @param {GrammarGenerator} jsccGenerator
342  * 			the JS/CC grammar generator instance
343  * @param {Deferred} promise
344  * 			the {@link #deferred} for resolving the generator
345  * 			as initialized.
346  *  
347  * 
348  * @private
349  * @function
350  * @memberOf JsccGenerator#
351  */
352 function _loadTemplate(url, jsccGenerator, promise){
353 	$.ajax({
354 		url: url,
355 		dataType: 'text',
356 		async: true,
357 		success: function(data){
358 			
359 			jsccGenerator.template = data;
360 			
361 			promise.resolve();
362 			
363 		},
364 		error: function(xhr, status, err){
365 			var msg = 'Failed to load grammar template file from "'+templatePath+'": '+status+', ERROR '+err;
366 			console.error(msg);
367 			promise.reject(msg);
368 		}
369 	});
370 }

428 					+ token_name.toLowerCase() + " = {};\n";

431 			
432 			for(var i=0, size = words.length; i < size ; ++i){
433 				if(i > 0){

435 				}
436 				grammar_token += words[i];
437 			}
438 			
439 			grammar_token += "'    " + token_name + " [* " + pref
440 					+ token_name.toLowerCase() + "[%match] = %match; *];\n";
441 			
442 			self.grammar_tokens += grammar_token;
443 		}
444 	},

446 		var self = this;
447 		var utt_index = 0;
448 		var json_utterances =  this.json_grammar_definition.utterances;
449 	
450 		for(var utterance_name in json_utterances){
451 			var utterance_def = json_utterances[utterance_name];
452 			if(utt_index > 0){

466 		var phrases = utterance_def.phrases;
467 		var semantic  = self.doCreateSemanticInterpretationForUtterance(utterance_name, utterance_def);
468 		
469 		for(var index=0,size=phrases.length; index < size; ++index){
470 			if(index > 0){

472 			}

542 	doCreateSemanticInterpretationForPhrase: function(utterance_name, utterance_def, phrase, semantic_as_string){
543 		var phraseList = phrase.split(/\s+/),
544 		length = phraseList.length,
545 		duplicate_helper = {};
546 		
547 		var result = " [* %% = ";
548 		var i = 0;
549 		while (i < length){

551 			result += "%"+i;
552 			if(i < length){
553 				result += " + ' ' + ";
554 			}
555 		}
556 		result += "; var "+utterance_name+"_temp = {}; "+utterance_name+"_temp['phrases'] = {};";
557 		for (i = 0; i < length; i += 1) {
558 			if (typeof(duplicate_helper[phraseList[i]]) == "undefined") {
559 				duplicate_helper[phraseList[i]] = 0;
560 				result += utterance_name+"_temp['phrases']['"+phraseList[i].toLowerCase()+"'] = [];";
561 			} else {

563 			}
