 32 	 * The DialogManager gives access to the most commonly used functions of

 35 	 * <p>

 40 	 * In addition, the DialogEngine is exported as module <code>"dialogEngine"</code> via
 41 	 * RequireJS' <code>define()</code> function.
 42 	 * 
 43 	 * @example
 44 	 * //initialization of inputManager
 45 	 * require('dialogManager').init().then( function(dialogManagerInstance, dialogEngineInstance){
 46 	 * 		//do something
 47 	 * });
 48 	 * 
 49 	 * @name mmir.DialogManager
 50 	 * @static
 51 	 * @class
 52 	 * 
 53 	 * @requires mmir.ControllerManager
 54 	 * @requires mmir.PresentationManager
 55 	 * @requires mmir.ModelManager

 58      * @requires jQuery.Deferred

103 		/**
104 		 * This function raises an event. 
105 		 * 
106 		 * @function raise
107 		 * @param {String} eventName
108 		 * 				The name of the event which is to be raised
109 		 * @param {Object} [eventData] OPTIONAL
110 		 * 				Data belonging to the event
111 		 * @throws {Error} if this function is invoked while the internal

113 		 * 				   is not initialized yet

116 		raise : function(eventName, eventData) {

124 		 * This function performs an action of a controller by calling
125 		 * the method {@link mmir.ControllerManager#perform} of the
126 		 * {@link mmir.ControllerManager}
127 		 * 
128 		 * @function perform

156 		 * This function performs an action of a helper-class for a
157 		 * controller by calling the method
158 		 * {@link mmir.ControllerManager#performHelper} of the
159 		 * {@link mmir.ControllerManager}
160 		 * 
161 		 * @function performHelper
162 		 * @param {String}
163 		 *            ctrlName Name of the controller to which the
164 		 *            helper action belongs
165 		 * @param {String}
166 		 *            helper_method_name Name of the action that should
167 		 *            be performed by the helper
168 		 * @param {Object}
169 		 *            data optional data that can be submitted to the
170 		 *            action
171 		 * @returns {Object} the return object of the performed action
172 		 * @public
173 		 */
174 		performHelper : function(ctrlName, helper_method_name, data) {
175 			
176 			if (arguments.length > 3) {
177 				
178 				return controllerManager.performHelper(
179 						ctrlName, helper_method_name, data, arguments[3]
180 				);
181 			}
182 			else {
183 				
184 				return controllerManager.performHelper(
185 						ctrlName, helper_method_name, data
186 				);
187 			}
188 		},

191 		 * This function displays a dialog of a controller by calling
192 		 * the method {@link mmir.PresentationManager#showDialog} of the

195 		 * @function showDialog

234 		 * @see mmir.PresentationManager#hideWaitDialog

249 		 * @function hideWaitDialog

251 		 * 
252 		 * @see mmir.PresentationManager#hideWaitDialog
253 		 * @see mmir.PresentationManager#showWaitDialog

255 		hideWaitDialog : function() {
256 			presentationManager.hideWaitDialog.apply(presentationManager, arguments);
257 		},
258 
259 		/**
260 		 * This function displays a view of a controller by calling the
261 		 * method {@link mmir.PresentationManager#renderView} of the
262 		 * {@link mmir.PresentationManager}.
263 		 * <br>
264 		 * And after rendering, the function set via #setOnPageRenderedHandler will
265 		 * called in context of the controller instance with arguments:
266 		 * <code>Controller.onPageRenderedFunc(ctrlName, viewName, data)</code>
267 		 * 
268 		 * 
269 		 * @function render
270 		 * @param {String}
271 		 *            ctrlName Name of the controller to which the view
272 		 *            belongs
273 		 * @param {String}
274 		 *            viewName Name of the view that should be rendered
275 		 * @param {Object}
276 		 *            data Optional data that can be submitted to the
277 		 *            generation of the view
278 		 * @public
279 		 */
280 		render : function(ctrlName, viewName, data) {
281 			
282 			presentationManager.renderView(ctrlName, viewName, data);
283 
284 			if (typeof onPageRenderedFunc === 'function') {
285 				var ctrl = controllerManager.getController(ctrlName);
286 				onPageRenderedFunc.call(ctrl, ctrlName, viewName, data);
287 			}
288 		},
289 		/**
290 		 * Get the current on-page-rendered hook function (if it was
291 		 * set).
292 		 * 
293 		 * @function getOnPageRenderedHandler
294 		 * @param {Function}
295 		 *            the onPageRendered handler (NOTE: this may not be
296 		 *            set, i.e. <tt>undefined</tt>)
297 		 */
298 		getOnPageRenderedHandler : function() {
299 			return onPageRenderedFunc;
300 		},
301 		/**
302 		 * Set the on_page_loaded callback function.
303 		 * 
304 		 * If <code>onPageRenderedHook</code> is a function object, it
305 		 * will be executed after a view is rendered and after the
306 		 * view's controller on_page_load function(s) has/have been
307 		 * executed.
308 		 * 
309 		 * <p>
310 		 * This function will be executed after the view's
311 		 * on_page_load()-function.<br>
312 		 * The <code>onPageRenderedHook</code> function takes 3
313 		 * arguments that refer to the parameters with which the
314 		 * render-function was invoked: <br>
315 		 * <code>{String} ctrlName </code> Name of the controller to
316 		 * which the view belongs <br>
317 		 * <code>{String} viewName</code> Name of the view that should
318 		 * be rendered <br>
319 		 * <code>{Object} [data]</code> <em>Optional</em> data that
320 		 * can be submitted to the generation of the view
321 		 * 
322 		 * @function setOnPageRenderedHandler
323 		 * @param {Function}
324 		 *            onPageRenderedHook a callback function that will
325 		 *            be executed after a view was rendered i.e. after a
326 		 *            page was loaded.
327 		 */
328 		setOnPageRenderedHandler : function(onPageRenderedHook) {
329 			onPageRenderedFunc = onPageRenderedHook;
330 		}
331 		
332 	};//END: _instance = {...

336 		init : function() {
337 			
338 			delete this.init;
339 			
340 			//"read" settings from requirejs' config (see mainConfig.js):
341 			var url = module.config().scxmlDoc;
342 			var mode = module.config().mode;
343 			
344 			//create a SCION engine:
345 			var engine = engineConfig(url, mode);
346 			
347 			this._log = Logger.create(module);
348 			engine._log = Logger.create(module.id+'Engine', module.config().logLevel);
349 
350 //			var _self = this;
351 
352 			return $.Deferred(function(theDeferredObj) {
353 				
354 				engine.load().done(function(_engine) {
355 					
356 					_instance.raise = function raise(){
357 						_engine.raise.apply(_engine, arguments);
358 					};
359 					
360 //					mmir.DialogEngine = _engine;
361 //					mmir.DialogEngine.gen('init', _self);
362 					delete _engine.gen;
363 					
364 					//register the DialogeEngine with requirejs as module "dialogEngine":
365 					define("dialogEngine", function(){
366 						return _engine;
367 					});
368 					//immediately load the module-definition:
369 					require(['dialogEngine'], function(){
370 						//signal end of initialization process:
371 						theDeferredObj.resolve(_instance, _engine);	
372 					});
373 				});
374 				
375 			}).promise();
376 			
377 		}//END: init()
378
