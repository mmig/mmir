 29 		initialize: function(callBack){//, mediaManager){//DISABLED this argument is currently un-used -> disabled
 30 			
 31 			/**  @memberOf CordovaAudioOutput# */
 32 			var _pluginName = 'codovaAudioOutput';
 33 			
 34 			//invoke the passed-in initializer-callback and export the public functions:
 35 			callBack({
 36 				/**
 37 				 * @public
 38 				 * @memberOf CordovaAudioOutput.prototype
 39 				 * @see mmir.MediaManager#playWAV
 40 				 */
 41 				playWAV: function(blob, successCallback, failureCallback){
 42 					try {
 43 						var blobURL = window.URL.createObjectURL(blob);
 44 						var my_media = new Media(
 45 								blobURL,
 46 								function(){ 
 47 //									console.log('WAV Audio created');
 48 
 49 									my_media.release();
 50 									if(successCallback){
 51 										successCallback();
 52 									}
 53 								},failureCallback
 54 						);
 55 
 56 						my_media.play();
 57 
 58 					} catch (e){
 59 						if(failureCallback){
 60 							failureCallback(e);
 61 						}
 62 					}
 63 				},
 64 				/**
 65 				 * @public
 66 				 * @memberOf CordovaAudioOutput.prototype
 67 				 * @see mmir.MediaManager#playURL
 68 				 */
 69 				playURL: function(url, successCallback, failureCallback){
 70 					try {
 71 //						console.log(url);
 72 						var my_media = new Media(
 73 								url, 
 74 								function(){ 
 75 //									console.log('Audio played');
 76 
 77 									my_media.release();
 78 									if(successCallback){
 79 										successCallback.apply(my_media,arguments);
 80 									}
 81 								} ,
 82 								failureCallback
 83 						);
 84 
 85 						my_media.play();
 86 					} catch (e){
 87 						if(failureCallback){
 88 							failureCallback.apply(my_media,arguments);
 89 						}
 90 					}
 91 				},
 92 				/**
 93 				 * @public
 94 				 * @memberOf CordovaAudioOutput.prototype
 95 				 * @see mmir.MediaManager#getURLAsAudio
 96 				 */
 97 				getURLAsAudio: function(url, onEnd, failureCallback, onCanPlay){
 98 					
 99 					try {
100 						
101 						/**
102 						 * @private
103 						 * @memberOf AudioCordovaImpl#
104 						 */
105 						var playStatus = 0;
106 						/**
107 						 * @private
108 						 * @memberOf AudioCordovaImpl#
109 						 */
110 						var my_media = new Media(
111 								url
112 								,null //DEBUG: function(){console.log('native onReady CB');}
113 								,failureCallback
114 								,function(status){
115 //									console.debug("media status change "+playStatus+" -> "+status+"  for: "+url);
116 
117 									playStatus = status;
118 
119 									if (status==1){
120 										if (onCanPlay){
121 											onCanPlay.apply(mediaImpl, arguments);
122 											onCanPlay = null;//remove onCanPlay callback after first invocation
123 										}
124 									} 
125 //									else if (status==2){
126 //									console.log("Audio started");
127 //									}
128 //									else if (status==3){
129 //									console.log("Audio paused");
130 //									}
131 									else if(status == 4){
132 										if (onEnd){
133 											onEnd.apply(mediaImpl, arguments);				    					 
134 										}
135 									}
136 								}
137 						);
138 						
139 						/**
140 						 * @private
141 						 * @memberOf AudioCordovaImpl#
142 						 */
143 						var enabled = true;
144 						
145 						
146 						/**
147 						 * The Audio abstraction that is returned by {@link mmir.MediaManager#getURLAsAudio}.
148 						 * 
149 						 * <p>
150 						 * NOTE: when an audio object is not used anymore, its {@link #release} method should
151 						 * 		 be called.
152 						 * 
153 						 * <p>
154 						 * This is the same interface as {@link AudioHtml5Impl}.
155 						 * 
156 						 * @class
157 						 * @name AudioCordovaImpl
158 						 * @public
159 						 */
160 						var mediaImpl = {
161 								/**
162 								 * Play audio.
163 								 * 
164 								 * @public
165 								 * @name play
166 								 * @function
167 								 * @memberOf AudioCordovaImpl.prototype
168 								 */
169 								play: function(){
170 									if (enabled){
171 										my_media.play();
172 									}
173 								},
174 								/**
175 								 * Stop playing audio.
176 								 * 
177 								 * @public
178 								 * @name stop
179 								 * @function
180 								 * @memberOf AudioCordovaImpl.prototype
181 								 */
182 								stop: function(){
183 									//use "manual" stop instead of Cordova's stop
184 									//in order to allow "forgiving" behavior when audio is already stopped
185 									//	-> Cordova's stop() requires the audio to be playing, otherwise an error is thrown/triggered
186 
187 //									console.info('CordovaAudio.stop[state '+playStatus
188 //											+', duration '+my_media.duration
189 //											+', position '+my_media.position
190 ////											+', currentPosition '+my_media.getCurrentPosition()
191 //											+']: '+url);
192 									
193 									//only try to stop if playing and/or paused
194 									if(playStatus == 2 || playStatus == 3){
195 										my_media.stop();
196 									}
197 									
198 //									if(playStatus == 2){//playing
199 //										my_media.stop();
200 //									}
201 //									else if(playStatus == 3){//paused
202 //										my_media.seekTo(0);
203 //									}
204 ////									my_media.stop();
205 								},
206 								/**
207 								 * Enable audio (should only be used internally).
208 								 * 
209 								 * @public
210 								 * @name enable
211 								 * @function
212 								 * @memberOf AudioCordovaImpl.prototype
213 								 */
214 								enable: function(){
215 									enabled = true;
216 								},
217 								/**
218 								 * Disable audio (should only be used internally).
219 								 * 
220 								 * @public
221 								 * @name disable
222 								 * @function
223 								 * @memberOf AudioCordovaImpl.prototype
224 								 */
225 								disable: function(){
226 									if(enabled){
227 										this.stop();
228 										enabled = false;
229 									}
230 								},
231 								/**
232 								 * Release audio: should be called when the audio
233 								 * file is not used any more.
234 								 * 
235 								 * NOTE Android has limited resources available - not releasing resources
236 								 *      may result in not being able to instantiate new (audio) resources.
237 								 * 
238 								 * @public
239 								 * @name release
240 								 * @function
241 								 * @memberOf AudioCordovaImpl.prototype
242 								 */
243 								release: function(){
244 									if(enabled && ! this.isPaused()){
245 										this.stop();
246 									}
247 									enabled= false;
248 									if(my_media){
249 										my_media.release();
250 									}
251 
252 								},
253 								/**
254 								 * Set the volume of this audio file
255 								 * 
256 								 * @param {Number} value
257 								 * 			the new value for the volume:
258 								 * 			a number between [0.0, 1.0]
259 								 * 
260 								 * @public
261 								 * @name setVolume
262 								 * @function
263 								 * @memberOf AudioCordovaImpl.prototype
264 								 */
265 								setVolume: function(value){
266 									if(my_media){
267 										my_media.setVolume(value);
268 									}
269 								},
270 								/**
271 								 * Get the duration of the audio file
272 								 * 
273 								 * @returns {Number} the duration in MS (or -1 if unknown)
274 								 * 
275 								 * @public
276 								 * @name getDuration
277 								 * @function
278 								 * @memberOf AudioCordovaImpl.prototype
279 								 */
280 								getDuration: function(){
281 									if(my_media){
282 										return my_media.duration;
283 									}
284 									return -1;
285 								},
286 								/**
287 								 * Check if audio is currently paused.
288 								 * 
289 								 * NOTE: "paused" is a different status than "stopped".
290 								 * 
291 								 * @returns {Boolean} TRUE if paused, FALSE otherwise
292 								 * 
293 								 * @public
294 								 * @name isPaused
295 								 * @function
296 								 * @memberOf AudioCordovaImpl.prototype
297 								 */
298 								isPaused: function(){
299 									if(my_media){
300 										return playStatus == 3;
301 									}
302 									return false;
303 								},
304 								/**
305 								 * Check if audio is currently enabled
306 								 * 
307 								 * @returns {Boolean} TRUE if enabled
308 								 * 
309 								 * @public
310 								 * @name isEnabled
311 								 * @function
312 								 * @memberOf AudioCordovaImpl.prototype
313 								 */
314 								isEnabled: function(){
315 									return enabled;
316 								}
317 						};
318 
319 						//WORK-AROUND for Android: need to invoke a method on the Media object in
320 						//							order to trigger the on-init callback.
321 						my_media.seekTo(0);
322 
323 
324 						return mediaImpl;
325 
326 					} catch (e){
327 						console.error(e);
328 						if(failureCallback){
329 							failureCallback(e);
330 						}
331 					}
332 				}//END: getURLAsAudio
333 				
334 			});//END: callBack({...
335 		}
336 };
