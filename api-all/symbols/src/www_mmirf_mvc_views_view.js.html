 41 /**
 42  * The View class is a kind of interface-class which gives access to the methods and data of a helper (which itself belongs to a controller)<br>
 43  * Apart from initializing some properties, the constructor also parses the view description and looks for needed helper methods.
 44  * 
 45  * @constructs View
 46  * @param {Object} ctrl 
 47  * 			Controller instance / object
 48  * @param {String} name
 49  * 			Name of the View 
 50  * @param {String} definition
 51  * 			View description, i.e. the raw template code that will be processed.
 52  * 			May be empty: in this case the processed contents must be
 53  * 						  added manually (cf. parser.StorageUtils)
 54  * 
 55  * @requires if param definition is NOT empty: parser.RenderUtils (must be loaded beforehand via <code>require(["renderUtils"]...</code>)
 56  * @requires if param definition is NOT empty: parser.ParseUtils (must be loaded beforehand via <code>require(["parseUtils"]...</code>)
 57  * 
 58  * @category core
 59  */
 60  function View(ctrl, name, definition){
 61     
 62 //	console.log("[View] '" + name + "' loaded.");
 63 	 if(definition){
 64 	    // remove HTML comments from View
 65 	    definition = definition.replace(commonUtils.regexHTMLComment, '');
 66 	}
 67 	 
 68     /**
 69      * The controller to which this view belongs.
 70      * 
 71      * @property controller
 72      * @type Object
 73      * @public
 74      */
 75     this.controller = ctrl;
 76 	
 77     /**
 78      * The description of the view in eHTML.
 79      * 
 80      * @property def
 81      * @type String
 82      * @public
 83      */
 84     this.def = definition;
 85 	
 86     /**
 87      * The name of the view.
 88      * 
 89      * @property name
 90      * @type String
 91      * @public
 92      */
 93     this.name = name;
 94    
 95 	
 96     /**
 97      * An array of all the views {@link mmir.ContentElement} objects.<br>
 98      * 
 99      * @property contentFors
100      * @type Array
101      * @public
102      */
103     this.contentFors = new Array();
104    
105 	
106     /**
107      *
108      * An array of all names of the for the view required helper methods.
109      * 
110      * @deprecated helper methods must now explicitly called in template definition (using syntax <code>@helper(name,args)</code>)
111      * 
112      * @property helperMethods
113      * @type Array
114      * @public
115      */
116     this.helperMethods = new Array();
117     
118     if(this.def){
119 	    
120 	    var parserUtils = require('parseUtils');
121 	    var renderUtils = require('renderUtils');
122 	    
123 	    var parseResult = parserUtils.parse(this.def, this);
124 	    
125 	    for(var i=0, size = parseResult.contentFors.length; i < size ; ++i){
126 	    	this.contentFors.push(new ContentElement(parseResult.contentFors[i], this, parserUtils, renderUtils));
127 	    }
128     }
129     
130 };
131

137  */

139     return this.def;
140 };

164 /**
165  * Gets a specific {@link mmir.ContentElement} object by name. 
166  * 
167  * @function getContentElement
168  * @param {String} name Name of the ContentElement object
169  * @returns {object} The wanted ContentElement object or null
170  */
171 View.prototype.getContentElement = function( name){
172     
173     for(var i=0, size = this.contentFors.length; i < size ; ++i){
174     	if(this.contentFors[i].getName() == name){
175     		return this.contentFors[i];/////////////////////// EARLY EXIT /////////////////////////////
176     	}
177     }
178     
179     return null;
180     
181 };

288 /**
289  * Gets an array of all helper methods. 
290  * 
291  * @deprecated helper methods must now explicitly called in template definition (using syntax <code>@helper(name,args)</code>)
292  * 
293  * @function getHelperMethods
294  * @returns {Array} Array of all helper methods
295  */
296 View.prototype.getHelperMethods = function(){
297 	return this.helperMethods;
298 };
299 
300 /**
301  * Executes all helper methods that were specified / referenced in the view; with **data** as parameter.
302  * 
303  * @deprecated helper methods must now explicitly called in template definition (using syntax <code>@helper(name,args)</code>)
304  * 
305  * @function executeHelperMethods
306  * @param {Object} data Parameter to pass to the helper methods
307  */
308 View.prototype.executeHelperMethods = function(data){
309 	for(var i=0, size = this.getHelperMethods().length; i < size ; ++i){
310 		this.controller.performHelper(this.getHelperMethods()[i], data);
311     }
312 };
313 
314 return View;
