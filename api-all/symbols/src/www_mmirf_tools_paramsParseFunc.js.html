  2 define(function() {
  3 	
  4 	////////////////////// exports ///////////////////////
  5 	
  6 	/**
  7      * Convert parameter-part of an URL to a "dictionary", containing
  8      * the parameter keys and values
  9      * 
 10      * @example <code>?id=5&name=heinz&name=kunz</code> →
 11      *          <code>dict['id']=5, dict['name'] = ['heinz', 'kunz']</code>
 12      * 
 13      * The returnd "dictionary" has the following functions:
 14      * <ul>
 15      * <li>has(String key): returns <code>true</code> if the
 16      * dictionary contains an entry for <code>key</code></li>
 17      * <li>isMultiple(String key): returns <code>true</code> if the
 18      * entry for <code>key</code> is an Array (i.e. the URL contained
 19      * multiple values for this key)</li>
 20      * <li>getKeys(): returns an Array with the keys (String) for all
 21      * entries</li>
 22      * </ul>
 23      * 
 24      * @function parseParamsToDictionary
 25      * @param {String}
 26      *            the parameter-part of the URL, i.e. <code>&...</code>
 27      * @return {Object} an "dictionary" for the parameters
 28      * @public
 29      */
 30     function parseParamsToDictionary(urlParamsPartStrings) {
 31 		var dict = new Object();
 32 
 33 		dict.has = function(key) {
 34 			return typeof dict[key] !== 'undefined';
 35 		};
 36 		dict.isMultiple = function(key) {
 37 			// use not-allowed-as-part-of-parameter-name char & as
 38 			// prefix for meta-data 'isMultiple' on field 'key':
 39 			return typeof dict['&' + key] !== 'undefined' && dict['&' + key] === true;
 40 		};
 41 		// use not-allowed-as-part-of-parameter-name char & as prefix
 42 		// for meta-data 'keys-list':
 43 		dict['&&keys'] = new Array();
 44 		dict.getKeys = function() {
 45 			return dict['&&keys'];
 46 		};
 47 
 48 		if (urlParamsPartStrings) {
 49 			if (typeof urlParamsPartStrings !== 'string') {
 50 				urlParamsPartStrings = urlParamsPartStrings.toString();
 51 			}
 52 			if (urlParamsPartStrings.length < 1) {
 53 				return dict;//////////////////////// EARLY EXIT /////////////////////////////
 54 			}
 55 			if (urlParamsPartStrings.charAt(0) === '?') {
 56 				urlParamsPartStrings = urlParamsPartStrings.substring(1);
 57 			}
 58 
 59 			var params = urlParamsPartStrings.split('&');
 60 			var cur = null;
 61 			var keyValue = null;
 62 			var theKey = null;
 63 			var theValue = null;
 64 			for (var i_params = 0, size_params = params.length; i_params < size_params; ++i_params) {
 65 
 66 				cur = params[i_params];
 67 
 68 				// "parse" parameter into key & value:
 69 				keyValue = cur.split('=');
 70 				theKey = keyValue[0];
 71 				if (keyValue.length > 1) {
 72 					theValue = keyValue[1];
 73 				}
 74 				else {
 75 					theValue = null;
 76 				}
 77 
 78 				// create entry in dict for the parameter
 79 				if (dict.has(theKey)) {
 80 
 81 					if (dict.isMultiple(theKey)) {
 82 						dict[theKey].push(theValue);
 83 					}
 84 					else {
 85 						// entry already exist, but is not multiple (=Array) yet:
 86 						var arr = new Array(2);
 87 						arr[0] = dict[theKey];
 88 						arr[1] = theValue;
 89 						dict[theKey] = arr;
 90 						dict['&' + theKey] = true;
 91 					}
 92 				}
 93 				else {
 94 					dict[theKey] = theValue;
 95 					dict['&&keys'].push(theKey);
 96 				}
 97 			}
 98 		}
 99 		return dict;
100     }
101 
102     return parseParamsToDictionary;
103 	
104 });
