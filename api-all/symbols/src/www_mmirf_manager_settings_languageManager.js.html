183 		    /**
184 		     * Request grammar for the provided language.
185 		     * 
186 		     * If there is no grammar available for the requested language, no new
187 		     * grammar is set.
188 		     * 
189 		     * A grammar is available, if at least one of the following is true for the
190 		     * requested language
191 		     * <ul>
192 		     * <li>there exists a JSON grammar file (with correct name and at the
193 		     * 		correct location)</li>
194 		     * <li>there exists a compiled JavaScript grammar file (with correct name
195 		     * 		and at the correct location)</li>
196 		     * </ul>
197 		     * 
198 		     * TODO document location for JSON and JavaScript grammar files
199 		     * 
200 		     * @function requestGrammar
201 		     * @param {String}
202 		     *            lang The language of the grammar which should be loaded.
203 		     * @returns {String} The current grammar language
204 		     * @async
205 		     * @private
206 		     * 
207 		     * @memberOf LanguageManager#
208 		     */
209 		    function requestGrammar(lang, doSetNextBestAlternative) {
210 
211 		        if (semanticInterpreter.hasGrammar(lang) || doCheckExistsGrammar(lang)) {
212 		            semanticInterpreter.setCurrentGrammar(lang);
213 		            return lang;
214 		        }
215 				else if (doSetNextBestAlternative) {
216 		            // try to find a language, for which a grammar is available
217 		            var grammarLang = null;
218 		            if (languages.length > 0) {
219 		                // first: try to find a language with COMPILED grammar
220 		                for ( var i = 0, size = languages.length; i < size; ++i) {
221 		                    grammarLang = languages[i];
222 		                    if (semanticInterpreter.hasGrammar(grammarLang)) {
223 		                        break;
224 		                    }
225 							else {
226 		                        grammarLang = null;
227 		                    }
228 		                }
229 
230 		                // ... otherwise: try to find a language with JSON grammar:
231 		                if (!grammarLang) {
232 		                    for ( var i = 0, size = languages.length; i < size; ++i) {
233 		                        grammarLang = languages[i];
234 		                        if (doCheckExistsGrammar(grammarLang)) {
235 		                            break;
236 		                        }
237 								else {
238 		                            grammarLang = null;
239 		                        }
240 		                    }
241 		                }
242 		            }
243 
244 		            if (grammarLang) {
245 		                logger.warn('Could not find grammar for selected language ' + lang + ', using grammar for language ' + grammarLang + ' instead.');
246 		                semanticInterpreter.setCurrentGrammar(grammarLang);
247 		            }
248 					else {
249 		                logger.warn('Could not find any grammar for one of [' + languages.join(', ') + '], disabling SemanticInterpret.');
250 		                semanticInterpreter.setEnabled(false);
251 		            }
252 		        }
253 
254 		        return semanticInterpreter.getCurrentGrammar();
255 		    }
256 		    /**
257 		     * Loads the speech-configuration for the provided language and updates the current
258 		     * language.
259 		     * 
260 		     * @function loadSpeechConfig

269 		    function loadSpeechConfig(lang) {
270 
271 		        if (lang && currentLanguage != lang) {
272 		            currentLanguage = lang;
273 		        }
274 		        var path = constants.getLanguagePath() + lang + "/" + constants.getSpeechConfigFileName();
275 		        $.ajax({
276 		            async : false,
277 		            dataType : "json",
278 		            url : path,
279 		            success : function(data) {
280 		                
281 		            	if(logger.isVerbose()) logger.v("[LanguageManager] Success. " + data);
282 		                
283 		            	currentSpeechConfig = data;// jQuery.parseJSON(data);
284 		                
285 		                if(logger.isVerbose()) logger.v("[LanguageManager] " + JSON.stringify(dictionary));
286 		            },
287 		            error : function(xhr, statusStr, error) {
288 		                logger.error("[LanguageManager] Error loading speech configuration from \""+path+"\": " + error? error.stack? error.stack : error : ''); // error
289 		            }
290 		        });
291 		        return currentLanguage;
292 		    }

326 		    /**
327 		     * Translates a keyword using the current dictionary and returns the
328 		     * translation.
329 		     * 
330 		     * @function internalGetText
331 		     * @param {String}
332 		     *            textVarName The keyword which should be looked up
333 		     * @returns {String} the translation of the keyword
334 		     * @private
335 		     * 
336 		     * @memberOf LanguageManager#
337 		     */
338 		    function internalGetText(textVarName) {
339 		        var translated = "";
340 		        if (dictionary[textVarName] && dictionary[textVarName].length > 0) {
341 		            translated = dictionary[textVarName];
342 		        }
343 				else if (textVarName === keyword_current_language){
344 		            translated = currentLanguage;
345 		        }
346 				else {
347 		            translated = "undefined";
348 		            logger.warn("[Dictionary] '" + textVarName + "' not found in " + JSON.stringify(dictionary));
349 		        }
350 		        return translated;
351 		    }

537 		            /**
538 		             * Chooses a language for the application.
539 		             * 
540 		             * <p>
541 		             * The language selection is done as follows:
542 		             * <ol>
543 		             * <li>check if a default language exists<br>
544 		             * if it does and if both (!) grammar and dictionary exist for this
545 		             * language, return this language </li>
546 		             * <li>walk through all languages alphabetically
547 		             * <ol>
548 		             * <li>if for a language both (!) grammar and dictionary exist,
549 		             * return this language memorize the first language with a grammar
550 		             * (do not care, if a dictionary exists) </li>
551 		             * </ol>
552 		             * <li>test if a grammar exists for the default language - do not
553 		             * care about dictionaries - if it does, return the default language
554 		             * </li>
555 		             * <li>If a language was found (in Step 2.1) return this language
556 		             * </li>
557 		             * <li>If still no language is returned take the default language
558 		             * if it has a dictionary </li>
559 		             * <li>If a language exists, take it (the first one) </li>
560 		             * <li>Take the default language - no matter what </li>
561 		             * </ol>
562 		             * 
563 		             * @function determineLanguage
564 		             * @returns {String} The determined language
565 		             * @public
566 		             */
567 		            determineLanguage : function(lang) {
568 		                var tempLanguage = lang;
569 		                var firstLanguageWithGrammar = null;
570 
571 		                // first check, if language - given in parameter - exists
572 		                if (tempLanguage != null) {
573 		                    // check if both grammar and dictionary exist for given
574 		                    // language
575 		                    if (instance.existsGrammar(tempLanguage) && instance.existsDictionary(tempLanguage)) {
576 		                        return tempLanguage;
577 		                    }
578 		                }
579 
580 		                tempLanguage = constants.getLanguage();
581 		                // then check, if default language exists
582 		                if (tempLanguage != null) {
583 		                    // check if both grammar and dictionary exist for default
584 		                    // language
585 		                    if (instance.existsGrammar(tempLanguage) && instance.existsDictionary(tempLanguage)) {
586 		                        return tempLanguage;
587 		                    }
588 		                }
589 		                // walk through the languages alphabetically
590 		                for ( var i = 0; i < languages.length; i++) {
591 		                    tempLanguage = languages[i];
592 		                    // check if a grammar and dictionary exists for every
593 		                    // language
594 		                    if (instance.existsGrammar(tempLanguage)) {
595 
596 		                        // memorize the first language with a grammar (for
597 		                        // later)
598 		                        if (firstLanguageWithGrammar == null) {
599 		                            firstLanguageWithGrammar = tempLanguage;
600 		                        }
601 
602 		                        if (instance.existsDictionary(tempLanguage)) {
603 		                            return tempLanguage;
604 		                        }
605 		                    }
606 		                }
607 
608 		                // still no language found - take the default language and test
609 		                // if a grammar exists
610 		                tempLanguage = constants.getLanguage();
611 		                if (tempLanguage != null) {
612 		                    // check if both grammar and dictionary exist for default
613 		                    // language
614 		                    if (instance.existsGrammar(tempLanguage)) {
615 		                        return tempLanguage;
616 		                    } else if (firstLanguageWithGrammar != null) {
617 		                        return firstLanguageWithGrammar;
618 		                    } else if (instance.existsDictionary(tempLanguage)) {
619 		                        return tempLanguage;
620 		                    }
621 		                }
622 
623 		                // still no language - take the first one
624 		                tempLanguage = languages[0];
625 		                if (tempLanguage != null) {
626 		                    return tempLanguage;
627 		                }
628 
629 		                return constants.getLanguage();
630 		            },

720 		            getLanguageConfig : function(pluginId, feature, separator) {
721 		                
722 		                //if nothing is specfied:
723 		            	//	return default language-setting
724 		                if(typeof pluginId === 'undefined'){
725 		                	return currentSpeechConfig.language; /////////// EARLY EXIT ///////////////
726 		                }
727 		                
728 		                //ASSERT pluginId is defined
729 		                
730 		                //default feature is language
731 		                if(typeof feature === 'undefined'){
732 		                	feature = 'language';
733 		                }
734 
735 		                var value;
736 		                if(currentSpeechConfig.plugins && currentSpeechConfig.plugins[pluginId] && typeof currentSpeechConfig.plugins[pluginId][feature] !== 'undefined'){
737 		                	//if there is a plugin-specific setting for this feature
738 		                	value = currentSpeechConfig.plugins[pluginId][feature];
739 		                }
740 		                else if(feature !== 'plugins' && typeof currentSpeechConfig[feature] !== 'undefined'){
741 		                	//otherwise take the default setting (NOTE: the name "plugins" is not allowed for features!)
742 		                	value = currentSpeechConfig[feature];
743 		                }
744 		                
745 		                //if there is a separator specified: replace default separator '-' with this one
746 		                if(value && typeof separator !== 'undefined'){
747 		                	value.replace(/-/, separator);
748 		                }
749 		                
750 		                return value;
751 		            }
