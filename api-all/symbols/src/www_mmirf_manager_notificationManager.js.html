160     	 * Factory function for creating "sounds objects",
161     	 * i.e. extend the basic Audio objects with needed functions/properties
162     	 * 
163     	 * @private
164     	 * @function
165     	 * 
166     	 * @param {AudioObject} audioObj
167     	 * @param {String} name
168     	 * 
169     	 * @returns {NotificationSound} the extended audio object, i.e. a NotificationSound
170     	 * 
171     	 * @memberOf mmir.NotificationManager#
172     	 */
173     	var initNotificationSound = function(audioObj, name){
174     		audioObj.name = name;
175     		audioObj.setVolume(beepVolume);
176 			audioObj.isNotificationPlaying = false;
177 			audioObj.repeatNotification = 0;
178 			audioObj.playNotification = function(repeatNTimes){
179 				
180 //				console.debug('isPlaying: '+this.isNotificationPlaying+', repeat: '+this.repeatNotification+',  args: '+repeatNTimes+'');
181 				
182 				//"blocking mode": only re-start, if not already playing
183 				if(!this.isNotificationPlaying){
184 					this.repeatNotification = repeatNTimes ? repeatNTimes : 0;
185 				}
186 				
187 				if( this.repeatNotification < 1){
188 					//end recurusion
189 					this.isNotificationPlaying = false;
190 					this.repeatNotification = 0;
191 				}
192 				else {
193 					this.isNotificationPlaying = true;
194 					--this.repeatNotification;
195 //					this.stop();
196 					this.play();
197 				}
198 			};
199 			
200 			audioObj.setCallbacks = function(onFinished, onError){
201 				this.onFinishedListener = onFinished;
202 				this.onErrorListener = onError;
203 			};
204 			audioObj.clearCallbacks = function(){
205 				this.onFinishedListener = null;
206 				this.onErrorListener = null;
207 			};
208 			audioObj.fireFinished = function(){
209 				
210 				var tempOnFinishedListener = this.onFinishedListener;
211 				//clear callbacks
212 				// NOTE: do this before triggering callback, in case the callback re-plays the notification with new callbacks!
213 				//       (if we would clear callbacks after invocation, we would delete the new callbacks!)
214 				this.clearCallbacks();
215 				if(tempOnFinishedListener){
216 					tempOnFinishedListener();
217 				}
218 			};
219 			audioObj.fireError = function(error){
220 				
221 				var tempOnErrorListener = this.onErrorListener;
222 				//clear callbacks
223 				// NOTE: do this before triggering callback, in case the callback re-plays the notification with new callbacks!
224 				//       (if we would clear callbacks after invocation, we would delete the new callbacks!)
225 				this.clearCallbacks();
226 				
227 				//create error message with details
228 				var id;
229 				if(this.name){
230 					var entry = soundMap.get(this.name);
231 					id = '"' + this.name + '" -> ' + (entry? '"'+entry.url+'"' : 'UNDEF'); 
232 				}
233 				else {
234 					id = '"BEEP" -> "'+constants.getBeepUrl()+'"'; 
235 				}
236 				var msg = 'Notification: Error playing the sound for notification '+id;
237 				
238 				//create Error object if necessary
239 				if(!error){
240 					error = new Error(msg);
241 					msg = null;
242 				}
243 				
244 				
245 				if(tempOnErrorListener){
246 					tempOnErrorListener(error, msg);
247 				}
248 				else {
249 					//if no callback: print debug output in error stream:
250 					console.error( (msg? msg + ': ' : '') + error);
251 				}
252 			};
253 			
254 			return audioObj;
255     	};

289     	 * @param {String} theUrl
290     	 * @param {Boolean} isKeepOnPause
291     	 *

302     	/**
303     	 * Helper for retrieving an existing sound from
304     	 * the {@link #soundMap}.
305     	 * 
306     	 * Initializes the sound if necessary.
307     	 * 
308     	 * @private
309     	 * @function
310     	 * 
311     	 * @param {String} name
312     	 * @param {Function} onErrorCallback
313     	 * 
314     	 * @memberOf mmir.NotificationManager#
315     	 */
316     	function doGetSoundFromMap(name, onErrorCallback){
317     		var audioObj = null;
318     		var audioUrl = null;
319     		var keepOnPause = false;
320     		
321     		//if no name: use default beep
322     		if(!name){
323     			audioObj = beepAudio;
324     			audioUrl = constants.getBeepUrl();
325     		}
326     		else {
327     			//retrieve information for sound
328     			var soundEntry = soundMap.get(name);
329     			if(soundEntry === INIT){
330     				//TODO set repeat-times?
331     				
332     				//sound is still initializing -> return
333     				return null; ////////////////////// EARLY EXIT //////////////////////
334     			}
335     			
336     			if(!soundEntry){
337     				var errMsg = 'mmir.NotificationManager: no sound "'+name+'" initialized!';
338     				if(onErrorCallback){
339     					onErrorCallback(errMsg);
340     				}
341     				else {
342     					console.error(errMsg);
343     				}
344 //    				throw new Error(errMsg);
345     				return null; ////////////////////// EARLY EXIT //////////////////////
346     			}
347     			
348     			audioObj = soundEntry.audio;//<- may be null
349     			audioUrl = soundEntry.url;//<- must NOT be null
350     			keepOnPause = soundEntry.isKeepOnPause? true : false;
351     		}
352     		
353     		return {
354     			sound: audioObj,
355     			url: audioUrl,
356     			isKeepOnPause: keepOnPause
357     		};
358     	}

360     	/**
361     	 * Helper for playing a registered notification sound.
362     	 * 
363     	 * Initializes the sound if necessary.
364     	 * 
365     	 * @private
366     	 * @function
367     	 * 
368     	 * @param {String} name
369     	 * 				ID of the sound
370     	 * @param {Number} times 
371     	 * @param {Function} onFinishedCallback
372     	 * @param {Function} onErrorCallback
373     	 * 
374     	 * @memberOf mmir.NotificationManager#
375     	 */
376     	function playAudioSound(name, times, onFinishedCallback, onErrorCallback){
377     		
378     		var soundEntry = doGetSoundFromMap(name, onErrorCallback);
379     		
380     		if(soundEntry === null){
381     			//could not retrieve sound-object
382     			// (error callback will already have been invoked, so just return)
383     			return;/////////////////////// EARYL EXIT ///////////
384     		}
385     		
386     		var audioObj = soundEntry.sound;
387     		var audioUrl = soundEntry.url;
388     		var isKeepOnPause = soundEntry.isKeepOnPause;
389 
390     		//create audio-object, if not existing yet
391     		if(audioObj === null){
392     			
393     			if(name){
394     				soundMap.put(name, INIT);
395     			}
396 
397     			audioObj = createAudio(
398     					audioUrl,
399     					function onFinished(){ 
400     						this.playNotification();
401 
402     						audioObj.fireFinished();
403     					}, 
404     					function onError(e){ 
405     						if(name) {
406     							soundMap.remove(name);
407     						};
408     						
409     						if(audioObj){
410     							audioObj.fireError(e);
411     						}
412     						else {
413     							if(onErrorCallback){
414     								onErrorCallback();
415     							}
416     							else {
417     								console.error('Notification: Error playing the sound from "'+audioUrl+'": '+e);
418     							}
419     						}
420     					},
421     					function onInit(){ 
422     						
423     						//FIX for Android/Cordova: return-value of createAudio will not set audioObj "fast enough"
424     						// 						   (i.e. may not use async-initialization, depending on where the audio-file is located)
425     						//						   ... in order to be able to use keep variable audioObj useful -> do assignment now/here
426     						audioObj = this;
427 
428     	    				initNotificationSound(audioObj, name);
429     	    				audioObj.setCallbacks(onFinishedCallback, onErrorCallback);
430     	    				
431     	    	    		//if no name: assume default beep
432     	    				if(!name){
433     	    					beepAudio = audioObj;
434     	    	    		}
435     	    				else {
436     	    					var theEntry = {url: audioUrl, audio: audioObj};
437     	    					if(isKeepOnPause){
438     	    						theEntry.isKeepOnPause = true;
439     	    					}
440     	    					soundMap.put(name, theEntry);
441     	    				}
442     	    				
443     						audioObj.playNotification(times);
444     	    				
445     	    			}
446     			);
447     			
448 //    			//FIXME this is a QUICK-FIX:
449 //    			//		Android needs invocation of a media-method, before it triggers the on-init callback.
450 //    			//		We need to do this here, not within the
451 //    			if(isCordovaEnv){
452 //    				audioObj.init();
453 //    			}
454     		}
455     		else {
456 				audioObj.setCallbacks(onFinishedCallback, onErrorCallback);
457     			audioObj.playNotification(times);
458     		}
459     		
460     	}

462     	/**
463     	 * Helper for stop playing a registered notification sound.
464     	 * 
465     	 * Initializes the sound if necessary.
466     	 * 
467     	 * @private
468     	 * @function
469     	 * 
470     	 * @param {String} name
471     	 * 				ID of the sound
472     	 * @param {Function} onFinishedCallback
473     	 * @param {Function} onErrorCallback
474     	 * 
475     	 * @memberOf mmir.NotificationManager#
476     	 */
477     	function stopAudioSound(name, onFinishedCallback, onErrorCallback){
478     		
479     		var soundEntry = doGetSoundFromMap(name, onErrorCallback);
480     		
481 
482 //			console.error('Notification: invoked stop on notification-sound '+name+' -> '+JSON.stringify(soundEntry));//FIXM debug
483     		
484     		if(soundEntry === null){
485     			//could not retrieve sound-object
486     			// (error callback will already have been invoked, so just return)
487     			return;/////////////////////// EARYL EXIT ///////////
488     		}
489     		
490     		var audioObj = soundEntry.sound;
491     		//NOTE audioObj may be null, e.g. if sound is still initializing.
492     		
493     		if(audioObj != null){
494     			if(audioObj.repeatNotification > 0)
495     				audioObj.repeatNotification = 0;
496     			if(audioObj.isNotificationPlaying === true)
497     				audioObj.isNotificationPlaying = false;
498     			if(audioObj.stop){
499 //  				console.error('Notification: stopping notification-sound -> '+name);//FIXM debug
500     				audioObj.stop();
501     			}
502     		}
503     		
504     		if(onFinishedCallback){
505     			onFinishedCallback.call(audioObj);
506     		}
507     		
508     	};
509     	
510     	//on Android: release resources on pause/exit, since they are limited
511     	if(isCordovaEnv){
512     		
513     		document.addEventListener("resume", function(event){
514     			//initialize beep sound:
515         		playAudioSound(null, 0);
516     		});
517     		
518     		document.addEventListener(
519     				"pause", 
520     				function(event){
521     					
522     					//use temporal variable for minimizing concurrency problems
523     					var temp;
524     					
525     					if(beepAudio !== null){
526     						
527     						temp = beepAudio;
528     						beepAudio = null;
529     						temp.release();
530     						
531     						console.info('Notification: released media resources for beep.');
532     					}
533     					
534     					var keys = soundMap.getKeys();
535     					for(var i = keys.length - 1; i >= 0; --i){
536     						
537     						var entry = soundMap.get(keys[i]);
538 	    					if(entry !== null && entry != INIT && ! entry.isKeepOnPause){
539 	    						
540 	    						temp = entry.audio;
541 	    						
542 	    						//audio may not be initialized yet:
543 	    						if(temp != null){
544 		    						entry.audio = null;
545 		    						temp.release();
546 	    						}
547 	    						
548 	    						console.info('Notification: released media resources for '+entry.url);
549 	    					}
550     					}
551     				},
552     				false
553     		);
554     	}

646             		throw new TypeError('argument vol (value: '+vol+') must be a number, but is instead: '+(typeof vol));
647             	}
648             	
649             	if(vol !== beepVolume){
650             		
651             		//set volume for beep notification
652 	            	beepVolume = vol;
653 	            	if(beepAudio){
654 	            		beepAudio.setVolume(beepVolume);
655 	            	}
656 	            	
657 	            	//set volume for notification sounds
658 	            	var keys = soundMap.getKeys();
659 	            	var entry = null;
660 	            	for(var i = 0, size = keys.length; i < size; ++i){
661 	            		entry = soundMap.get(keys[i]);
662 	            		if(entry.audio !== null){
663 	            			entry.audio.setVolume(vol);
664 	            		}
665 	            	}
666 	            	
667             	}
668             }

708             }
