 33    * Initializes the MMIR framework:
 34    * triggers {@link mmir.ready} when initialization has finished.

 38    * Otherwise initialization starts when the <code>domready</code> event was fired
 39    * (using jQuery's ready function).

 43    * @exports main as mmir.main
 44    * 
 45    * @requires jQuery.Deferred
 46    * 
 47    * @requires jQuery#selector	(for initial CSS/script injections; TODO change mechanism)
 48    * @requires jQuery.is			(for initial script injections; TODO change mechanism)
 49    * @requires jQuery.append		(for initial CSS injections; TODO change mechanism)

 51    */

117 				var checkInitCompleted = function(){

120 							&&	isModelsLoaded
121 							&& 	isVisualsLoaded 
122 							&&	isInputManagerLoaded
123 							&&	isDialogManagerLoaded
124 							&&	isSemanticsLoaded
125 					){
126 						//get additional configuration for requirejs
127 						//from configuration.json:
128 						// -> if property "config" is set, apply it as requirejs-config
129 						//    before signaling READY
130 						// EXAMPLE:
131 						// the following entry (in config/configuration.json) would add

143 						}
144 						
145 						//"give signal" that the framework is now initialized / ready
146 						mmir.setInitialized();
147 					}
148 				};
149

152 					isSemanticsLoaded = true;
153 					
154 					mmir.SemanticInterpreter = semanticInterpreter;
155 					var grammarEngine = configurationManager.get('grammarCompiler');
156 					if(grammarEngine){
157 						semanticInterpreter.setGrammarEngine(grammarEngine);
158 					}
159 					
160 					checkInitCompleted();
161 				});
162 
163 				// start the MediaManager
164 				mediaManager.init().then(function() {
165 					isMediaManagerLoaded = true;
166 					
167 					//initialize BEEP notification (after MediaManager was initialized)
168 					notificationManager.initBeep();
169 					
170 					mmir.MediaManager = mediaManager;
171 					checkInitCompleted();
172 				});
173 				
174 				//TODO models may access views etc. during their initialization
175 				//	   --> there should be a way to configure startup, so that models may only be loaded, after everything else was loaded
176 				modelManager.init().then(function(){
177 					isModelsLoaded = true;
178 					mmir.ModelManager = modelManager;
179 					checkInitCompleted();
180 				});
181 				
182 				presentationManager.init().then(function(){
183 					
184 					//FIXME impl. mechanism where this is done for each view/layout rendering 
185 					//   (i.e. in presentationManager's rendering function and not here)
186 					//
187 					//initialize with default layout contents:
188 					var headerContents = $( presentationManager.getLayout(null, true).getHeaderContents() );
189 					//NOTE: need to handle scripts separately, since some browsers may refuse to "simply append" script TAGs...
190 					var scriptList = [];
191 					var stylesheetList = headerContents.filter(function(index){
192 						var tis = $(this);
193 						if( tis.is('script') ){
194 							scriptList.push(tis.attr('src'));
195 							return false;
196 						}
197 						return true;
198 					});
199 					$("head").append( stylesheetList );
200 					commonUtils.loadImpl(scriptList, true)//load serially, since scripts may depend on each other; TODO should processing wait, until these scripts have been finished! (i.e. add callbacks etc.?)
201 							
202 						.then(function(){//<- need to wait until all referenced scripts form LAYOUT have been loaded (may be used/required when views get rendered)
203 
204 								isVisualsLoaded = true;//<- set to "LOADED" after all scripts have been loaded / TODO impl. better mechanism for including application-scripts...
205 					
206 								mmir.PresentationManager = presentationManager;
207 								checkInitCompleted();
208 					});
209 				});
210 				//TODO handle reject/fail of the presentationManager's init-Promise!
211 				
212 				dialogManager.init().then(function(_dlgMng, _dialogEngine){
213 					isDialogManagerLoaded = true;
214 					mmir.DialogManager = dialogManager;
215 					mmir.DialogEngine = _dialogEngine;
216 					checkInitCompleted();
217 				});
218 				
219 				inputManager.init().then(function(_inputMng, _inputEngine){
220 					isInputManagerLoaded = true;
221 					mmir.InputManager = inputManager;
222 					mmir.InputEngine  = _inputEngine; 
223 					checkInitCompleted();
224 				});
225 				
226 				
227 				
228 			});
229 			
230 
231 	};//END: mainInit(){...
