 28 /**

 31 
 32 define(['parsingResult', 'storageUtils'],
 33 	/**
 34 	 * @name YieldDeclaration
 35 	 * @class
 36 	 */
 37 	function(
 38 		ParsingResult, parser
 39 ) {
 40 	/** @scope YieldDeclaration.prototype */
 41 	/**
 42 	 * #@+
 43 	 * @memberOf YieldDeclaration.prototype
 44 	 */
 45 	
 46 	/**
 47 	 * The YieldDeclaration class holds the name of the yield-declaration (which is a place-holder for the contentFor-fields and is used in the layouts: content, header, footer, dialogs, ...)
 48 	 * and its starting and ending position within the content-definition.
 49 	 * 
 50 	 * @constructs YieldDeclaration
 51 	 * @param {Object} parsingElement with properties <code>name</code> {String}, <code>start</code> {Integer}, <code>end</code> {Integer}
 52 	 * @param {Integer} contentAreaType the type of the content area within the layout that this yield-declaration refers to (e.g. ViewConstants.CONTENT_AREA_BODY )
 53 	 * @category core
 54 	 * 
 55 	 * borrows the getValue function from ParsingResult:
 56 	 * 	@requires mmir.parser.ParsingResult#getValue
 57 	 * 
 58 	 */ 
 59 	function YieldDeclaration(parsingElement, contentAreaType){
 60 		
 61 		if(parsingElement){
 62 			this.name     = parsingElement.name;
 63 			this.nameType = parsingElement.nameType;
 64 			
 65 			this.start    = parsingElement.start;
 66 			this.end      = parsingElement.end;
 67 		}
 68 		
 69 		this.contentAreaType = contentAreaType;
 70 	    
 71 		return this;
 72 	}
 73 
 74 
 75 
 76 	/**
 77 	 * Gets the name of a {@link mmir.YieldDeclaration} object (e.g. content, header, footer, dialogs, ...).
 78 	 * 
 79 	 * @function getName
 80 	 * @returns {String} Name - used by yield tags in layout
 81 	 * @public
 82 	 */ 
 83 	YieldDeclaration.prototype.getName = function(){
 84 	    return this.name;
 85 	};
 86 
 87 	YieldDeclaration.prototype.getNameType = function(){
 88 	    return this.nameType;
 89 	};
 90 
 91 	/**
 92 	 * Gets the type of the content area that this {@link mmir.YieldDeclaration} object refers to (i.e. "areas" in the layout, e.g. bodyContents, dialogsContent).
 93 	 * 
 94 	 * @function getAreaType
 95 	 * @returns {Integer} Content area type (see {@link mmir.Layout}, e.g. ViewConstants.CONTENT_AREA_BODY)
 96 	 * @public
 97 	 */ 
 98 	YieldDeclaration.prototype.getAreaType = function(){
 99 	    return this.contentAreaType;
100 	};
101 
102 	/**
103 	 * Gets the start position (index) of a {@link mmir.YieldDeclaration} object.
104 	 * 
105 	 * @function getStart
106 	 * @returns {Integer} Start position of the Yield within the content (e.g. the bodyContent or the dialogsContent)
107 	 * @public
108 	 */ 
109 	YieldDeclaration.prototype.getStart = function(){
110 	    return this.start;
111 	};
112 
113 	/**
114 	 * Gets the end position (index) of a {@link mmir.YieldDeclaration} object.
115 	 * 
116 	 * @function getEnd
117 	 * @returns {Integer} End position of the Yield within the content (e.g. the bodyContent or the dialogsContent)
118 	 * @public
119 	 */ 
120 	YieldDeclaration.prototype.getEnd = function(){
121 	    return this.end;
122 	};
123 	
124 
125 	/**
126 	 * Get the value for property <code>name</code> with the proper type
127 	 * (as specified by <code>nameType</code>).
128 	 * This may be neccessary, if the nameType is not e.g. STRING but a VARIABLE, 
129 	 * in which case <code>name</code> does not reference the value itself, but the name 
130 	 * for the variable
131 	 * 
132 	 * This is a shortcut to the function 
133 	 * mobileDS.parser.ParsingResult.prototype.getValue
134 	 * 
135 	 * I.e. for YieldDeclration yield with a nameType of VARIABLE, to not use:
136 	 * <s><code>yield.getName()</code></s>
137 	 * 
138 	 * but
139 	 * 
140 	 * <code>yield.getValue(yield.getName(), yield.getNameType(), theRenderingData)</code>
141 	 * where theRenderingData is an object that contains a property from which the variable value can be retrieved, i.e.
142 	 * where <em>theRenderingData[yield.getName()]</em> contains the YieldDeclaration's name.
143 	 * 
144 	 */
145 	YieldDeclaration.prototype.getValue = ParsingResult.prototype.getValue; 
146 
147 	YieldDeclaration.prototype.stringify = function(){
148 		
149 		// "plain properties" list
150 		var propList = [
151 	   	     'name', 
152 	   	     'nameType',
153 	   	     'start',
154 	   	     'end',
155 	   	     'contentAreaType'
156 	   	];
157 
158 	   	//function for iterating over the property-list and generating JSON-like entries in the string-buffer
159 	   	var appendStringified = parser.appendStringified;
160 	   	
161 	   	var sb = ['require("storageUtils").restoreObject({ classConstructor: "yield"', ','];
162 	   	
163 	   	appendStringified(this, propList, sb);
164 	   	
165 	   	//if last element is a comma, remove it
166 	   	if(sb[sb.length - 1] === ','){
167 	   		sb.splice( sb.length - 1, 1);
168 	   	}
169 	   	
170 	   	sb.push(' })');
171 	   	return sb.join('');
172 	};
173 		
174 	return YieldDeclaration;
175 	
176 	/** #@- */
177 });
